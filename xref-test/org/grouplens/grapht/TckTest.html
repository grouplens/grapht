27  public class TckTest extends TestCase {

29      public static Test suite() throws InjectionException {
30          InjectorBuilder ib = InjectorBuilder.create()
31              .setDefaultCachePolicy(CachePolicy.NEW_INSTANCE)
32              .setProviderInjectionEnabled(true);
33          
34          ib.bind(Car.class).to(Convertible.class);
35          ib.bind(Seat.class).withQualifier(Drivers.class).to(DriversSeat.class);
36          ib.bind(Seat.class).to(Seat.class);
37          ib.bind(Tire.class).to(Tire.class);
38          ib.bind(Engine.class).to(V8Engine.class);
39          ib.bind(Tire.class).withQualifier(Names.named("spare")).to(SpareTire.class);
40          
41          Car car = ib.build().getInstance(Car.class);
42          
43          // Support for private methods, but no support for static methods,
44          // that doesn't make sense with grapht's design principles
45          return Tck.testsFor(car, false, true);
46      }
