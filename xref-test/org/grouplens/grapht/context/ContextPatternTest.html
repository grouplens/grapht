57      @Test
58      public void testSingleton() {
59          // test a single type pattern
60          // should match itself
61          ContextPattern initial = ContextPattern.empty().append(ContextElements.matchAny());
62          assertThat(initial.append(A.class).matches(makeContext(A.class)),
63                     notNullValue());
64          // should not match other type
65          assertThat(initial.append(A.class).matches(makeContext(B.class)),
66                     nullValue());
67          // should not match empty
68          assertThat(initial.append(A.class).matches(makeContext()),
69                     nullValue());
70          // should not match too long
71          assertThat(initial.append(A.class).matches(makeContext(A.class, B.class)),
72                     nullValue());
73          // either way
74          assertThat(initial.append(A.class).matches(makeContext(B.class, A.class)),
75                     nullValue());
76          // either way
77          assertThat(initial.append(A.class).matches(makeContext(B.class, A.class)),
78                     nullValue());
79          // or duplicated
80          assertThat(initial.append(A.class).matches(makeContext(A.class, A.class)),
81                     nullValue());
82      }

213     @Test
214     public void testOrderByCloseness() {
215         ContextPattern patA = ContextPattern.subsequence(A.class);
216         ContextPattern patB = ContextPattern.subsequence(B.class);
217         InjectionContext ctx1 = makeContext(A.class, B.class);
218         // B matches more closely than A
219         assertThat(patB.matches(ctx1), lessThan(patA.matches(ctx1)));
220         // A matches the same as itself
221         assertThat(patA.matches(ctx1),
222                    allOf(lessThanOrEqualTo(patA.matches(ctx1)),
223                          greaterThanOrEqualTo(patA.matches(ctx1))));
224     }

277     private InjectionContext makeContext(Class<?>... types) {
278         InjectionContext context = DependencySolver.initialContext();
279         for (Class<?> type: types) {
280             MockSatisfaction sat = new MockSatisfaction(type, new ArrayList<Desire>());
281             context = context.extend(sat, new MockInjectionPoint(type, null, false));
282         }
283         return context;
284     }
