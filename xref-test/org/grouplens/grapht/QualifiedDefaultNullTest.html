32  /**
33   * Tests for <a href="https://github.com/grouplens/grapht/issues/73">#73</a>.
34   *
35   * @author <a href="http://www.grouplens.org">GroupLens Research</a>
36   */
37  public class QualifiedDefaultNullTest {
38      @Test
39      public void testQualifiedDefaultNull() throws InjectionException {
40          InjectorBuilder bld = InjectorBuilder.create();
41          Injector inj = bld.build();
42  
43          // With no bindings, we should get a null inner class.
44          MainC main = inj.getInstance(MainC.class);
45          assertThat(main, notNullValue());
46          assertThat(main.dependency, nullValue());
47      }
48  
49      @Test
50      public void testQualifiedDefaultNullWithBinding() throws InjectionException {
51          InjectorBuilder bld = InjectorBuilder.create();
52          bld.bind(TestQ.class, DepC.class)
53             .to(DepC.class);
54          Injector inj = bld.build();
55  
56          // With a bindings, we should get an instance
57          MainC main = inj.getInstance(MainC.class);
58          assertThat(main, notNullValue());
59          assertThat(main.dependency,
60                     allOf(notNullValue(),
61                           instanceOf(DepC.class)));
62      }
63  
64      /**
65       * Qualifier with default of {@code null}.
66       */
67      @Qualifier
68      @Retention(RetentionPolicy.RUNTIME)
69      @Target(ElementType.PARAMETER)
70      @DefaultNull
71      @Documented
72      public static @interface TestQ {}
73  
74      /**
75       * Component with qualified dependency on concrete class.
76       */
77      public static class MainC {
78          private final DepC dependency;
79  
80          @Inject
81          public MainC(@Nullable @TestQ DepC dep) {
82              dependency = dep;
83          }
84      }
85  
86      public static class DepC {
87          @Inject
88          public DepC() {}
89      }
90  }
