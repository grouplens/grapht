47      /**
48       * Test bindings with a null, and a deeper context using a binding. To make things
49       * interesting, the thing tweaked is a concrete class.
50       */
51      @Test
52      public void testDeeperOverride() throws InjectionException {
53          build.within(Outer.class)
54               .bind(IPlug.class)
55               .to(PlugH.class);
56          build.within(Inner.class)
57               .bind(IPlug.class)
58               .to(PlugA.class);
59          Injector inj = build.build();
60  
61          Inner in = inj.getInstance(Inner.class);
62          assertThat(in, notNullValue());
63          assertThat(in.plug, instanceOf(PlugA.class));
64  
65          Outer out = inj.getInstance(Outer.class);
66          assertThat(out, notNullValue());
67          assertThat(out.plug, instanceOf(PlugH.class));
68          assertThat(out.inner, notNullValue());
69          assertThat(out.inner.plug, instanceOf(PlugA.class));
70      }

75      @Test

123     /**
124      * Test anchored root binding.
125      */
126     @Test
127     public void testAnchoredToRoot() throws InjectionException {
128         build.bind(IPlug.class)
129              .to(PlugA.class);
130         build.at(null)
131              .bind(IPlug.class)
132              .to(PlugH.class);
133         Injector inj = build.build();
134 
135         // Does directly requesting a plug get us the anchored binding?
136         assertThat(inj.getInstance(IPlug.class),
137                    instanceOf(PlugH.class));
138 
139         // Is the non-anchored binding used for dependencies?
140         Outer out = inj.getInstance(Outer.class);
141         assertThat(out.plug,
142                    instanceOf(PlugA.class));
143         assertThat(out.inner.plug,
144                    instanceOf(PlugA.class));
145         assertThat(out.plug, sameInstance(out.inner.plug));
146 
147         // quick check this again, make sure nothing changed
148         assertThat(inj.getInstance(IPlug.class),
149                    instanceOf(PlugH.class));
150     }

152     @Test
153     public void testPatternForPlug() throws InjectionException {
154         build.matching(ContextPattern.any()
155                                      .append(CInner.class)
156                                      .append(ContextElements.invertMatch(ContextElements.matchType(PlugW.class)),
157                                              Multiplicity.ZERO_OR_MORE))
158              .bind(Plug.class)
159              .to(PlugW.class);
160         Injector inj = build.build();
161         CInner c = inj.getInstance(CInner.class);
162         assertThat(c.plug, instanceOf(PlugW.class));
163         assert c.plug != null;
164         assertThat(((PlugW) c.plug).inner.getClass(),
165                    equalTo((Class) Plug.class));
166     }
