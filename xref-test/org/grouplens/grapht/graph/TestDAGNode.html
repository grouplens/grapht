144     @Test
145     public void testReplaceDualUseNode() {
146         DAGNode<String,String> foo, fooP, bar, blatz, graph;
147         foo = DAGNode.singleton("foo");
148         // this node should be replaced once, not twice
149         fooP = DAGNode.<String,String>newBuilder("foo'")
150                 .addEdge(foo, "-> foo")
151                 .build();
152         bar = DAGNode.<String,String>newBuilder("bar")
153                      .addEdge(fooP, "bar -> foo")
154                      .build();
155         blatz = DAGNode.<String,String>newBuilder("blatz")
156                        .addEdge(fooP, "blatz -> foo")
157                        .build();
158         graph = DAGNode.<String,String>newBuilder("graph")
159                 .addEdge(bar, "-> bar")
160                 .addEdge(blatz, "-> blatz")
161                 .build();
162 
163         DAGNode<String,String> foo2 = DAGNode.singleton("foo2");
164         HashMap<DAGNode<String,String>,DAGNode<String,String>> memory = Maps.newHashMap();
165         DAGNode<String,String> replaced = graph.replaceNode(foo, foo2, memory);
166         assertThat(replaced, not(sameInstance(graph)));
167         // if we have > 5 nodes, then fooP was duplicated!
168         assertThat(replaced.getReachableNodes(),
169                    hasSize(5));
170         assertThat(foo2, isIn(replaced.getReachableNodes()));
171         assertThat(foo, not(isIn(replaced.getReachableNodes())));
172         assertThat(fooP, not(isIn(replaced.getReachableNodes())));
173         assertThat(replaced.getSortedNodes().get(0),
174                    equalTo(foo2));
175         assertThat(replaced.getSortedNodes().get(1).getLabel(),
176                    equalTo("foo'"));
177     }

202     @Test
203     public void testTransformDeeperEdge() {
204         DAGNode<String,String> foo = DAGNode.singleton("foo");
205         DAGNode<String,String> bar = DAGNode.singleton("bar");
206         DAGNode<String,String> wombat = DAGNode.<String,String>newBuilder("wombat")
207                                                .addEdge(foo, "wombat")
208                                                .build();
209         DAGNode<String,String> graph = DAGNode.<String,String>newBuilder("graph")
210                                               .addEdge(wombat, "wumpus")
211                                               .addEdge(bar, "woozle")
212                                               .build();
213         DAGNode<String,String> g2 =
214                 graph.transformEdges(new Function<DAGEdge<String, String>, DAGEdge<String, String>>() {
215                     @Nullable
216                     @Override
217                     public DAGEdge<String, String> apply(@Nullable DAGEdge<String, String> input) {
218                         if (input != null && input.getLabel().equals("woozle")) {
219                             return DAGEdge.create(input.getHead(), input.getTail(), "hatrack");
220                         } else {
221                             return null;
222                         }
223                     }
224                 });
225         assertThat(g2, not(sameInstance(graph)));
226         assertThat(g2.getOutgoingEdge(bar, "hatrack"),
227                    notNullValue());
228         assertThat(g2.getOutgoingEdge(bar, "woozle"),
229                    nullValue());
230         assertThat(g2.getOutgoingEdge(wombat, "wumpus"),
231                    notNullValue());
232     }

306     @Test
307     public void testEdgeBFSByName() {
308         DAGNode<String,String> foo = DAGNode.singleton("foo");
309         DAGNode<String,String> baz = DAGNode.singleton("baz");
310         DAGNode<String,String> bar = DAGNode.<String,String>newBuilder("bar")

313                                             .build();

315         DAGEdge<String,String> edge = bar.findEdgeBFS(pred);
