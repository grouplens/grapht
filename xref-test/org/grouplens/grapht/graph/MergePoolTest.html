31  public class MergePoolTest {
32      MergePool<String,String> pool;
33  
34      @Before
35      public void createPool() {
36          pool = MergePool.create();
37      }
38  
39      @Test
40      public void testSingletonNode() {
41          DAGNode<String,String> node = DAGNode.singleton("foo");
42          DAGNode<String,String> merged = pool.merge(node);
43          // merging a singleton node should just return the node
44          assertThat(merged, sameInstance(node));
45      }
46  
47      @Test
48      public void testReuseSingleton() {
49          DAGNode<String,String> node = DAGNode.singleton("foo");
50          DAGNode<String,String> node2 = DAGNode.singleton("foo");
51          // simplify first so it's in the pool
52          pool.merge(node);
53          // and merge the second
54          DAGNode<String,String> merged = pool.merge(node2);
55          // the first node should have been reused
56          assertThat(merged, sameInstance(node));
57      }

59      @Test
60      public void testMergeDescendants() {
61          DAGNode<String,String> node = DAGNode.singleton("foo");
62          DAGNode<String,String> node2 = DAGNode.singleton("foo");
63          DAGNode<String,String> root =
64                  DAGNode.<String,String>newBuilder("root")
65                         .addEdge(node, "hello")
66                         .addEdge(node2, "goodbye")
67                         .build();
68  
69          DAGNode<String,String> merged = pool.merge(root);
70  
71          // now, node and node2 should be merged
72          // new graph (since they're merged)
73          assertThat(merged, not(sameInstance(root)));
74          // only 2 nodes (root has 2 edges to same node)
75          assertThat(merged.getReachableNodes(), hasSize(2));
76          // two edges have same target
77          List<DAGEdge<String,String>> nbrs =
78                  Lists.newArrayList(merged.getOutgoingEdges());
79          // both edges have the same instance
80          assertThat(nbrs.get(1).getTail(),
81                     sameInstance(nbrs.get(0).getTail()));
82          // it's one of the nodes we gave it
83          assertThat(nbrs.get(0).getTail(),
84                     anyOf(sameInstance(node),
85                           sameInstance(node2)));
86  
87          // and a new singleton should be merged
88          assertThat(pool.merge(DAGNode.<String, String>singleton("foo")),
89                     anyOf(sameInstance(node),
90                           sameInstance(node2)));
91  
92          // re-merging our old root should give us the same node
93          assertThat(pool.merge(root), sameInstance(merged));
94      }

96      @Test
97      public void testMergeWithChildren() {
98          DAGNode<String,String> node = DAGNode.singleton("foo");
99          DAGNode<String,String> p1 =
100                 DAGNode.<String,String>newBuilder("child")
101                        .addEdge(node, "k1")
102                        .build();
103         DAGNode<String,String> node2 = DAGNode.singleton("foo");
104         // second parent node. it will have different label to test that labels are ignored
105         DAGNode<String,String> p2 =
106                 DAGNode.<String,String>newBuilder("child")
107                        .addEdge(node, "k2")
108                        .build();
109         DAGNode<String,String> root =
110                 DAGNode.<String,String>newBuilder("root")
111                        .addEdge(p1, "hello")
112                        .addEdge(p2, "fish")
113                        .build();
114 
115         DAGNode<String,String> merged = pool.merge(root);
116 
117         // now, p1 and p2 should be merged
118         // new graph (since they're merged)
119         assertThat(merged, not(sameInstance(root)));
120         // only 3 nodes (root has 2 edges to now-merged nodes)
121         assertThat(merged.getReachableNodes(), hasSize(3));
122         // it has one of the leaves
123         assertThat(merged.getReachableNodes(),
124                    (Matcher) anyOf(hasItem(sameInstance(node)),
125                                    hasItem(sameInstance(node2))));
126         // two edges have same target
127         List<DAGEdge<String,String>> nbrs =
128                 Lists.newArrayList(merged.getOutgoingEdges());
129         // both edges have the same instance
130         assertThat(nbrs.get(1).getTail(),
131                    sameInstance(nbrs.get(0).getTail()));
132 
133         // and a new singleton should be merged
134         assertThat(pool.merge(DAGNode.<String, String>singleton("foo")),
135                    anyOf(sameInstance(node),
136                          sameInstance(node2)));
137 
138         // re-merging a child should give us a previously-merged node
139         assertThat(pool.merge(p2), isIn(merged.getReachableNodes()));
140         assertThat(pool.merge(p1), isIn(merged.getReachableNodes()));
141     }
