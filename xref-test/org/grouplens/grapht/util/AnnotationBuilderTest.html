29      @Test
30      public void testArrayCloned() {
31          // verify that array values are clones of the original
32          // to prevent contamination, and that returned values are also
33          // clones to prevent post-contamination
34          
35          double[] original = new double[] { 4.0 };
36          A3 built = new AnnotationBuilder<A3>(A3.class).set("otherValue", original).build();
37          double[] fromAnnot = built.otherValue();
38  
39          original[0] = 5.0;
40          Assert.assertEquals(4.0, fromAnnot[0], 0.00001);
41          
42          fromAnnot[0] = 6.0;
43          Assert.assertEquals(4.0, built.otherValue()[0], 0.00001);
44      }

74      @Test
75      public void testSetValues() {
76          // verify that all setters set the expected values
77          Types t = new AnnotationBuilder<Types>(Types.class)
78              .set("v1", (byte) 1)
79              .set("v2", (short) 2)
80              .set("v3", 3)
81              .set("v4", 4L)
82              .set("v5", '5')
83              .set("v6", 6f)
84              .set("v7", 7.0)
85              .set("v8", "8")
86              .set("v9", new byte[] { 9 })
87              .set("v10", new short[] { 10 })
88              .set("v11", new int[] { 11 })
89              .set("v12", new long[] { 12 })
90              .set("v13", new char[] { 't' })
91              .set("v14", new float[] { 14f })
92              .set("v15", new double[] { 15.0 })
93              .set("v16", new String[] { "16" })
94              .set("v17", new AnnotationBuilder<A2>(A2.class).set("value", 17).build())
95              .set("v18", new A2[] { new AnnotationBuilder<A2>(A2.class).set("value", 18).build() })
96              .set("v19", true)
97              .set("v20", new boolean[] { true, false })
98              .build();
99          
100         Assert.assertEquals((byte) 1, t.v1());
101         Assert.assertEquals((short) 2, t.v2());
102         Assert.assertEquals(3, t.v3());
103         Assert.assertEquals(4L, t.v4());
104         Assert.assertEquals('5', t.v5());
105         Assert.assertEquals(6f, t.v6(), 0.00001f);
106         Assert.assertEquals(7.0, t.v7(), 0.00001);
107         Assert.assertEquals("8", t.v8());
108         
109         Assert.assertEquals(1, t.v9().length);
110         Assert.assertEquals((byte) 9, t.v9()[0]);
111         
112         Assert.assertEquals(1, t.v10().length);
113         Assert.assertEquals((short) 10, t.v10()[0]);
114         
115         Assert.assertEquals(1, t.v11().length);
116         Assert.assertEquals(11, t.v11()[0]);
117         
118         Assert.assertEquals(1, t.v12().length);
119         Assert.assertEquals(12L, t.v12()[0]);
120         
121         Assert.assertEquals(1, t.v13().length);
122         Assert.assertEquals('t', t.v13()[0]);
123         
124         Assert.assertEquals(1, t.v14().length);
125         Assert.assertEquals(14f, t.v14()[0], 0.00001f);
126         
127         Assert.assertEquals(1, t.v15().length);
128         Assert.assertEquals(15.0, t.v15()[0], 0.00001);
129         
130         Assert.assertEquals(1, t.v16().length);
131         Assert.assertEquals("16", t.v16()[0]);
132         
133         Assert.assertEquals(17, t.v17().value());
134         
135         Assert.assertEquals(1, t.v18().length);
136         Assert.assertEquals(18, t.v18()[0].value());
137         
138         Assert.assertTrue(t.v19());
139         
140         Assert.assertEquals(2, t.v20().length);
141         Assert.assertTrue(t.v20()[0]);
142         Assert.assertFalse(t.v20()[1]);
143     }
