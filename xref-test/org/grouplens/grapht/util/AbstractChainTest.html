33  public class AbstractChainTest {
34      static class TestChain extends AbstractChain<String> {
35          TestChain(AbstractChain<String> prev, String tv) {
36              super(prev, tv);
37          }
38          public TestChain extend(String tok) {
39              return new TestChain(this, tok);
40          }
41      }
42  
43      static TestChain singleton(String tok) {
44          return new TestChain(null, tok);
45      }
46      static TestChain chain(String t1, String... toks) {
47          TestChain chain = singleton(t1);
48          for (String tok: toks) {
49              chain = chain.extend(tok);
50          }
51          return chain;
52      }

54      @Test
55      public void testSingletonBasicProps() throws Exception {
56          TestChain chain = singleton("foo");
57          assertThat(chain.size(), equalTo(1));
58          assertThat(chain.isEmpty(), equalTo(false));
59          assertThat(chain.contains("foo"), equalTo(true));
60          assertThat(chain, contains("foo"));
61      }

63      @Test
64      public void testSingletonsEqual() {
65          TestChain foo1 = singleton("foo");
66          TestChain foo2 = singleton("foo");
67          TestChain bar = singleton("bar");
68  
69          assertThat(foo1.equals(foo1), equalTo(true));
70          assertThat(foo1.equals(foo2), equalTo(true));
71          assertThat(foo1.equals(bar), equalTo(false));
72      }

74      @Test
75      public void testSingletonGet() {
76          TestChain foo = singleton("foo");
77  
78          assertThat(foo.get(0), equalTo("foo"));
79      }

81      @Test
82      public void testExtend() {
83          TestChain foo = singleton("foo");
84          TestChain foobar = foo.extend("bar");
85  
86          assertThat(foobar.size(), equalTo(2));
87  
88          Iterator<String> iter = foobar.iterator();
89          assertThat(iter.next(), equalTo("foo"));
90          assertThat(iter.next(), equalTo("bar"));
91          assertThat(iter.hasNext(), equalTo(false));
92  
93          assertThat(foobar.get(1), equalTo("bar"));
94          assertThat(foobar.get(0), equalTo("foo"));
95      }
96  
97      @Test
98      public void testReverseIteration() {
99          TestChain chain = chain("foobie", "bletch", "forever");
100         assertThat(chain, hasSize(3));
101         assertThat(chain.reverse(), contains("forever", "bletch", "foobie"));
102 
103         Iterator<String> iter = chain.reverseIterator();
104         iter.next();
105         iter.next();
106         iter.next();
107         try {
108             iter.next();
109             fail("must throw NoSuchElement");
110         } catch (Throwable th) {
111             assertThat(th, instanceOf(NoSuchElementException.class));
112         }
113     }

115     @Test
116     public void testChainsEqual() {
117         TestChain base = chain("foo", "bar", "blatz");
118         TestChain copy = chain("foo", "bar", "blatz");
119         TestChain chend = chain("foo", "bar", "blam");
120         TestChain chstart = chain("f00", "bar", "blatz");
121 
122         assertThat(base.equals(base), equalTo(true));
123         assertThat(base.equals(copy), equalTo(true));
