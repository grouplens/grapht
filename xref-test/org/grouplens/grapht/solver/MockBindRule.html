19  package org.grouplens.grapht.solver;
20  
21  import org.grouplens.grapht.CachePolicy;
22  import org.grouplens.grapht.reflect.Desire;
23  import org.grouplens.grapht.reflect.MockQualifierMatcher;
24  import org.grouplens.grapht.reflect.QualifierMatcher;
25  
26  import java.util.EnumSet;
27  import java.util.HashMap;
28  import java.util.Map;
29  
30  /**
31   * MockBindRule is a simple implementation of BindRule where the matching
32   * bindings are represented as a map from input Desires to output Desires, via
33   * {@link #addMapping(Desire, Desire)}.
34   * 
35   * @author <a href="http://grouplens.org">GroupLens Research</a>
36   */
37  public class MockBindRule implements BindRule {
38      private final Map<Desire, Desire> bindings;
39      private boolean terminate;
40      private CachePolicy policy;
41      
42      public MockBindRule() {
43          bindings = new HashMap<Desire, Desire>();
44          policy = CachePolicy.NO_PREFERENCE;
45      }
46      
47      public MockBindRule(Desire in, Desire out) {
48          this();
49          addMapping(in, out);
50      }
51      
52      public MockBindRule setCachePolicy(CachePolicy policy) {
53          this.policy = policy;
54          return this;
55      }
56      
57      public MockBindRule setTerminatesChain(boolean terminate) {
58          this.terminate = terminate;
59          return this;
60      }
61      
62      public MockBindRule addMapping(Desire in, Desire out) {
63          bindings.put(in, out);
64          return this;
65      }
66      
67      @Override
68      public CachePolicy getCachePolicy() {
69          return policy;
70      }
71      
72      @Override
73      public boolean matches(Desire desire) {
74          return bindings.containsKey(desire);
75      }
76  
77      @Override
78      public Desire apply(Desire desire) {
79          return bindings.get(desire);
80      }
81  
82      @Override
83      public EnumSet<BindingFlag> getFlags() {
84          if (terminate) {
85              return EnumSet.of(BindingFlag.TERMINAL);
86          } else {
87              return BindingFlag.emptySet();
88          }
89      }
90  
91      @Override
92      public boolean isTerminal() {
93          return terminate;
94      }
95      
96      public QualifierMatcher getQualifier() {
97          return MockQualifierMatcher.any();
98      }
99  
100     @Override
101     public BindRuleBuilder newCopyBuilder() {
102         throw new UnsupportedOperationException("cannot configure mock bind rules");
103     }
104 
105     public int compareTo(BindRule other) {
106         return getQualifier().compareTo(((MockBindRule) other).getQualifier());
107     }
108 
109     @Override
110     public String toString() {
111         return getClass() + "@" + Integer.toHexString(System.identityHashCode(this));
112     }
113 }
