50      private DependencySolver createSolver(ListMultimap<ContextMatcher, BindRule> rules) {
51          return DependencySolver.newBuilder()
52                  .addBindingFunction(new RuleBasedBindingFunction(rules))
53                  .setDefaultPolicy(CachePolicy.NO_PREFERENCE)
54                  .setMaxDepth(100)
55                  .build();
56      }

58      // bypass synthetic root and return node that resolves the desire 
59      private DAGNode<Component, Dependency> getRoot(DependencySolver r, Desire d) {
60          return r.getGraph().getOutgoingEdgeWithLabel(Dependency.hasInitialDesire(d)).getTail();
61      }

63      @Test
64      public void testCachePolicySuccess() throws Exception {
65          // Test that satisfactions formed with different cache policies 
66          // correctly restrict merging of nodes
67          Satisfaction sa = new MockSatisfaction(A.class, new ArrayList<Desire>());
68          Desire da = new MockDesire();
69          Satisfaction sb = new MockSatisfaction(B.class, Arrays.asList(da));
70          
71          Desire ra = new MockDesire(sa);
72          Desire rb = new MockDesire(sb);
73          
74          ImmutableListMultimap.Builder<ContextMatcher, BindRule> bindings = ImmutableListMultimap.builder();
75          final Class<?> type = B.class;
76          final MockQualifierMatcher qualifier = MockQualifierMatcher.any();
77          bindings.put(ContextPattern.subsequence(ContextElements.matchType(type, qualifier)),
78                       new MockBindRule(da, ra).setCachePolicy(CachePolicy.MEMOIZE));
79          bindings.put(ContextPattern.any(),
80                       new MockBindRule(da, ra).setCachePolicy(CachePolicy.NEW_INSTANCE));
81          
82          DependencySolver r = createSolver(bindings.build());
83          r.resolve(rb);
84          r.resolve(da); // should create a single extra sa node with different cache policy
85  
86          assertThat(r.getGraph().getReachableNodes(),
87                     hasSize(3 + 1));
88  
89          DAGNode<Component, Dependency> bnode = getRoot(r, rb);
90          Assert.assertEquals(CachePolicy.NO_PREFERENCE, bnode.getLabel().getCachePolicy());
91          
92          DAGNode<Component, Dependency> adepnode = getNode(bnode, sa, da);
93          Assert.assertNotNull(adepnode);
94          Assert.assertEquals(CachePolicy.MEMOIZE, adepnode.getLabel().getCachePolicy());
95          
96          DAGNode<Component, Dependency> anode = getRoot(r, da);
97          Assert.assertNotSame(adepnode, anode);
98          Assert.assertEquals(CachePolicy.NEW_INSTANCE, anode.getLabel().getCachePolicy());
99      }
