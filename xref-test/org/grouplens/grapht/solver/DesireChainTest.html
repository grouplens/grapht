25  import java.io.FileInputStream;
26  import java.io.InputStream;

29  import static org.hamcrest.Matchers.*;
30  import static org.junit.Assert.assertThat;

32  /**
33   * @author <a href="http://www.grouplens.org">GroupLens Research</a>
34   */
35  public class DesireChainTest {
36      @Test
37      public void testSingleDesire() {
38          Desire desire = new MockDesire(String.class, null, null);
39          DesireChain chain = DesireChain.singleton(desire);
40          assertThat(chain.getCurrentDesire(),
41                     equalTo(desire));
42          assertThat(chain.getInitialDesire(),
43                     equalTo(desire));
44          assertThat(chain, contains(desire));
45          assertThat(chain.getPreviousDesires(), hasSize(0));
46      }
47  
48      @Test
49      public void testMultipleDesires() {
50          Desire d1 = new MockDesire(InputStream.class, null, null);
51          Desire d2 = new MockDesire(FileInputStream.class, null, null);
52          DesireChain chain = DesireChain.singleton(d1).extend(d2);
53          assertThat(chain.getCurrentDesire(),
54                     equalTo(d2));
55          assertThat(chain.getInitialDesire(),
56                     equalTo(d1));
57          assertThat(chain.getPreviousDesires(),
58                     equalTo((List<Desire>) DesireChain.singleton(d1)));
59          assertThat(chain, contains(d1, d2));
60      }
61  }
