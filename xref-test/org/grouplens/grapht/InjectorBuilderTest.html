182     @Test
183     public void testInjectorBuilderWithAnnotatedBindings() throws Exception {
184         // Test that injector building works as expected
185         TypeA a1 = new TypeA();
186         InterfaceA a2 = new TypeA();
187         TypeB b1 = new TypeB();
188         InterfaceB b2 = new TypeB();
189         
190         InjectorBuilder b = InjectorBuilder.create();
191         b.bind(TypeA.class).to(a1);
192         b.bind(InterfaceA.class).withQualifier(RoleA.class).to(a2);
193         b.bind(TypeB.class).to(b1);
194         b.bind(InterfaceB.class).withQualifier(RoleD.class).to(b2);
195         Injector i = b.build();
196         
197         TypeC c = i.getInstance(TypeC.class);
198         Assert.assertEquals(5, c.getIntValue());
199         Assert.assertSame(a1, c.getTypeA());
200         Assert.assertSame(a2, c.getInterfaceA());
201         Assert.assertSame(b1, c.getTypeB());
202         Assert.assertSame(b2, c.getInterfaceB());
203         
204         // now assert that it memoizes instances and merges graphs properly
205         Assert.assertSame(a1, i.getInstance(TypeA.class));
206         Assert.assertSame(a2, i.getInstance(new AnnotationBuilder<RoleA>(RoleA.class).build(), InterfaceA.class));
207         Assert.assertSame(b1, i.getInstance(TypeB.class));
208         Assert.assertSame(b2, i.getInstance(new AnnotationBuilder<RoleD>(RoleD.class).build(), InterfaceB.class));
209     }

212     public void testInjectorBuilderWithNamedBindings() throws Exception {
213         InjectorBuilder b = InjectorBuilder.create();
214         b.bind(String.class).withQualifier(new AnnotationBuilder<Named>(Named.class).set("value", "unused").build()).to("shouldn't see this"); // extra binding to make sure it's skipped
215         b.bind(String.class).withQualifier(new AnnotationBuilder<Named>(Named.class).set("value", "test1").build()).to("hello world");
216         Injector i = b.build();
217         
218         NamedType c = i.getInstance(NamedType.class);
219         Assert.assertEquals("hello world", c.getNamedString());
220         Assert.assertEquals("hello world", i.getInstance(new AnnotationBuilder<Named>(Named.class).set("value", "test1").build(), String.class));
