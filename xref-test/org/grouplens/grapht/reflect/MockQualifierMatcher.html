19  package org.grouplens.grapht.reflect;
20  
21  import java.lang.annotation.Annotation;
22  import java.util.HashSet;
23  import java.util.Set;
24  
25  public class MockQualifierMatcher implements QualifierMatcher {
26      private static final long serialVersionUID = 1L;
27  
28      private Set<Annotation> qualifiers;
29      
30      private MockQualifierMatcher() {
31          qualifiers = null;
32      }
33  
34      public static MockQualifierMatcher any() {
35          return new MockQualifierMatcher();
36      }
37      
38      public static MockQualifierMatcher none() {
39          return match((Annotation) null);
40      }
41  
42      @Override
43      public int getPriority() {
44          return 0;
45      }
46  
47      public static MockQualifierMatcher match(Annotation... qualifiers) {
48          MockQualifierMatcher q = new MockQualifierMatcher();
49          q.qualifiers = new HashSet<Annotation>();
50          for (Annotation m: qualifiers) {
51              q.qualifiers.add(m);
52          }
53          return q;
54      }
55      
56      @Override
57      public int compareTo(QualifierMatcher o) {
58          MockQualifierMatcher mq = (MockQualifierMatcher) o;
59          
60          if (qualifiers == null) {
61              if (mq.qualifiers == null) {
62                  return 0;
63              } else {
64                  return 1;
65              }
66          } else {
67              if (mq.qualifiers == null) {
68                  return -1;
69              } else {
70                  return 0;
71              }
72          }
73      }
74  
75      @Override
76      @Deprecated
77      public boolean matches(Annotation q) {
78          return apply(q);
79      }
80  
81      @Override
82      public boolean apply(Annotation q) {
83          return qualifiers == null || qualifiers.contains(q);
84      }
85  }
