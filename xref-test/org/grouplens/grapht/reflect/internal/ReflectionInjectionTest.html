44      @Test
45      public void testProviderCycleInjection() throws Exception {
46          InjectorBuilder b = InjectorBuilder.create().setProviderInjectionEnabled(true);
47          Injector i = b.build();
48          
49          i.getInstance(CycleA.class);
50          DAGNode<Component,Dependency> root = ((DefaultInjector) i).getSolver().getGraph();
51  
52          assertThat(root.getReachableNodes(), hasSize(3 + 1));
53  
54          assertThat(root.getOutgoingEdges(), hasSize(1));
55          DAGNode<Component, Dependency> anode = root.getOutgoingEdges().iterator().next().getTail();
56          Assert.assertEquals(CycleA.class, anode.getLabel().getSatisfaction().getErasedType());
57          
58          Assert.assertEquals(1, anode.getOutgoingEdges().size());
59          DAGNode<Component, Dependency> bnode = anode.getOutgoingEdges().iterator().next().getTail();
60          Assert.assertEquals(CycleB.class, bnode.getLabel().getSatisfaction().getErasedType());
61          
62          Assert.assertEquals(1, bnode.getOutgoingEdges().size());
63          DAGNode<Component, Dependency> pnode = bnode.getOutgoingEdges().iterator().next().getTail();
64          Assert.assertEquals(Provider.class, pnode.getLabel().getSatisfaction().getErasedType());
65  
66          // no outgoing edges...
67          Assert.assertEquals(0, pnode.getOutgoingEdges().size());
68          // but a back edge
69          SetMultimap<DAGNode<Component,Dependency>,DAGEdge<Component, Dependency>> backEdges = ((DefaultInjector) i).getSolver().getBackEdges();
70          assertThat(backEdges.entries(), hasSize(1));
71          DAGEdge<Component, Dependency> edge = backEdges.values().iterator().next();
72          Assert.assertSame(anode, edge.getTail());
73      }

75      @Test
76      public void testTypeCInjectionWithDefaults() throws Exception {
77          // Test that TypeC can be resolved successfully without any bind rules.
78          // All of TypeC's dependencies have defaults or are satisfiable.
79          Desire rootDesire = Desires.create(null, TypeC.class, false);
80          DefaultInjector r = new DefaultInjector(DefaultDesireBindingFunction.create());
81          
82          TypeC instance = r.getInstance(TypeC.class);
83          Assert.assertEquals(5, instance.getIntValue());
84          Assert.assertNotNull(instance.getInterfaceA());
85          Assert.assertTrue(instance.getInterfaceA() instanceof TypeB); // ProviderA actually creates TypeB's
86          Assert.assertSame(instance.getInterfaceA(), instance.getTypeA());
87          Assert.assertNotNull(instance.getInterfaceB());
88          Assert.assertTrue(instance.getInterfaceB() instanceof TypeB);
89          Assert.assertSame(instance.getInterfaceB(), instance.getTypeB());
90          
91          // also verify memoization
92          Assert.assertSame(instance, r.getInstance(TypeC.class));
93  
94          DAGNode<Component, Dependency> resolvedRoot =
95                  r.getSolver().getGraph().getOutgoingEdgeWithLabel(Dependency.hasInitialDesire(rootDesire)).getTail();
96          assertThat(resolvedRoot.getOutgoingEdges(),
97                     hasSize(5));
98  
99          Map<InjectionPoint, DAGNode<Component, Dependency>> deps = Maps.newHashMap();
100         for (DAGEdge<Component, Dependency> e: resolvedRoot.getOutgoingEdges()) {
101             ReflectionDesire d = (ReflectionDesire) e.getLabel().getInitialDesire();
102             
103             if (d.getInjectionPoint().equals(TypeC.CONSTRUCTOR)) {
104                 // CycleA ParameterA defaults to 5
105                 Assert.assertFalse(deps.containsKey(TypeC.CONSTRUCTOR));
106                 Assert.assertTrue(e.getTail().getLabel().getSatisfaction() instanceof InstanceSatisfaction);
107                 Assert.assertEquals(5, ((InstanceSatisfaction) e.getTail().getLabel().getSatisfaction()).getInstance());
108                 deps.put(TypeC.CONSTRUCTOR, e.getTail());
109             } else if (d.getInjectionPoint().equals(TypeC.INTERFACE_A)) {
110                 // An InterfaceA is implemented by TypeA, which is then provided by Provider CycleA
111                 Assert.assertFalse(deps.containsKey(TypeC.INTERFACE_A));
112                 Assert.assertTrue(e.getTail().getLabel().getSatisfaction() instanceof ProviderClassSatisfaction);
113                 Assert.assertEquals(ProviderA.class, ((ProviderClassSatisfaction) e.getTail().getLabel().getSatisfaction()).getProviderType());
114                 deps.put(TypeC.INTERFACE_A, e.getTail());
115             } else if (d.getInjectionPoint().equals(TypeC.TYPE_A)) {
116                 // CycleA TypeA is provided by a ProviderA
117                 Assert.assertFalse(deps.containsKey(TypeC.TYPE_A));
118                 Assert.assertTrue(e.getTail().getLabel().getSatisfaction() instanceof ProviderClassSatisfaction);
119                 Assert.assertEquals(ProviderA.class, ((ProviderClassSatisfaction) e.getTail().getLabel().getSatisfaction()).getProviderType());
120                 deps.put(TypeC.TYPE_A, e.getTail());
121             } else if (d.getInjectionPoint().equals(TypeC.INTERFACE_B)) {
122                 // RoleE inherits RoleD and that defaults to TypeB
123                 Assert.assertFalse(deps.containsKey(TypeC.INTERFACE_B));
124                 Assert.assertTrue(e.getTail().getLabel().getSatisfaction() instanceof ClassSatisfaction);
125                 Assert.assertEquals(TypeB.class, e.getTail().getLabel().getSatisfaction().getErasedType());
126                 deps.put(TypeC.INTERFACE_B, e.getTail());
127             } else if (d.getInjectionPoint().equals(TypeC.TYPE_B)) {
128                 // TypeB is satisfiable on its own
129                 Assert.assertFalse(deps.containsKey(TypeC.TYPE_B));
130                 Assert.assertTrue(e.getTail().getLabel().getSatisfaction() instanceof ClassSatisfaction);
131                 Assert.assertEquals(TypeB.class, e.getTail().getLabel().getSatisfaction().getErasedType());
132                 deps.put(TypeC.TYPE_B, e.getTail());
133             } else {
134                 Assert.fail();
135             }
136         }
137         
138         // verify that all injection points were tested
139         Assert.assertTrue(deps.containsKey(TypeC.CONSTRUCTOR));
140         Assert.assertTrue(deps.containsKey(TypeC.TYPE_A));
141         Assert.assertTrue(deps.containsKey(TypeC.INTERFACE_A));
142         Assert.assertTrue(deps.containsKey(TypeC.TYPE_B));
143         Assert.assertTrue(deps.containsKey(TypeC.INTERFACE_B));
144         
145         // make sure that nodes are shared where appropriate
146         Assert.assertSame(deps.get(TypeC.INTERFACE_A), deps.get(TypeC.TYPE_A));
147         Assert.assertSame(deps.get(TypeC.INTERFACE_B), deps.get(TypeC.TYPE_B));
148     }

150     @Test
151     public void testTypeCInjectionWithBindings() throws Exception {
152         // Test that TypeC can be injected correctly using bind rules, although
153         // the bind rule configuration does not need to be very complicated, since
154         // the resolver and bind rules are already tested.
155         Desire rootDesire = Desires.create(null, TypeC.class, false);
156         
157         TypeA a = new TypeA();
158         TypeB b = new TypeB();
159         
160         BindingFunctionBuilder bindRules = new BindingFunctionBuilder(false);
161         bindRules.getRootContext().bind(Integer.class).withQualifier(ParameterA.class).to(10);
162         bindRules.getRootContext().bind(InterfaceA.class).withQualifier(RoleA.class).to(PrimeA.class);
163         bindRules.getRootContext().bind(InterfaceB.class).withQualifier(RoleD.class).to(PrimeB.class);
164         bindRules.getRootContext().bind(TypeA.class).to(a);
165         bindRules.getRootContext().bind(TypeB.class).to(b);
166         
167         DefaultInjector r = new DefaultInjector(bindRules.build(RuleSet.EXPLICIT), DefaultDesireBindingFunction.create());
168 
169         TypeC instance = r.getInstance(TypeC.class);
170         Assert.assertEquals(10, instance.getIntValue());
171         Assert.assertNotNull(instance.getInterfaceA());
172         Assert.assertTrue(instance.getInterfaceA() instanceof PrimeA);
173         Assert.assertSame(a, instance.getTypeA());
174         Assert.assertNotNull(instance.getInterfaceB());
175         Assert.assertTrue(instance.getInterfaceB() instanceof PrimeB);
176         Assert.assertSame(b, instance.getTypeB());
177         
178         // also verify memoization
179         Assert.assertSame(instance, r.getInstance(TypeC.class));
180 
181         DAGNode<Component, Dependency> resolvedRoot =
182                 r.getSolver().getGraph().getOutgoingEdgeWithLabel(Dependency.hasInitialDesire(rootDesire)).getTail();
183         assertThat(resolvedRoot.getOutgoingEdges(),
184                    hasSize(5));
185         
186         Map<InjectionPoint, DAGNode<Component, Dependency>> deps = Maps.newHashMap();
187         for (DAGEdge<Component, Dependency> e: resolvedRoot.getOutgoingEdges()) {
188             ReflectionDesire d = (ReflectionDesire) e.getLabel().getInitialDesire();
189             
190             if (d.getInjectionPoint().equals(TypeC.CONSTRUCTOR)) {
191                 // ParameterA was set to 10
192                 Assert.assertFalse(deps.containsKey(TypeC.CONSTRUCTOR));
193                 Assert.assertTrue(e.getTail().getLabel().getSatisfaction() instanceof InstanceSatisfaction);
194                 Assert.assertEquals(10, ((InstanceSatisfaction) e.getTail().getLabel().getSatisfaction()).getInstance());
195                 deps.put(TypeC.CONSTRUCTOR, e.getTail());
196             } else if (d.getInjectionPoint().equals(TypeC.INTERFACE_A)) {
197                 // An InterfaceA has been bound to PrimeA
198                 Assert.assertFalse(deps.containsKey(TypeC.INTERFACE_A));
199                 Assert.assertTrue(e.getTail().getLabel().getSatisfaction() instanceof ClassSatisfaction);
200                 Assert.assertEquals(PrimeA.class, e.getTail().getLabel().getSatisfaction().getErasedType());
201                 deps.put(TypeC.INTERFACE_A, e.getTail());
202             } else if (d.getInjectionPoint().equals(TypeC.TYPE_A)) {
203                 // CycleA TypeA has been bound to an instance
204                 Assert.assertFalse(deps.containsKey(TypeC.TYPE_A));
205                 Assert.assertTrue(e.getTail().getLabel().getSatisfaction() instanceof InstanceSatisfaction);
206                 Assert.assertSame(a, ((InstanceSatisfaction) e.getTail().getLabel().getSatisfaction()).getInstance());
207                 deps.put(TypeC.TYPE_A, e.getTail());
208             } else if (d.getInjectionPoint().equals(TypeC.INTERFACE_B)) {
209                 // RoleE has been bound to PrimeB
210                 Assert.assertFalse(deps.containsKey(TypeC.INTERFACE_B));
211                 Assert.assertTrue(e.getTail().getLabel().getSatisfaction() instanceof ClassSatisfaction);
212                 Assert.assertEquals(PrimeB.class, e.getTail().getLabel().getSatisfaction().getErasedType());
213                 deps.put(TypeC.INTERFACE_B, e.getTail());
214             } else if (d.getInjectionPoint().equals(TypeC.TYPE_B)) {
215                 // TypeB has been bound to an instance
216                 Assert.assertFalse(deps.containsKey(TypeC.TYPE_B));
217                 Assert.assertTrue(e.getTail().getLabel().getSatisfaction() instanceof InstanceSatisfaction);
218                 Assert.assertSame(b, ((InstanceSatisfaction) e.getTail().getLabel().getSatisfaction()).getInstance());
219                 deps.put(TypeC.TYPE_B, e.getTail());
220             } else {
221                 Assert.fail();
222             }
223         }
224         
225         // verify that all injection points were tested
226         Assert.assertTrue(deps.containsKey(TypeC.CONSTRUCTOR));
227         Assert.assertTrue(deps.containsKey(TypeC.TYPE_A));
228         Assert.assertTrue(deps.containsKey(TypeC.INTERFACE_A));
229         Assert.assertTrue(deps.containsKey(TypeC.TYPE_B));
230         Assert.assertTrue(deps.containsKey(TypeC.INTERFACE_B));
231     }
