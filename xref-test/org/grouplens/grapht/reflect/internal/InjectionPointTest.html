64      @Test
65      public void testConstructorParameterInjectionPoint() throws Exception {
66          // created expected injection points
67          Constructor<CtorType> ctor = CtorType.class.getConstructor(Object.class, String.class);
68          ConstructorParameterInjectionPoint p1 = new ConstructorParameterInjectionPoint(ctor, 0);
69          ConstructorParameterInjectionPoint p2 = new ConstructorParameterInjectionPoint(ctor, 1);
70          
71          Set<InjectionPoint> expected = new HashSet<InjectionPoint>();
72          expected.add(p1);
73          expected.add(p2);
74          
75          // verify that the qualifiers and types are identified properly
76          Assert.assertThat(p1.getQualifier(), instanceOf(RoleA.class));
77          Assert.assertThat(p2.getQualifier(), instanceOf(RoleB.class));
78          Assert.assertThat(p1.getAttribute(Transient.class), notNullValue());
79          Assert.assertThat(p1.getAttributes(), contains(instanceOf(Transient.class)));
80          Assert.assertThat(p2.getAttributes(), hasSize(0));
81          Assert.assertThat(p2.getAttribute(Transient.class), nullValue());
82  
83          Assert.assertEquals(Object.class, p1.getType());
84          Assert.assertEquals(String.class, p2.getType());
85          
86          // verify nullability and transience
87          Assert.assertFalse(p1.isNullable());
88          Assert.assertTrue(p2.isNullable());
89          
90          Assert.assertEquals(expected, getInjectionPoints(CtorType.class));
91      }

218     private Set<InjectionPoint> getInjectionPoints(Class<?> types) {
219         List<Desire> desires = ReflectionDesire.getDesires(types);
220         Set<InjectionPoint> points = new HashSet<InjectionPoint>();
221         for (Desire rd: desires) {
222             points.add(rd.getInjectionPoint());
223         }
224         return points;
225     }
