28  import static org.junit.Assert.assertThat;

30  public class FieldInjectionPointTest {
31      // a field to test serialization
32      String foo;
33      @Nullable
34      String nullable;
35  
36      @Test
37      public void testNotNullable() throws NoSuchFieldException {
38          FieldInjectionPoint nonnull = new FieldInjectionPoint(getClass().getDeclaredField("foo"));
39          assertThat(nonnull.isNullable(), equalTo(false));
40      }
41  
42      @Test
43      public void testNullable() throws NoSuchFieldException {
44          FieldInjectionPoint field = new FieldInjectionPoint(getClass().getDeclaredField("nullable"));
45          assertThat(field.isNullable(), equalTo(true));
46      }
47  
48      @Test
49      public void testSerialize() throws NoSuchFieldException, IOException, ClassNotFoundException {
50          FieldInjectionPoint fip = new FieldInjectionPoint(getClass().getDeclaredField("foo"));
51          FieldInjectionPoint fip2 = SerializationUtils.clone(fip);
52          assertThat(fip2, equalTo(fip));
53          assertThat(fip2, not(sameInstance(fip)));
54      }
