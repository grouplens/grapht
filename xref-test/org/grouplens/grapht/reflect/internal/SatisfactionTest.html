62      private Set<InjectionPoint> getInjectionPoints(Satisfaction s) {
63          Set<InjectionPoint> detected = new HashSet<InjectionPoint>();
64          for (Desire d: s.getDependencies()) {
65              detected.add(((ReflectionDesire) d).getInjectionPoint());
66          }
67          return detected;
68      }

73          Set<InjectionPoint> d = getInjectionPoints(s);
74          Assert.assertEquals(typeCInjectPoints, d);
75          Assert.assertEquals(TypeC.class, s.getErasedType());

78      @Test
79      public void testClassSatisfactionProvider() throws Exception {
80          InterfaceA a1 = new TypeA();
81          TypeA a2 = new TypeA();
82          InterfaceB b1 = new TypeB();
83          TypeB b2 = new TypeB();
84  
85          Map<Desire,Instantiator> providers = Maps.newHashMap();
86          providers.put(new ReflectionDesire(TypeC.CONSTRUCTOR), Instantiators.ofInstance(10));
87          providers.put(new ReflectionDesire(TypeC.INTERFACE_A), Instantiators.ofInstance(a1));
88          providers.put(new ReflectionDesire(TypeC.TYPE_A), Instantiators.ofInstance(a2));
89          providers.put(new ReflectionDesire(TypeC.INTERFACE_B), Instantiators.ofInstance(b1));
90          providers.put(new ReflectionDesire(TypeC.TYPE_B), Instantiators.ofInstance(b2));
91          
92          Instantiator provider = new ClassSatisfaction(TypeC.class).makeInstantiator(providers);
93          Object o = provider.instantiate();
94          
95          Assert.assertTrue(o instanceof TypeC);
96          
97          TypeC c = (TypeC) o;
98          Assert.assertSame(a1, c.getInterfaceA());
99          Assert.assertSame(a2, c.getTypeA());
100         Assert.assertSame(b1, c.getInterfaceB());
101         Assert.assertSame(b2, c.getTypeB());
102         Assert.assertEquals(10, c.getIntValue());
103     }
