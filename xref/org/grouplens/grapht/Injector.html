19  package org.grouplens.grapht;

21  import javax.inject.Qualifier;
22  import java.lang.annotation.Annotation;
23  
24  /**
25   * <p>
26   * Injector uses dependency injection to act as a factory for creating instances
27   * with complex dependencies. A default implementation of Injector can easily be
28   * created by using an {@link InjectorBuilder}:
29   * 
30   * <pre>
31   * InjectorBuilder b = new InjectorBuilder();
32   * b.bind(Foo.class).to(Bar.class);
33   * b.applyModule(new MyCustomModule());
34   * // other bindings
35   * Injector i = b.build();
36   * assert (i.getInstance(Foo.class) instanceof Bar);
37   * </pre>
38   * <p>
39   * Alternatively, {@link BindingFunctionBuilder} and {@link org.grouplens.grapht.solver.DependencySolver}
40   * can be used to create your own Injector implementations.
41   * 
42   * @author <a href="http://grouplens.org">GroupLens Research</a>
43   */
44  public interface Injector {
45      /**
46       * <p>
47       * Get an instance of T based on the bindings that this Injector was
48       * configured with. An exception is thrown if the request type cannot be
49       * instantiated with dependency injection.
50       * <p>
51       * Injectors may memoize or cache previously created objects. As an example,
52       * the Injector created by {@link InjectorBuilder} reuses instances where
53       * possible.
54       * 
55       * @param <T> The object type being created
56       * @param type The class type
57       * @return An instance of type T
58       * @throws ConstructionException if type cannot be instantiated
59       */
60      <T> T getInstance(Class<T> type) throws InjectionException;
61  
62      /**
63       * <p>
64       * Get an instance of T with the given {@link Qualifier} annotation.
65       * 
66       * @param <T> The object type
67       * @param qualifier The qualifier on of the returned instance
68       * @param type The class type
69       * @return An instance of type T
70       * @throws ConstructionException if type cannot be instantiated
71       */
72      <T> T getInstance(Annotation qualifier, Class<T> type) throws ConstructionException, InjectionException;
73  }
