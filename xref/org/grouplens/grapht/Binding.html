19  package org.grouplens.grapht;
20  
21  import org.grouplens.grapht.solver.BindRule;
22  import org.grouplens.grapht.reflect.Satisfaction;
23  
24  import javax.annotation.Nonnull;
25  import javax.annotation.Nullable;
26  import javax.inject.Provider;
27  import javax.inject.Qualifier;
28  import java.lang.annotation.Annotation;
29  
30  /**
31   * Binding is part of the fluent API used for configuring an {@link Injector}.
32   * It represents a binding action from one type to another type.
33   * 
34   * @author <a href="http://grouplens.org">GroupLens Research</a>
35   * @param <T> The source type
36   */
37  public interface Binding<T> {
38      /**
39       * <p>
40       * Configure the binding to match the given {@link Qualifier} annotation.
41       * The given annotation type must be annotated with {@link Qualifier}. The
42       * created binding will match injection points only if the qualifier is
43       * applied to the injection point, unless the annotation inherits from the
44       * default qualifier.
45       * <p>
46       * This will override any previous name or qualifier annotation.
47       * 
48       * @param qualifier The Qualifier that must match
49       * @return A newly configured Binding
50       */
51      Binding<T> withQualifier(@Nonnull Class<? extends Annotation> qualifier);
52  
53      /**
54       * Configure the binding to match injection points that have been annotated with the exact
55       * annotation instance.
56       *
57       * <p>This will override any previous name or qualifier annotation.
58       *
59       * <p>The annotation provided must be serializable.  Annotations built by {@link
60       * org.grouplens.grapht.annotation.AnnotationBuilder} (recommended) or retrieved from the Java
61       * reflection API are serializable; if you use some other annotation implementation, it must be
62       * serializable.
63       *
64       * @param annot The annotation instance to match.
65       * @return A newly configured Binding
66       */
67      Binding<T> withQualifier(@Nonnull Annotation annot);
68  
69      /**
70       * Configure the binding to match injection points that have any qualifier annotation (including
71       * no qualifier).
72       *
73       * <p>This will override any previous name or qualifier annotation.
74       *
75       * @return A newly configured Binding
76       */
77      Binding<T> withAnyQualifier();
78  
79      /**
80       * <p>
81       * Configure the binding to only match injection points that have no
82       * qualifier. By default, the binding matches any injection point with the
83       * given type, whether or not its been qualified. A qualified binding to the
84       * same type will still be preferred first when resolving a qualified
85       * injection point.
86       * 
87       * @return A newly configured binding
88       */
89      Binding<T> unqualified();
90  
91      /**
92       * Exclude the provided type from being matched when examining injection
93       * points. Bindings can generate multiple {@link BindRule BindRules} for
94       * super and sub types. Excluded classes for a binding will not have
95       * BindRules generated for them.
96       * 
97       * @param exclude The type to exclude from automated rule generation
98       * @return A newly configured Binding
99       */
100     Binding<T> exclude(@Nonnull Class<?> exclude);
101     
102     /**
103      * Configure the binding so that a shared instance is always used when
104      * satisfying matched injection points, effectively making it a singleton or
105      * memoized within its container.
106      * 
107      * @return A newly configured Binding
108      */
109     Binding<T> shared();
110     
111     /**
112      * Configure the binding so that new instances are always created when
113      * satisfying matched injection.
114      * 
115      * @return A newly configured binding
116      */
117     Binding<T> unshared();
118 
119     /**
120      * Configure the binding so that its results are 'fixed'.  Fixed results will not be rewritten
121      * by the dependency solver in rewrite mode.
122      * @return A newly configured binding.
123      */
124     Binding<T> fixed();
125 
126     /**
127      * <p>
128      * Complete this binding by specifying a subtype that will satisfy the
129      * desired type. The implementation does not have to be instantiable; if
130      * it's not then additional bindings must be configured to bind to reach an
131      * instantiable type. It is recommended for types to be instantiable.
132      * <p>
133      * The given type may have its own dependencies that will have to be
134      * satisfied by other bindings.
135      * <p>It is permissible to have two bindings forming a chain, like
136      * <code>A &rarr; B &rarr; C</code>. The {@code chained} parameter controls
137      * whether the chain is followed.  If {@code chained == false} for the {@code A &rarr; B}
138      * binding, then the {@code B &rarr; C} binding is not followed (and the {@code A &rarr; B}
139      * binding is called <em>terminal</em>).
140      * 
141      * @param impl The implementation type
142      * @param chained Whether further binding lookup will be done on the implementation type.
143      *                {@code true} allows lookup, {@code false} creates a terminal binding.
144      */
145     void to(@Nonnull Class<? extends T> impl, boolean chained);
146 
147     /**
148      * Bind to an implementation type non-terminally.  This calls {@link #to(Class, boolean)}
149      * as {@code this.to(impl, true)}.
150      *
151      * @param impl The implementation type.
152      */
153     void to(@Nonnull Class<? extends T> impl);
154 
155     /**
156      * Complete this binding by specifying an instance to use. The instance will
157      * be used to satisfy matched injection points. Because the instance never
158      * changes, any cache policy assigned by {@link #shared()} or
159      * {@link #unshared()} is effectively ignored.
160      * 
161      * @param instance The instance to use. If {@code null}, binds explicitly to
162      *                 null.
163      */
164     void to(@Nullable T instance);
165 
166     /**
167      * Complete this binding by specifying a Provider class to be instantiated
168      * and used to create instances of type T. The Provider class may have its
169      * own dependencies that will be resolved by the injector.
170      * 
171      * @param provider The provider type that will satisfy this binding
172      */
173     void toProvider(@Nonnull Class<? extends Provider<? extends T>> provider);
174 
175     /**
176      * Complete this binding by specifying a Provider instance that will be used
177      * to create instances of type T to satisfy this binding.
178      * 
179      * @param provider The provider instance
180      */
181     void toProvider(@Nonnull Provider<? extends T> provider);
182 
183 
184     /**
185      * Complete this binding by explicitly binding to {@code null}. The resulting
186      * bindings may not create an instantiable graph, as non-nullable injection points
187      * still require a non-null instance.
188      */
189     void toNull();
190 
191     /**
192      * Complete this binding by explicitly binding to {@code null} with a type.
193      * @param type The type of {@code null} to bind.
194      * @see #toNull()
195      */
196     void toNull(Class<? extends T> type);
197 
198     /**
199      * Bind this binding directly to a satisfaction.
200      *
201      * <p><strong>Note:</strong> this method is intended for use by applications that extend Grapht,
202      * or for bindings to other JVM languages. Most applications will have no use for this method,
203      * and developers consider if one of the other methods is more applicable for their situation.
204      * </p>
205      *
206      * <p>Bindings to satisfactions are always {@linkplain org.grouplens.grapht.solver.BindRule#isTerminal() terminal}.
207      * </p>
208      *
209      * @param sat The satisfaction to bind to.
210      */
211     void toSatisfaction(@Nonnull Satisfaction sat);
212 }
