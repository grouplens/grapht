28  /**
29   * Utilities and methods for building and working with {@link org.grouplens.grapht.Instantiator}s.

37      /**
38       * Create an instantiator that returns an instance.
39       * @param inst The instance to return (must be non-null).
40       * @return An instantiator that returns {@code inst}.
41       */
42      public static Instantiator ofInstance(Object inst) {
43          Preconditions.checkNotNull(inst, "instance");
44          return new InstanceInstantiator(inst);
45      }
46  
47      /**
48       * Create an instantiator that returns a null value.
49       * @param type The type of null value to return.
50       * @return An instantiator that returns {@code null}.
51       */
52      public static Instantiator ofNull(Class<?> type) {
53          Preconditions.checkNotNull(type, "instance");
54          return new InstanceInstantiator(null, type);
55      }

57      /**
58       * Convert a providerInstantiator to an instantiator.  Any exception thrown by the provider - including a
59       * runtime exception - is wrapped in an {@link ConstructionException}.
60       * @param provider The providerInstantiator to wrap.
61       * @return An instantiator wrapping {@code providerInstantiator}.
62       */
63      public static Instantiator ofProvider(Provider<?> provider) {
64          Preconditions.checkNotNull(provider, "provider");
65          return new ProviderInstantiator(ofInstance(provider));
66      }
67  
68      /**
69       * Flatten an instnatiator of providers into an instantiator of the provided type.  Any
70       * exception thrown by the provider - including a runtime exception - is wrapped in an
71       * {@link ConstructionException}.
72       * @param pinst The providerInstantiator instantiator to wrap.
73       * @return An instantiator wrapping {@code providerInstantiator}.
74       */
75      public static Instantiator ofProviderInstantiator(Instantiator pinst) {
76          Preconditions.checkNotNull(pinst, "provider instantiator");
77          Preconditions.checkArgument(Provider.class.isAssignableFrom(pinst.getType()),
78                                      "instantiator is not of type Provider");
79          return new ProviderInstantiator(pinst);
80      }
81  
82      /**
83       * Convert an instantiator to a provider.
84       * @param instantiator The instantiator to convert.
85       * @return A provider whose {@link javax.inject.Provider#get()} method invokes the instantiator.
86       */
87      public static Provider<?> toProvider(Instantiator instantiator) {
88          // First try to unpack the instantiator
89          if (instantiator instanceof ProviderInstantiator) {
90              Instantiator itor = ((ProviderInstantiator) instantiator).providerInstantiator;
91              if (itor instanceof InstanceInstantiator) {
92                  return (Provider) ((InstanceInstantiator) itor).instance;
93              }
94          }
95          // Otherwise, wrap it.
96          return new InstantiatorProvider(instantiator);
97      }
98  
99      /**
100      * Memoize an instantiator.
101      * @param instantiator The instantiator to memoize.
102      * @return An instantiator that memoizes {@code instantiator}.
103      */
104     public static Instantiator memoize(Instantiator instantiator) {
105         Preconditions.checkNotNull(instantiator, "instantiator");
106         return new MemoizingInstantiator(instantiator);
107     }

109     private static final class InstanceInstantiator implements Instantiator {
110         private final Object instance;
111         private final Class<?> type;
112 
113         public InstanceInstantiator(Object inst) {
114             this(inst, (Class) inst.getClass());
115         }
116 
117         public InstanceInstantiator(Object inst, Class<?> typ) {
118             instance = inst;
119             type = typ;
120         }
121 
122         @Override
123         public Object instantiate() throws ConstructionException {
124             return instance;
125         }
126 
127         @Override
128         public Class getType() {
129             return type;
130         }
131     }
132 
133     private static class ProviderInstantiator implements Instantiator {
134         private final Instantiator providerInstantiator;
135 
136         public ProviderInstantiator(Instantiator prov) {
137             providerInstantiator = prov;
138         }
139 
140         @Override
141         public Object instantiate() throws ConstructionException {
142             Provider<?> provider = (Provider) providerInstantiator.instantiate();
143             try {
144                 return provider.get();
145             } catch (Throwable th) {
146                 throw new ConstructionException(getType(), "Error invoking provider " + providerInstantiator, th);
147             }
148         }
149 
150         @SuppressWarnings("unchecked")
151         @Override
152         public Class<?> getType() {
153             return Types.getProvidedType(providerInstantiator.getType());
154         }
155     }

157     private static class MemoizingInstantiator implements Instantiator {
158         private final Instantiator delegate;
159         private volatile boolean instantiated = false;
160         private Object instance = null;
161         private Throwable error = null;
162 
163         public MemoizingInstantiator(Instantiator inst) {
164             delegate = inst;
165         }
166 
167         @Override
168         public Object instantiate() throws ConstructionException {
169             if (!instantiated) {
170                 synchronized (this) {
171                     if (!instantiated) {
172                         try {
173                             instance = delegate.instantiate();
174                         } catch (Throwable th) {
175                             error = th;
176                         }
177                         instantiated = true;
178                     }
179                 }
180             }
181 
182             if (error != null) {
183                 Throwables.propagateIfPossible(error, ConstructionException.class);
184                 // shouldn't happen, but hey.
185                 throw new RuntimeException("Unexpected instantiation exception", error);
186             } else {
187                 return instance;
188             }
189         }
190 
191         @Override
192         public Class getType() {
193             return delegate.getType();
194         }
195     }

197     private static class InstantiatorProvider implements TypedProvider {
198         private final Instantiator instantiator;
199 
200         public InstantiatorProvider(Instantiator itor) {
201             instantiator = itor;
202         }
203 
204         @Override
205         public Class<?> getProvidedType() {
206             return instantiator.getType();
207         }
208 
209         @Override
210         public Object get() {
211             try {
212                 return getProvidedType().cast(instantiator.instantiate());
213             } catch (ConstructionException ex) {
214                 throw new RuntimeException(ex);
215             }
216         }
217     }
