19  package org.grouplens.grapht;
20  
21  import org.apache.commons.lang3.builder.EqualsBuilder;
22  import org.apache.commons.lang3.builder.HashCodeBuilder;
23  import org.grouplens.grapht.reflect.Satisfaction;
24  import org.grouplens.grapht.util.Preconditions;
25  
26  import java.io.Serializable;
27  
28  /**
29   * A component to be instantiated in the final dependency plan.  A component consists of a {@link
30   * Satisfaction} and related information for instantiating it (such as the {@link CachePolicy}).
31   *
32   * @author <a href="http://grouplens.org">GroupLens Research</a>
33   */
34  public class Component implements Serializable {
35      private static final long serialVersionUID = 5L;
36      
37      private final Satisfaction satisfaction;
38      private final CachePolicy cachePolicy;
39  
40      private Component(Satisfaction satisfaction, CachePolicy policy) {
41          Preconditions.notNull("satisfaction", satisfaction);
42          Preconditions.notNull("policy", policy);
43  
44          this.satisfaction = satisfaction;
45          cachePolicy = policy;
46      }
47  
48      /**
49       * Create a new Component wrapping the given satisfaction and cache policy.  The injector is
50       * responsible for using the satisfaction to implement this component consistent with its
51       * cache policy.
52       *
53       * @param satisfaction The satisfaction to wrap
54       * @param policy       The policy used with this satisfaction
55       * @throws NullPointerException the satisfaction or policy is null
56       */
57      public static Component create(Satisfaction satisfaction, CachePolicy policy) {
58          return new Component(satisfaction, policy);
59      }
60  
61      /**
62       * @return The Satisfaction stored in this pair
63       */
64      public Satisfaction getSatisfaction() {
65          return satisfaction;
66      }
67      
68      /**
69       * @return The CachePolicy that must be implemented on top of the providers
70       *         from this {@link #getSatisfaction() satisfaction}
71       */
72      public CachePolicy getCachePolicy() {
73          return cachePolicy;
74      }
75  
76      @Override
77      public boolean equals(Object o) {
78          if (!(o instanceof Component)) {
79              return false;
80          }
81              
82          Component c = (Component) o;
83          EqualsBuilder eqb = new EqualsBuilder();
84          return eqb.append(satisfaction, c.satisfaction)
85                    .append(cachePolicy, c.cachePolicy)
86                    .isEquals();
87      }
88      
89      @Override
90      public int hashCode() {
91          HashCodeBuilder hcb = new HashCodeBuilder();
92          return hcb.append(satisfaction)
93                    .append(cachePolicy)
94                    .toHashCode();
95      }
96      
97      @Override
98      public String toString() {
99          return "(" + satisfaction + ", " + cachePolicy + ")";
100     }
101 }
