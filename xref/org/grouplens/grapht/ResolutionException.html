29  /**
30   * Exception thrown when there is a dependency resolution error.
31   *
32   * @since 0.9
33   * @author <a href="http://grouplens.org">GroupLens Research</a>
34   */
35  @SuppressWarnings("deprecation")
36  public class ResolutionException extends SolverException {
37      private static final long serialVersionUID = 1L;
38  
39      public ResolutionException() {
40          super();
41      }
42  
43      public ResolutionException(String msg) {
44          super(msg);
45      }
46  
47      public ResolutionException(Throwable throwable) {
48          super(throwable);
49      }
50  
51      public ResolutionException(String msg, Throwable throwable) {
52          super(msg, throwable);
53      }
54      
55      protected String format(InjectionContext ctx, DesireChain desires) {
56          StringBuilder sb = new StringBuilder();
57          
58          // type path
59          sb.append("Context:\n");
60          sb.append("  Type path:\n");
61          for (Pair<Satisfaction, InjectionPoint> path: ctx) {
62              Satisfaction sat = path.getLeft();
63              Class<?> type = sat == null ? null : sat.getErasedType();
64              sb.append("    ")
65                .append(format(path.getRight(), type))
66                .append('\n');
67          }
68          sb.append('\n');
69          
70          // desire chain
71          sb.append("  Prior desires:\n");
72          for (Desire desire: desires.getPreviousDesires()) {
73              sb.append("    ")
74                .append(format(desire.getInjectionPoint(), desire.getDesiredType()))
75                .append('\n');
76          }
77  
78          return sb.toString();
79      }
80  
81      protected String format(InjectionPoint ip) {
82          return format(ip, ip.getErasedType());
83      }
84  
85      protected String format(InjectionPoint ip, Class<?> type) {
86          if (type == null) {
87              type = ip.getErasedType();
88          }
89          String base = (ip.getQualifier() != null ? ip.getQualifier() + ":" : "");
90          String name = type == null ? null : type.getName();
91          return base + name;
92      }
93  }
