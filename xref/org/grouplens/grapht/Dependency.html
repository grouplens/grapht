19  package org.grouplens.grapht;
20  
21  import com.google.common.base.Predicate;
22  import org.grouplens.grapht.reflect.Desire;
23  import org.grouplens.grapht.solver.DesireChain;
24  
25  import javax.annotation.Nullable;
26  import java.io.Serializable;
27  import java.util.EnumSet;
28  
29  /**
30   * Track information about a particular resolved dependency. Used as the edge of DI graph nodes
31   *
32   * @author <a href="http://www.grouplens.org">GroupLens Research</a>
33   */
34  public class Dependency implements Serializable {
35      private static final long serialVersionUID = 1L;
36  
37      private final DesireChain desireChain;
38      private final EnumSet<Flag> flags;
39  
40      private Dependency(DesireChain chain, EnumSet<Flag> flagSet) {
41          desireChain = chain;
42          flags = flagSet.clone();
43      }
44  
45      public static Dependency create(DesireChain desires, EnumSet<Flag> flags) {
46          return new Dependency(desires, flags);
47      }
48  
49      /**
50       * Get the desire chain associated with this dependency.
51       * @return The chain of desires followed in resolving this dependency.
52       */
53      public DesireChain getDesireChain() {
54          return desireChain;
55      }
56  
57      /**
58       * Convenience method to get the initial desire that prompted this dependency.
59       * @return The initial desire that prompted this dependency.
60       */
61      public Desire getInitialDesire() {
62          return desireChain.getInitialDesire();
63      }
64  
65      /**
66       * Get the flags associated with this dependency.
67       * @return The flags associated with this dependency.
68       */
69      public EnumSet<Flag> getFlags() {
70          return flags;
71      }
72  
73      /**
74       * Query whether this dependency is immune to rewriting.
75       * @return {@code true} if this dependency is immune to rewriting.
76       */
77      public boolean isFixed() {
78          return flags.contains(Flag.FIXED);
79      }
80  
81      /**
82       * Create a predicate matching dependencies with the specified initial desire.
83       * @param d The desire.
84       * @return A predicate that accepts dependencies whose initial desire is {@code d}.
85       */
86      public static Predicate<Dependency> hasInitialDesire(final Desire d) {
87          return new Predicate<Dependency>() {
88              @Override
89              public boolean apply(@Nullable Dependency input) {
90                  return input != null && input.getInitialDesire().equals(d);
91              }
92          };
93      }
94  
95      @Override
96      public boolean equals(Object o) {
97          if (this == o) return true;
98          if (o == null || getClass() != o.getClass()) return false;
99  
100         Dependency that = (Dependency) o;
101 
102         if (!desireChain.equals(that.desireChain)) return false;
103         if (!flags.equals(that.flags)) return false;
104 
105         return true;
106     }
107 
108     @Override
109     public int hashCode() {
110         int result = desireChain.hashCode();
111         result = 31 * result + flags.hashCode();
112         return result;
113     }
114 
115     @Override
116     public String toString() {
117         return "Dependency(" + desireChain + ", " + flags + ")";
118     }
119 
120     /**
121      * Flags associated with a dependency.
122      */
123     public static enum Flag {
124         /**
125          * Indicates that a dependency is immune to rewriting.
126          */
127         FIXED;
128 
129         public static EnumSet<Flag> emptySet() {
130             return EnumSet.noneOf(Flag.class);
131         }
132     }
133 }
