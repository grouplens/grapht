19  package org.grouplens.grapht.graph;

21  import com.google.common.base.Functions;
22  import com.google.common.collect.FluentIterable;
23  import com.google.common.collect.Maps;
24  import com.google.common.collect.Sets;
25  import org.apache.commons.lang3.tuple.Pair;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  import java.util.List;
30  import java.util.Map;
31  import java.util.Set;

33  /**
34   * Merges graphs to remove redundant nodes.  This takes graphs and merges them, pruning redundant
35   * nodes within the graphs and between graphs previously merged.  It remembers graphs it has
36   * previously seen to allow nodes to be reused across multiple graphs.
37   *
38   * @param <V> The vertex type of graphs to merge.
39   * @param <E> The edge type of graphs to merge.
40   * @since 0.7
41   * @author <a href="http://grouplens.org">GroupLens Research</a>
42   * @todo Allow arbitrary equivalence relations over graph nodes so this class is less specialized.
43   */
44  public class MergePool<V,E> {
45      private static final Logger logger = LoggerFactory.getLogger(MergePool.class);
46  
47      private final Set<DAGNode<V,E>> pool;
48  
49      private MergePool() {
50          pool = Sets.newLinkedHashSet();
51      }
52  
53      /**
54       * Create a merge pool that checks node labels for equality.
55       *
56       * @param <V> The node label type.
57       * @param <E> The edge label type.
58       * @return A new merge pool.
59       */
60      public static <V,E> MergePool<V,E> create() {
61          return new MergePool<V, E>();
62      }
63  
64      /**
65       * Merge and simplify a graph.  This will coalesce redundant nodes (equivalent labels and
66       * outgoing edge destinations), and will prefer to use nodes from graphs seen previously.
67       * This allows deduplication across multiple graphs.
68       *
69       * <p><strong>Noteo:</strong> edge labels are ignored for the purpose of merging.</p>
70       *
71       * @param graph The graph to simplify.
72       * @return The new simplified, merged graph.
73       */
74      public DAGNode<V,E> merge(DAGNode<V, E> graph) {
75          List<DAGNode<V, E>> sorted = graph.getSortedNodes();
76  
77          Map<Pair<V,Set<DAGNode<V,E>>>, DAGNode<V,E>> nodeTable = Maps.newHashMap();
78          for (DAGNode<V, E> node: pool) {
79              Pair<V, Set<DAGNode<V, E>>> key =
80                      Pair.of(node.getLabel(), node.getAdjacentNodes());
81              assert !nodeTable.containsKey(key);
82              nodeTable.put(key, node);
83          }
84  
85          // We want to map nodes to their previous merged versions
86          Map<DAGNode<V,E>, DAGNode<V,E>> mergedMap = Maps.newHashMap();
87          // Now start processing nodes
88          for (DAGNode<V, E> toMerge: sorted) {
89              V sat = toMerge.getLabel();
90              // Resolve the merged neighbors of this node.  They have already been
91              // merged, since we are going in topological order.
92              Set<DAGNode<V, E>> neighbors =
93                      FluentIterable.from(toMerge.getOutgoingEdges())
94                                    .transform(DAGEdge.<V,E>extractTail())
95                                    .transform(Functions.forMap(mergedMap))
96                                    .toSet();
97  
98              // See if we have already created an equivalent to this node
99              DAGNode<V, E> newNode = nodeTable.get(Pair.of(sat, neighbors));
100             if (newNode == null) {
101                 // No, let's start building one
102                 DAGNodeBuilder<V,E> bld = DAGNode.newBuilder();
103 
104                 boolean changed = false;
105                 bld.setLabel(sat);
106                 logger.debug("Adding new node to merged graph for satisfaction: {}", sat);
107 
108                 for (DAGEdge<V, E> edge: toMerge.getOutgoingEdges()) {
109                     // create a new edge with the merged tail and same label
110                     DAGNode<V, E> filtered = mergedMap.get(edge.getTail());
111                     bld.addEdge(filtered, edge.getLabel());
112                     // have we made a change to this node?
113                     changed |= !filtered.equals(edge.getTail());
114                 }
115 
116                 if (changed) {
117                     // one of the node's neighbors has been replaced with merged version
118                     // so use the new node
119                     newNode = bld.build();
120                 } else {
121                     // no edges were changed, leave the node unmodified
122                     newNode = toMerge;
123                 }
124                 nodeTable.put(Pair.of(sat, neighbors), newNode);
125             } else {
126                 logger.debug("Node already in merged graph for satisfaction: {}", toMerge.getLabel());
127             }
128 
129             // update merge map so future equivalent nodes get replaced with this one
130             mergedMap.put(toMerge, newNode);
131         }
132 
133         // now let's find our return value - what did we merge the graph root to?
134         DAGNode<V, E> newRoot = mergedMap.get(graph);
135         // remember all its nodes for future merge operations
136         pool.addAll(newRoot.getReachableNodes());
137         // and we're done
138         return newRoot;
139     }
