33  /**
34   * A node in a (rooted) DAG.  Since DAGs are rooted, a full graph is just represented by its root
35   * node.  Nodes are compared using reference equality, so distinct nodes do not compare equal even
36   * if they have identical labels and edge sets.
37   *
38   * <p>Nodes and edges may not have null labels.  There <em>may</em> be multiple edges from one
39   * node to another, so long as those edges have distinct labels.
40   *
41   * <p>Nodes know about all nodes reachable from them, and the edges connecting those nodes.
42   *
43   * <p>DAGs and their nodes are immutable.  You can build them using a {@linkplain DAGNodeBuilder builder},
44   * obtained from {@link #newBuilder(Object)}.
45   *
46   * @param <V> The type of node (vertex) labels.
47   * @param <E> The type of edge labels.
48   * @since 0.7.0
49   * @author <a href="http://www.grouplens.org">GroupLens Research</a>
50   */
51  @Immutable
52  public class DAGNode<V,E> implements Serializable {

96      /**
97       * Create a new builder initialized to build a copy of the specified node.
98       * @param node The node to copy.
99       * @param <V> The type of node labels.
100      * @param <E> The type of edge labels.
101      * @return A new builder initialized with the labels and edges of {@code node}.
102      */
103     public static <V,E> DAGNodeBuilder<V,E> copyBuilder(DAGNode<V,E> node) {
104         DAGNodeBuilder<V,E> bld = newBuilder(node.getLabel());
105         for (DAGEdge<V,E> edge: node.getOutgoingEdges()) {
106             bld.addEdge(edge.getTail(), edge.getLabel());
107         }
108         return bld;
109     }

111     /**
112      * Construct a new DAG node.
113      * @param lbl The label.
114      * @param edges The edges.  This takes pairs, not actual edge objects, because the edge objects
115      *              need to be constructed within the constructor in order to create the circular
116      *              references back to the head nodes properly.
117      */
118     DAGNode(@Nonnull V lbl, Iterable<Pair<DAGNode<V,E>,E>> edges) {
119         label = lbl;
120         ImmutableSet.Builder<DAGEdge<V,E>> bld = ImmutableSet.builder();
121         for (Pair<DAGNode<V,E>,E> pair: edges) {
122             DAGEdge<V,E> edge = new DAGEdge<V, E>(this, pair.getLeft(), pair.getRight());
123             bld.add(edge);
124         }
125         outgoingEdges = bld.build();
126         initializeCaches();
127     }

129     /**
130      * Initialize caches for traversing this node.
131      */
132     private void initializeCaches() {
133         reverseEdgeCache = Suppliers.memoize(new EdgeMapSupplier());
134         reachableNodeCache = Suppliers.memoize(new NodeSetSupplier());
135         topologicalSortCache = Suppliers.memoize(new TopologicalSortSupplier());
136     }

171      * @param target The target node.

176     public DAGEdge<V,E> getOutgoingEdge(DAGNode<V,E> target, E label) {
177         for (DAGEdge<V,E> edge: outgoingEdges) {
178             if (edge.getTail().equals(target) && edge.getLabel().equals(label)) {
179                 return edge;
180             }
181         }

249     /**
250      * Helper mode for {@link #getSortedNodes()}, via {@link TopologicalSortSupplier}.  This method
251      * does a depth-first traversal of the nodes, adding each to the {@code visited} set when it is
252      * left.  This results in {@code visited} being a topological sort.
253      *
254      * @param visited The set of nodes seen so far.
255      */
256     private void sortVisit(LinkedHashSet<DAGNode<V,E>> visited) {
257         if (!visited.contains(this)) {
258             for (DAGEdge<V,E> nbr: outgoingEdges) {
259                 nbr.getTail().sortVisit(visited);
260             }
261             // neighbors won't have added this, or we have an impossible cycle
262             assert !visited.contains(this);
263             visited.add(this);
264         }
265     }

276     /**
277      * Replace one node with another in this graph.  All edges referencing {@code node} are replaced
278      * with edges referencing {@code replacement}.
279      *
280      * @param node The node to replace.
281      * @param replacement The replacement node.
282      * @param memory A table to remember node replacements.  It maintains a mapping of every node
283      *               that has to be replaced with the node that replaces it.  This map should
284      *               usually be empty on the initial call to this method.  In particular, it should
285      *               not contain any reachable nodes on the initial call, or unexpected behavior
286      *               may arise.  Recursive calls of this method to itself do contain such nodes.
287      * @return The graph with the replaced node.
288      */
289     public DAGNode<V,E> replaceNode(DAGNode<V,E> node, DAGNode<V,E> replacement,
290                                     Map<DAGNode<V,E>,DAGNode<V,E>> memory) {
291         if (this.equals(node)) {
292             memory.put(node, replacement);
293             return replacement;
294         } else if (memory.containsKey(this)) {
295             // we have already been replaced, reuse the replacement
296             return memory.get(this);
297         } else if (getReachableNodes().contains(node)) {
298             DAGNodeBuilder<V,E> bld = newBuilder(label);
299             for (DAGEdge<V,E> edge: outgoingEdges) {
300                 DAGNode<V,E> newTail = edge.getTail().replaceNode(node, replacement, memory);
301                 bld.addEdge(newTail, edge.getLabel());
302             }
303             DAGNode<V,E> repl = bld.build();
304             memory.put(this, repl);
305             return repl;
306         } else {
307             return this;
308         }
309     }

311     /**
312      * Do a breadth-first search for a node.
313      *
314      * @param pred The predicate for matching nodes.
315      * @return The first node matching {@code pred} in a breadth-first search, or {@code null} if no
316      *         such node is found.
317      */
318     public DAGNode<V, E> findNodeBFS(@Nonnull Predicate<? super DAGNode<V, E>> pred) {
319         if (pred.apply(this)) {
320             return this;
321         }
322 
323         Queue<DAGNode<V, E>> work = Lists.newLinkedList();
324         Set<DAGNode<V, E>> seen = Sets.newHashSet();
325         work.add(this);
326         seen.add(this);
327         while (!work.isEmpty()) {
328             DAGNode<V, E> node = work.remove();
329             for (DAGEdge<V, E> e : node.getOutgoingEdges()) {
330                 // is this the node we are looking for?
331                 DAGNode<V, E> nbr = e.getTail();
332                 if (!seen.contains(nbr)) {
333                     if (pred.apply(nbr)) {
334                         return nbr;
335                     } else {
336                         seen.add(nbr);
337                         work.add(nbr);
338                     }
339                 }
340             }
341         }
342 
343         // no node found
344         return null;
345     }

347     /**
348      * Do a breadth-first search for an edge.
349      *
350      * @param pred The predicate for matching nodes.
351      * @return The first node matching {@code pred} in a breadth-first search, or {@code null} if no
352      *         such node is found.
353      */
354     public DAGEdge<V, E> findEdgeBFS(@Nonnull Predicate<? super DAGEdge<V, E>> pred) {
355         Queue<DAGNode<V, E>> work = Lists.newLinkedList();
356         Set<DAGNode<V, E>> seen = Sets.newHashSet();
357         work.add(this);
358         seen.add(this);
359         while (!work.isEmpty()) {
360             DAGNode<V, E> node = work.remove();
361             for (DAGEdge<V, E> e : node.getOutgoingEdges()) {
362                 // is this the edge we are looking for?
363                 if (pred.apply(e)) {
364                     return e;
365                 } else if (!seen.contains(e.getTail())) {
366                     seen.add(e.getTail());
367                     work.add(e.getTail());
368                 }
369             }
370         }
371 
372         // no node found
373         return null;
374     }

376     /**
377      * Transform the edges in this graph.  Edges in parent nodes are passed <em>after</em> their
378      * target nodes are rewritten, if necessary.
379      *
380      * @param function The edge transformation function.  Any edge returned by this function must
381      *                 have the same head node as the function it was passed.  The transform
382      *                 function may need to replace the head node on a returned edge; the label and
383      *                 tail will be preserved.  If the function returns {@code null}, that is the
384      *                 same as returning its input unmodified.
385      * @return The rewritten graph.
386      */
387     public DAGNode<V,E> transformEdges(Function<? super DAGEdge<V,E>, ? extends DAGEdge<V,E>> function) {
388         // builder for new node
389         DAGNodeBuilder<V,E> builder = null;
390         // intact edges (unmodified edges)
391         List<DAGEdge<V,E>> intact = Lists.newArrayListWithCapacity(outgoingEdges.size());
392         for (DAGEdge<V,E> edge: outgoingEdges) {
393             DAGNode<V,E> tail = edge.getTail();
394             DAGNode<V,E> transformedTail = tail.transformEdges(function);
395             DAGEdge<V,E> toQuery = edge;
396             if (transformedTail != tail) {
397                 // the node changed, query with the updated edge
398                 toQuery = DAGEdge.create(this, transformedTail, edge.getLabel());
399             }
400             DAGEdge<V,E> transformedEdge = function.apply(toQuery);
401             if (transformedEdge == null) {
402                 transformedEdge = toQuery;
403             }
404             if (edge.equals(transformedEdge)) {
405                 // edge unmodified
406                 if (builder == null) {
407                     intact.add(transformedEdge);
408                 } else {
409                     builder.addEdge(transformedEdge.getTail(), transformedEdge.getLabel());
410                 }
411             } else {
412                 // modified, need to transform this node
413                 if (builder == null) {
414                     builder = newBuilder(label);
415                     for (DAGEdge<V,E> done: intact) {
416                         builder.addEdge(done.getTail(), done.getLabel());
417                     }
418                 }
419                 builder.addEdge(transformedEdge.getTail(), transformedEdge.getLabel());
420             }
421         }
422 
423         if (builder != null) {
424             return builder.build();
425         } else {
426             return this;
427         }
428     }

443     public static <V> Predicate<DAGNode<V,?>> labelMatches(final Predicate<? super V> pred) {
444         return new Predicate<DAGNode<V, ?>>() {
445             @Override
446             public boolean apply(@Nullable DAGNode<V, ?> input) {
447                 V lbl = input == null ? null : input.getLabel();
448                 return pred.apply(lbl);
449             }
450         };

453     /**
454      * Supplier to compute the map of incoming edges.  Used to implement {@link #getIncomingEdgeMap()}.
455      */
456     private class EdgeMapSupplier implements Supplier<SetMultimap<DAGNode<V, E>, DAGEdge<V, E>>> {
457         @Override
458         public SetMultimap<DAGNode<V, E>, DAGEdge<V, E>> get() {
459             ImmutableSetMultimap.Builder<DAGNode<V,E>,DAGEdge<V,E>> bld = ImmutableSetMultimap.builder();
460             for (DAGEdge<V,E> nbr: outgoingEdges) {
461                 bld.put(nbr.getTail(), nbr);
462                 bld.putAll(nbr.getTail().getIncomingEdgeMap());
463             }
464             return bld.build();
465         }
466     }
