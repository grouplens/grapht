19  package org.grouplens.grapht.graph;
20  
21  import com.google.common.base.Preconditions;
22  import com.google.common.collect.Sets;
23  import org.apache.commons.lang3.tuple.Pair;
24  
25  import javax.annotation.Nonnull;
26  import java.util.Set;
27  
28  /**
29   * A builder for {@linkplain DAGNode DAG nodes}.  You can create one with {@link DAGNode#newBuilder()}
30   * or {@link DAGNode#newBuilder(Object)}.
31   *
32   * @since 0.7.0
33   * @author <a href="http://www.grouplens.org">GroupLens Research</a>
34   */
35  public class DAGNodeBuilder<V,E> {
36      private V label;
37      private Set<Pair<DAGNode<V,E>,E>> edges;
38  
39      public DAGNodeBuilder() {
40          this(null);
41      }
42  
43      public DAGNodeBuilder(V lbl) {
44          label = lbl;
45          edges = Sets.newHashSet();
46      }
47  
48      /**
49       * Set the node's label.
50       * @param lbl The node's label.
51       * @return The builder (for chaining).
52       */
53      @Nonnull
54      public DAGNodeBuilder<V,E> setLabel(@Nonnull V lbl) {
55          Preconditions.checkNotNull(lbl, "node label");
56          label = lbl;
57          return this;
58      }
59  
60      /**
61       * Get the label set for this node.
62       * @return The label currently set for the node builder.
63       */
64      public V getLabel() {
65          return label;
66      }
67  
68      /**
69       * Add an edge.
70       * @param target The target node.
71       * @param label The edge label.
72       * @return The builder (for chaining).
73       */
74      @Nonnull
75      public DAGNodeBuilder<V,E> addEdge(@Nonnull DAGNode<V,E> target,
76                                         @Nonnull E label) {
77          Preconditions.checkNotNull(target, "edge target");
78          Preconditions.checkNotNull(label, "edge label");
79          return addEdge(Pair.of(target, label));
80      }
81  
82      /**
83       * Add an edge.
84       * @param edge The target node and label for the edge.
85       * @return The builder (for chaining).
86       */
87      @Nonnull
88      public DAGNodeBuilder<V,E> addEdge(Pair<DAGNode<V,E>,E> edge) {
89          Preconditions.checkNotNull(edge, "edge");
90          Preconditions.checkNotNull(edge.getLeft(), "edge target");
91          Preconditions.checkNotNull(edge.getRight(), "edge label");
92          edges.add(edge);
93          return this;
94      }
95  
96      /**
97       * Get the set of edges.  This set is live, and can be used to modify the edges that will
98       * be put in the final builder.
99       *
100      * @return The set of edges.
101      */
102     @Nonnull
103     public Set<Pair<DAGNode<V,E>,E>> getEdges() {
104         return edges;
105     }
106 
107     @Nonnull
108     public DAGNode<V,E> build() {
109         Preconditions.checkState(label != null, "no node label set");
110         return new DAGNode<V,E>(label, edges);
111     }
112 
113     @Override
114     public String toString() {
115         return String.format("node builder with label %s and %d edges", label, edges.size());
116     }
117 }
