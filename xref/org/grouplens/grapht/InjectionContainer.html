37  /**
38   * Container for dependency-injected components.  A container is the scope of memoization, so
39   * components with a cache policy of {@link CachePolicy#MEMOIZE} will share an instance so long
40   * as they are instantiated by the same instantiator.
41   *
42   * @since 0.9
43   * @author <a href="http://www.grouplens.org">GroupLens Research</a>
44   */
45  public class InjectionContainer {
46      private static final Logger logger = LoggerFactory.getLogger(InjectionContainer.class);
47  
48      private final CachePolicy defaultCachePolicy;
49      private final Map<DAGNode<Component, Dependency>, Instantiator> providerCache;
50  
51      /**
52       * Create a new instantiator with a default policy of {@code MEMOIZE}.
53       * @return The instantiator.
54       */
55      public static InjectionContainer create() {
56          return create(CachePolicy.MEMOIZE);
57      }
58  
59      /**
60       * Create a new instantiator.
61       * @param dft The default cache policy.
62       * @return The instantiator.
63       */
64      public static InjectionContainer create(CachePolicy dft) {
65          return new InjectionContainer(dft);
66      }
67  
68      private InjectionContainer(CachePolicy dft) {
69          defaultCachePolicy = dft;
70          providerCache = new WeakHashMap<DAGNode<Component, Dependency>, Instantiator>();
71      }
72  
73      /**
74       * Get a provider that, when invoked, will return an instance of the component represented
75       * by a graph.
76       *
77       *
78       * @param node The graph.
79       * @return A provider to instantiate {@code graph}.
80       * @see #makeInstantiator(DAGNode, SetMultimap)
81       */
82      public Instantiator makeInstantiator(DAGNode<Component, Dependency> node) {
83          return makeInstantiator(node, ImmutableSetMultimap.<DAGNode<Component, Dependency>, DAGEdge<Component, Dependency>>of());
84      }
85  
86      /**
87       * Get a provider that, when invoked, will return an instance of the component represented
88       * by a graph with back edges.  The provider will implement the cache policy, so cached nodes
89       * will return a memoized provider.
90       *
91       * @param node The graph.
92       * @param backEdges A multimap of back edges for cyclic dependencies.
93       * @return A provider to instantiate {@code graph}.
94       */
95      public Instantiator makeInstantiator(DAGNode<Component, Dependency> node,
96                                           SetMultimap<DAGNode<Component, Dependency>, DAGEdge<Component, Dependency>> backEdges) {
97          Instantiator cached = providerCache.get(node);
98          if (cached == null) {
99              logger.debug("Node has not been memoized, instantiating: {}", node.getLabel());
100             Map<Desire, Instantiator> depMap = makeDependencyMap(node, backEdges);
101 
102             Instantiator raw = node.getLabel().getSatisfaction().makeInstantiator(depMap);
103 
104             CachePolicy policy = node.getLabel().getCachePolicy();
105             if (policy.equals(CachePolicy.NO_PREFERENCE)) {
106                 policy = defaultCachePolicy;
107             }
108             if (policy.equals(CachePolicy.MEMOIZE)) {
109                 // enforce memoization on providers for MEMOIZE policy
110                 cached = Instantiators.memoize(raw);
111             } else {
112                 // Satisfaction.makeInstantiator() returns providers that are expected
113                 // to create new instances with each invocation
114                 assert policy.equals(CachePolicy.NEW_INSTANCE);
115                 cached = raw;
116             }
117             providerCache.put(node, cached);
118         }
119         return cached;
120     }
121 
122     private Map<Desire, Instantiator> makeDependencyMap(DAGNode<Component, Dependency> node, SetMultimap<DAGNode<Component, Dependency>, DAGEdge<Component, Dependency>> backEdges) {
123         Set<DAGEdge<Component,Dependency>> edges = node.getOutgoingEdges();
124         if (backEdges.containsKey(node)) {
125             ImmutableSet.Builder<DAGEdge<Component,Dependency>> bld = ImmutableSet.builder();
126             edges = bld.addAll(edges)
127                        .addAll(backEdges.get(node))
128                        .build();
129         }
130 
131         ImmutableSet.Builder<Desire> desires = ImmutableSet.builder();
132         for (DAGEdge<Component,Dependency> edge: edges) {
133             desires.add(edge.getLabel().getInitialDesire());
134         }
135 
136         return Maps.asMap(desires.build(), new DepLookup(edges, backEdges));
137     }
138 
139     /**
140      * Function to look up a desire in a set of dependency edges.
141      */
142     private class DepLookup implements Function<Desire,Instantiator> {
143         private final Set<DAGEdge<Component, Dependency>> edges;
144         private final SetMultimap<DAGNode<Component, Dependency>, DAGEdge<Component, Dependency>> backEdges;
145 
146         /**
147          * Construct a depenency lookup funciton.
148          * @param edges The set of edges to consult.
149          * @param backEdges The back edge map (to pass to {@link #makeInstantiator(DAGNode,SetMultimap)}).
150          */
151         public DepLookup(Set<DAGEdge<Component,Dependency>> edges,
152                          SetMultimap<DAGNode<Component, Dependency>, DAGEdge<Component, Dependency>> backEdges) {
153             this.edges = edges;
154             this.backEdges = backEdges;
155         }
156 
157         @Nullable
158         @Override
159         public Instantiator apply(@Nullable Desire input) {
160             for (DAGEdge<Component,Dependency> edge: edges) {
161                 if (edge.getLabel().getInitialDesire().equals(input)) {
162                     return makeInstantiator(edge.getTail(), backEdges);
163                 }
164             }
165             return null;
166         }
167     }
