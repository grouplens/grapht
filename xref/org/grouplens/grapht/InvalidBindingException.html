19  package org.grouplens.grapht;
20  
21  import javax.inject.Qualifier;
22  
23  /**
24   * Thrown when a binding configuration is invalid, which often occurs when an
25   * implementation type is bound to a type that it is not a subclass of, or when
26   * an annotation is intended to be used as a qualifier but has not been
27   * annotated with {@link Qualifier}.
28   * 
29   * @author <a href="http://grouplens.org">GroupLens Research</a>
30   */
31  public class InvalidBindingException extends RuntimeException {
32      private static final long serialVersionUID = 1L;
33  
34      private final Class<?> type;
35      
36      public InvalidBindingException(Class<?> type) {
37          this(type, "");
38      }
39      
40      public InvalidBindingException(Class<?> type, String message) {
41          this(type, message, null);
42      }
43      
44      public InvalidBindingException(Class<?> type, Throwable t) {
45          this(type, "", t);
46      }
47      
48      public InvalidBindingException(Class<?> type, String message, Throwable t) {
49          super(message, t);
50          this.type = type;
51      }
52      
53      /**
54       * @return The type that is configured incorrectly, or is the cause of
55       *         configuration errors
56       */
57      public Class<?> getType() {
58          return type;
59      }
60      
61      @Override
62      public String getMessage() {
63          return String.format("Error configuring %s: %s", type, super.getMessage());
64      }
65  }
