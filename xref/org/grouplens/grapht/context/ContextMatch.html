19  package org.grouplens.grapht.context;
20  
21  import com.google.common.collect.ImmutableList;
22  import com.google.common.collect.Ordering;
23  
24  import java.util.List;
25  
26  /**
27   * Interface for context matches. A context match is the result of successfully
28   * matching a {@link ContextMatcher}.
29   *
30   * @author <a href="http://grouplens.org">GroupLens Research</a>
31   */
32  public class ContextMatch implements Comparable<ContextMatch> {
33      private final ImmutableList<MatchElement> matchElements;
34  
35      private ContextMatch(List<MatchElement> matches) {
36          matchElements = ImmutableList.copyOf(matches);
37      }
38  
39      /**
40       * Create a new context match.
41       * @param matches The list of match elements.
42       * @return The context match.
43       */
44      static ContextMatch create(List<MatchElement> matches) {
45          return new ContextMatch(matches);
46      }
47  
48      @Override
49      public int compareTo(ContextMatch o) {
50          return Ordering.from(MatchElement.Order.PRIORITY_ONLY)
51                         .lexicographical()
52                         .compound(Ordering.from(MatchElement.Order.PRIORITY_AND_DISTANCE)
53                                           .lexicographical())
54                         .compare(matchElements.reverse(), o.matchElements.reverse());
55      }
56  
57      @Override
58      public boolean equals(Object o) {
59          if (o == this) {
60              return true;
61          } else if (o instanceof ContextMatch) {
62              ContextMatch om = (ContextMatch) o;
63              return matchElements.equals(om.matchElements);
64          } else {
65              return false;
66          }
67      }
68  
69      @Override
70      public int hashCode() {
71          return matchElements.hashCode();
72      }
73  
74      @Override
75      public String toString() {
76          return "ContextMatch" + matchElements.toString();
77      }
78  }
