19  package org.grouplens.grapht.context;
20  
21  import com.google.common.collect.Ordering;
22  
23  import javax.annotation.Nullable;
24  import java.util.Comparator;
25  
26  /**
27   * A single element in a context match.  Used in {@link ContextMatch}.
28   *
29   * @since 0.7
30   * @author <a href="http://www.grouplens.org">GroupLens Research</a>
31   */
32  public interface MatchElement {
33      /**
34       * Get the priority of this element matcher.
35       *
36       * @return The element matcher's priority.
37       */
38      ContextElements.MatchPriority getPriority();
39  
40      /**
41       * Get the type distance of this match.
42       *
43       * @return The type distance in this match, or empty if the type distance is irrelevant
44       *         for this type of matcher.
45       */
46      @Nullable
47      Integer getTypeDistance();
48  
49      /**
50       * Orderings for the match order.
51       */
52      public static enum Order implements Comparator<MatchElement> {
53          /**
54           * Priority-only ordering, used for first-pass comparison of context matchers.
55           */
56          PRIORITY_ONLY {
57              @Override
58              public int compare(MatchElement e1, MatchElement e2) {
59                  return e1.getPriority().compareTo(e2.getPriority());
60              }
61          },
62          /**
63           * Full ordering of match elements, including type distance.  Closer compares lower (higher
64           * priority).
65           */
66          PRIORITY_AND_DISTANCE {
67              @Override
68              public int compare(MatchElement e1, MatchElement e2) {
69                  int cmp = e1.getPriority().compareTo(e2.getPriority());
70                  if (cmp == 0) {
71                      Ordering<Integer> distOrder = Ordering.<Integer>natural().nullsLast();
72                      cmp = distOrder.compare(e1.getTypeDistance(), e2.getTypeDistance());
73                  }
74                  return cmp;
75              }
76          }
77      }
78  }
