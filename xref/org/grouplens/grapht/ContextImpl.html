19  package org.grouplens.grapht;
20  
21  import org.grouplens.grapht.context.ContextElementMatcher;
22  import org.grouplens.grapht.context.ContextElements;
23  import org.grouplens.grapht.context.ContextPattern;
24  import org.grouplens.grapht.reflect.QualifierMatcher;
25  import org.grouplens.grapht.context.Multiplicity;
26  import org.grouplens.grapht.reflect.Qualifiers;
27  
28  import javax.annotation.Nullable;
29  import java.lang.annotation.Annotation;
30  
31  /**
32   * ContextImpl is the basic implementation of Context.
33   * 
34   * @author <a href="http://grouplens.org">GroupLens Research</a>
35   */
36  class ContextImpl extends AbstractContext {
37      private final BindingFunctionBuilder config;
38      private final ContextPattern pattern;
39      private final boolean anchored;
40  
41      /**
42       * Construct a new context implementation.
43       * @param config The function builder.
44       * @param context The context pattern.
45       * @param anchored Whether this is anchored.  If it is not anchored, {@code .*} will be appended
46       *                 to the context pattern for any calls other than to {@link #matching(ContextPattern)}.
47       */
48      private ContextImpl(BindingFunctionBuilder config, ContextPattern context, boolean anchored) {
49          this.config = config;
50          this.pattern = context;
51          this.anchored = anchored;
52      }
53  
54      public static ContextImpl root(BindingFunctionBuilder config) {
55          return new ContextImpl(config, ContextPattern.empty(), false);
56      }
57      
58      public BindingFunctionBuilder getBuilder() {
59          return config;
60      }
61      
62      /**
63       * Get the context pattern for this builder.  It will never return {@code null}; for the root
64       * context, it will return {@link ContextPattern#any()}.
65       *
66       * @return The context pattern for this builder.
67       */
68      public ContextPattern getContextPattern() {
69          if (anchored) {
70              return pattern;
71          } else {
72              return pattern.appendDotStar();
73          }
74      }
75      
76      @Override
77      public <T> Binding<T> bind(Class<T> type) {
78          return new BindingImpl<T>(this, type);
79      }
80  
81      @Override
82      public Context within(Class<?> type) {
83          return in(Qualifiers.matchDefault(), type, false);
84      }
85  
86      @Override
87      public Context within(@Nullable Class<? extends Annotation> qualifier, Class<?> type) {
88          return in(Qualifiers.match(qualifier), type, false);
89      }
90      
91      @Override
92      public Context within(@Nullable Annotation annot, Class<?> type) {
93          return in(Qualifiers.match(annot), type, false);
94      }
95  
96      @Override
97      public Context matching(ContextPattern pat) {
98          return new ContextImpl(config, pattern.append(pat), true);
99      }
100 
101     @Override
102     public Context at(Class<?> type) {
103         return in(Qualifiers.matchDefault(), type, true);
104     }
105 
106     @Override
107     public Context at(@Nullable Class<? extends Annotation> qualifier, Class<?> type) {
108         return in(Qualifiers.match(qualifier), type, true);
109     }
110 
111     @Override
112     public Context at(@Nullable Annotation annot, Class<?> type) {
113         return in(Qualifiers.match(annot), type, true);
114     }
115     
116     private Context in(QualifierMatcher q, Class<?> type, boolean anchored) {
117         ContextElementMatcher nextMatcher = ContextElements.matchType(type, q);
118         ContextPattern pat = getContextPattern().append(nextMatcher, Multiplicity.ONE);
119 
120         return new ContextImpl(config, pat, anchored);
121     }
122 }
