19  package org.grouplens.grapht.annotation;
20  
21  import javax.annotation.processing.AbstractProcessor;
22  import javax.annotation.processing.Messager;
23  import javax.annotation.processing.RoundEnvironment;
24  import javax.inject.Qualifier;
25  import javax.lang.model.SourceVersion;
26  import javax.lang.model.element.Element;
27  import javax.lang.model.element.TypeElement;
28  import javax.tools.Diagnostic;
29  import java.lang.annotation.Documented;
30  import java.lang.annotation.Retention;
31  import java.lang.annotation.RetentionPolicy;
32  import java.util.HashSet;
33  import java.util.Set;
34  
35  /**
36   * Annotation processor that checks and validates DI annotations.
37   */
38  public class AnnotationValidator extends AbstractProcessor {
39      @Override
40      public SourceVersion getSupportedSourceVersion() {
41          // support version 6 or 7
42          // we can't compile against RELEASE_7 and maintain Java 6 compatibility, but the
43          // processor is Java 7-compatible. We have not tested against Java 8, however.
44          SourceVersion[] versions = SourceVersion.values();
45          SourceVersion v6 = SourceVersion.RELEASE_6;
46          assert v6.ordinal() < versions.length;
47          // we support up through Java 8
48          return versions[Math.min(v6.ordinal() + 2, versions.length - 1)];
49      }
50  
51      @Override
52      public Set<String> getSupportedAnnotationTypes() {
53          Set<String> atypes = new HashSet<String>();
54          atypes.add(Qualifier.class.getName());
55          atypes.add(Attribute.class.getName());
56          return atypes;
57      }
58  
59      private void warning(Element e, String fmt, Object... args) {
60          Messager log = processingEnv.getMessager();
61          String msg = String.format(fmt, args);
62          log.printMessage(Diagnostic.Kind.MANDATORY_WARNING, msg, e);
63      }
64  
65      private void error(Element e, String fmt, Object... args) {
66          Messager log = processingEnv.getMessager();
67          String msg = String.format(fmt, args);
68          log.printMessage(Diagnostic.Kind.ERROR, msg, e);
69      }
70  
71      @Override
72      public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
73          analyzeQualifiers(roundEnv);
74          analyzeAttributes(roundEnv);
75          return false; // let other processors work too
76      }
77  
78      private void analyzeQualifiers(RoundEnvironment re) {
79          Set<? extends Element> elts = re.getElementsAnnotatedWith(Qualifier.class);
80          for (Element elt: elts) {
81              if (elt.getAnnotation(Documented.class) == null) {
82                  warning(elt, "qualifier annotation should be @Documented");
83              }
84              Retention ret = elt.getAnnotation(Retention.class);
85              if (ret == null || !ret.value().equals(RetentionPolicy.RUNTIME)) {
86                  error(elt, "qualifer annotation must have RUNTIME retention");
87              }
88          }
89      }
90  
91      private void analyzeAttributes(RoundEnvironment re) {
92          Set<? extends Element> elts = re.getElementsAnnotatedWith(Qualifier.class);
93          for (Element elt: elts) {
94              if (elt.getAnnotation(Documented.class) == null) {
95                  warning(elt, "attribute annotation should be @Documented");
96              }
97              Retention ret = elt.getAnnotation(Retention.class);
98              if (ret == null || !ret.value().equals(RetentionPolicy.RUNTIME)) {
99                  error(elt, "attribute annotation must have RUNTIME retention");
100             }
101         }
102     }
