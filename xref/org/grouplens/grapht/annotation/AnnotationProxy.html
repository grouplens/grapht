33  /**
34   * Proxy used to implement annotation interfaces.  It implements the {@link Annotation}
35   * contract by delegating to a map of named attribute values.  A new AnnotationProxy instance
36   * should be created for each proxy annotation.
37   *
38   * @see AnnotationBuilder
39   */
40  class AnnotationProxy<T extends Annotation> implements InvocationHandler, Serializable {
41      private static final long serialVersionUID = 1L;
42      private final ClassProxy annotationType;
43      private final Map<String, Object> attributes;
44      private transient Class<T> cachedType;
45  
46      public AnnotationProxy(Class<T> type, Map<String, Object> attrs) {
47          annotationType = ClassProxy.of(type);
48          cachedType = type;
49          attributes = ImmutableMap.copyOf(attrs);
50      }
51  
52      /**
53       * Customized {@code readObject} implementation to ensure the cached type is resolved.
54       *
55       * @param in The stream.
56       * @throws java.io.ObjectStreamException If there is an error reading the object from the stream.
57       */
58      @SuppressWarnings("unchecked")
59      private void readObject(ObjectInputStream in) throws ObjectStreamException {
60          try {
61              in.defaultReadObject();
62              cachedType = (Class<T>) annotationType.resolve();
63          } catch (IOException e) {
64              ObjectStreamException ex = new StreamCorruptedException("IO exception");
65              ex.initCause(e);
66              throw ex;
67          } catch (ClassNotFoundException e) {
68              ObjectStreamException ex = new InvalidObjectException("IO exception");
69              ex.initCause(e);
70              throw ex;
71          }
72      }
73  
74      @Override
75      public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
76          if (isHashCode(method)) {
77              return proxyHashCode(proxy);
78          } else if (isEquals(method)) {
79              return proxyEquals(proxy, args[0]);
80          } else if (isAnnotationType(method)) {
81              return proxyAnnotationType();
82          } else if (isToString(method)) {
83              return proxyToString(proxy);
84          } else if (attributes.containsKey(method.getName()) && method.getParameterTypes().length == 0) {
85              return copyAnnotationValue(attributes.get(method.getName()));
86          } else {
87              // fall back to the default
88              return copyAnnotationValue(method.getDefaultValue());
89          }
90          // wait() and other Object methods do not get sent to the InvocationHandler
91          // so we don't have any other cases
92      }
93  
94      private boolean isEquals(Method m) {
95          return m.getName().equals("equals") && m.getReturnType().equals(boolean.class)
96              && m.getParameterTypes().length == 1 && m.getParameterTypes()[0].equals(Object.class);
97      }
98  
99      private boolean isHashCode(Method m) {
100         return m.getName().equals("hashCode") && m.getReturnType().equals(int.class)
101             && m.getParameterTypes().length == 0;
102     }
103 
104     private boolean isAnnotationType(Method m) {
105         return m.getName().equals("annotationType") && m.getReturnType().equals(Class.class)
106             && m.getParameterTypes().length == 0;
107     }
108 
109     private boolean isToString(Method m) {
110         return m.getName().equals("toString") && m.getReturnType().equals(String.class)
111             && m.getParameterTypes().length == 0;
112     }
113 
114     private Class<? extends Annotation> proxyAnnotationType() {
115         return cachedType;
116     }
117 
118     private String proxyToString(Object o) {
119         return AnnotationUtils.toString((Annotation) o);
120     }
121 
122     private int proxyHashCode(Object proxy) {
123         return AnnotationUtils.hashCode((Annotation) proxy);
124     }
125 
126     private boolean proxyEquals(Object o1, Object o2) {
127         return AnnotationUtils.equals((Annotation) o1, (Annotation) o2);
128     }
129 
130     /**
131      * Safe clone of an object.  If the object is an array, it is copied; otherwise, it is
132      * returned as-is.  This object is only applicable to valid annotation value types, which
133      * are all either arrays or immutable.
134      * @param o The annotation value.
135      * @return A copy of the value.
136      */
137     @SuppressWarnings("unchecked")
138     static Object copyAnnotationValue(Object o) {
139         if (o.getClass().isArray()) {
140             // make a shallow copy of the array
141             if (o instanceof boolean[]) {
142                 boolean[] a = (boolean[]) o;
143                 return Arrays.copyOf(a, a.length);
144             } else if (o instanceof byte[]) {
145                 byte[] a = (byte[]) o;
146                 return Arrays.copyOf(a, a.length);
147             } else if (o instanceof short[]) {
148                 short[] a = (short[]) o;
149                 return Arrays.copyOf(a, a.length);
150             } else if (o instanceof int[]) {
151                 int[] a = (int[]) o;
152                 return Arrays.copyOf(a, a.length);
153             } else if (o instanceof long[]) {
154                 long[] a = (long[]) o;
155                 return Arrays.copyOf(a, a.length);
156             } else if (o instanceof char[]) {
157                 char[] a = (char[]) o;
158                 return Arrays.copyOf(a, a.length);
159             } else if (o instanceof float[]) {
160                 float[] a = (float[]) o;
161                 return Arrays.copyOf(a, a.length);
162             } else if (o instanceof double[]) {
163                 double[] a = (double[]) o;
164                 return Arrays.copyOf(a, a.length);
165             } else {
166                 Object[] a = (Object[]) o;
167                 return Arrays.copyOf(a, a.length, (Class<? extends Object[]>) o.getClass());
168             }
169         } else if (o instanceof String
170                    || o instanceof Annotation
171                    || ClassUtils.isPrimitiveOrWrapper(o.getClass())) {
172             // the value is immutable and a copy is not necessary
173             return o;
174         } else {
175             throw new IllegalArgumentException("not an annotation value");
176         }
177     }
