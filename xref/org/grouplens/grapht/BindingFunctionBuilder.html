21  import com.google.common.collect.ArrayListMultimap;
22  import com.google.common.collect.Multimap;
23  import org.grouplens.grapht.context.ContextMatcher;
24  import org.grouplens.grapht.solver.BindRule;
25  import org.grouplens.grapht.solver.BindingFunction;
26  import org.grouplens.grapht.solver.RuleBasedBindingFunction;
27  
28  import java.io.Externalizable;
29  import java.io.Serializable;
30  import java.util.Collections;
31  import java.util.HashSet;
32  import java.util.Set;

34  /**
35   * BindingFunctionBuilder provides a convenient access to the fluent API and
36   * converts calls to {@link Context} and {@link Binding} methods into multiple
37   * {@link BindingFunction BindingFunctions}.
38   * 
39   * @author <a href="http://grouplens.org">GroupLens Research</a>
40   */
41  public class BindingFunctionBuilder implements Cloneable {
42      /**
43       * BindingFunctionBuilder generates three binding functions at separate
44       * priorities.
45       */
46      public static enum RuleSet {
47          /**
48           * Rule set for the explicitly configured rules with the fluent API.
49           */
50          EXPLICIT,
51          /**
52           * Rule set for the intermediate types between a source type (
53           * {@link Context#bind(Class)}) and the target type (
54           * {@link Binding#to(Class)})
55           */
56          INTERMEDIATE_TYPES,
57          /**
58           * Rule set for the super types of the source types of bindings (e.g.
59           * {@link Context#bind(Class)})
60           */
61          SUPER_TYPES
62      }
63      
64      private final Context root;
65      
66      private final Set<Class<?>> defaultExcludes;
67      private final boolean generateRules;
68      
69      private final Multimap<ContextMatcher,BindRule> manualRules;
70      private final Multimap<ContextMatcher,BindRule> intermediateRules; // "generated"
71      private final Multimap<ContextMatcher,BindRule> superRules; // "generated"
72  
73      /**
74       * Create a new InjectorConfigurationBuilder that automatically generates bind rules for
75       * super and intermediate types.
76       */
77      public BindingFunctionBuilder() {
78          this(true);
79      }
80  
81      /**
82       * Create a new InjectorConfigurationBuilder. If <tt>generateRules</tt> is true, bind
83       * rules for super and intermediate types are generated. If it is false,
84       * only one bind rule is created per binding.
85       * 
86       * @param generateRules True if additional bind rules should be generated
87       */
88      public BindingFunctionBuilder(boolean generateRules) {
89          this.generateRules = generateRules;
90  
91          defaultExcludes = new HashSet<Class<?>>();
92          defaultExcludes.add(Object.class);
93          defaultExcludes.add(Comparable.class);
94          defaultExcludes.add(Serializable.class);
95          defaultExcludes.add(Externalizable.class);
96          defaultExcludes.add(Cloneable.class);
97  
98          manualRules = ArrayListMultimap.create();
99          intermediateRules = ArrayListMultimap.create();
100         superRules = ArrayListMultimap.create();
101 
102         root = ContextImpl.root(this);
103     }
104     
105     private BindingFunctionBuilder(BindingFunctionBuilder clone) {
106         generateRules = clone.generateRules;
107         defaultExcludes = new HashSet<Class<?>>(clone.defaultExcludes);
108         manualRules = ArrayListMultimap.create(clone.manualRules);
109         intermediateRules = ArrayListMultimap.create(clone.intermediateRules);
110         superRules = ArrayListMultimap.create(clone.superRules);
111         root = ContextImpl.root(this);
112     }
113     
114     @Override
115     public BindingFunctionBuilder clone() {
116         return new BindingFunctionBuilder(this);
117     }
118     
119     /**
120      * @return True if bind rules for super and intermediate types should be
121      *         generated
122      */
123     public boolean getGenerateRules() {
124         return generateRules;
125     }
126     
127     /**
128      * @return The root context managed by this builder
129      */
130     public Context getRootContext() {
131         return root;
132     }
133 
134     /**
135      * Run the module's {@link Module#configure(Context) bind()} method on the root
136      * context of this builder.
137      * 
138      * @param module The module to apply
139      */
140     public void applyModule(Module module) {
141         module.configure(getRootContext());
142     }
143 
144     /**
145      * Add a type to be excluded from when generating bind rules. This does not
146      * invalidate bindings that bind directly to this type.
147      * 
148      * @param type The type to exclude
149      * @throws NullPointerException if type is null
150      */
151     public void addDefaultExclusion(Class<?> type) {
152         if (type == null) {
153             throw new NullPointerException("Exclusion type cannot be null");
154         }
155         defaultExcludes.add(type);
156     }
157 
158     /**
159      * Remove a type that is currently being excluded.
160      * 
161      * @see #addDefaultExclusion(Class)
162      * @param type The type that should no longer be excluded
163      * @throws NullPointerException if type is null
164      */
165     public void removeDefaultExclusion(Class<?> type) {
166         if (type == null) {
167             throw new NullPointerException("Exclusion type cannot be null");
168         }
169         defaultExcludes.remove(type);
170     }
171     
172     /**
173      * Return the built BindingFunction for the given RuleSet.
174      * 
175      * @param set
176      * @return
177      */
178     public BindingFunction build(RuleSet set) {
179         return new RuleBasedBindingFunction(getMap(set));
180     }
181     
182     void addBindRule(RuleSet set, ContextMatcher context, BindRule rule) {
183         Multimap<ContextMatcher, BindRule> map = getMap(set);
184         map.put(context, rule);
185     }
186 
187     Set<Class<?>> getDefaultExclusions() {
188         return Collections.unmodifiableSet(defaultExcludes);
189     }
190 
191     private Multimap<ContextMatcher, BindRule> getMap(RuleSet set) {
192         switch(set) {
193         case EXPLICIT:
194             return manualRules;
195         case INTERMEDIATE_TYPES:
196             return intermediateRules;
197         case SUPER_TYPES:
198             return superRules;
199         default:
200             throw new RuntimeException("Should not happen");
201         }
202     }
