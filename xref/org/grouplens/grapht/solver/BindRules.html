19  package org.grouplens.grapht.solver;

21  import org.grouplens.grapht.CachePolicy;
22  import org.grouplens.grapht.reflect.QualifierMatcher;
23  import org.grouplens.grapht.reflect.Satisfaction;
24  
25  /**
26   * Utility methods for {@link BindRule}.
27   */
28  public final class BindRules {
29      private BindRules() {}
30  
31      /**
32       * Construct a new bind rule that binds a dependency to a satisfaction.
33       *
34       *
35       * @param depType      The dependency type.
36       * @param qualifier    The qualifier matcher to match the dependency type.
37       * @param satisfaction The satisfaction.
38       * @param policy       The cache policy.
39       * @param terminal     Whether this binding is terminal.
40       * @return A new bind rule.
41       * @deprecated Use {@link BindRuleBuilder}.
42       */
43      @Deprecated
44      public static BindRule toSatisfaction(Class<?> depType, QualifierMatcher qualifier,
45                                            Satisfaction satisfaction, CachePolicy policy,
46                                            boolean terminal) {
47          return BindRuleBuilder.create()
48                  .setDependencyType(depType)
49                  .setQualifierMatcher(qualifier)
50                  .setSatisfaction(satisfaction)
51                  .setCachePolicy(policy)
52                  .setTerminal(terminal)
53                  .build();
54      }
55  
56      /**
57       * Construct a new bind rule that binds a dependency to a class.
58       *
59       *
60       * @param depType   The dependency type.
61       * @param qualifier The qualifier matcher to match the dependency type.
62       * @param implType  The implementation type to use.
63       * @param policy    The cache policy.
64       * @param terminal  Whether this binding is terminal.
65       * @return A new bind rule.
66       * @deprecated Use {@link BindRuleBuilder}.
67       */
68      @Deprecated
69      public static BindRule toClass(Class<?> depType, QualifierMatcher qualifier,
70                                     Class<?> implType, CachePolicy policy,
71                                     boolean terminal) {
72          return BindRuleBuilder.create()
73                                .setDependencyType(depType)
74                                .setQualifierMatcher(qualifier)
75                                .setImplementation(implType)
76                                .setCachePolicy(policy)
77                                .setTerminal(terminal)
78                                .build();
79      }
