42  /**
43   * <p>
44   * DependencySolver is a utility for resolving Desires into a dependency graph,
45   * where nodes are shared when permitted by a Satisfaction's dependency
46   * configuration. It supports qualified and context-aware injection, and
47   * just-in-time injection if the type has an injectable constructor.
48   * <p>
49   * The conceptual binding function used by this solver is represented as a list
50   * of prioritized {@link BindingFunction functions}. Functions at the start of
51   * the list are used first, which makes it easy to provide custom functions that
52   * override default behaviors.
53   * <p>
54   * This solver does not support cyclic dependencies because of the possibility
55   * that a context later on might activate a bind rule that breaks the cycle. To
56   * ensure termination, it has a maximum context depth that is configurable.
57   * 
58   * @see DefaultInjector
59   * @author <a href="http://grouplens.org">GroupLens Research</a>
60   */
61  public class DependencySolver {
62      private static final Logger logger = LoggerFactory.getLogger(DependencySolver.class);
63      public static final Component ROOT_SATISFACTION =
64              Component.create(new NullSatisfaction(Void.TYPE), CachePolicy.NO_PREFERENCE);

92      /**
93       * Create a DependencySolver that uses the given functions, and max
94       * depth of the dependency graph.
95       * 
96       * @param bindFunctions The binding functions that control desire bindings
97       * @param maxDepth A maximum depth of the graph before it's determined that
98       *            a cycle exists
99       * @throws IllegalArgumentException if maxDepth is less than 1
100      * @throws NullPointerException if bindFunctions is null
101      */
102     DependencySolver(List<BindingFunction> bindFunctions,
103                      List<BindingFunction> triggers,
104                      CachePolicy defaultPolicy, int maxDepth) {
105         Preconditions.notNull("bindFunctions", bindFunctions);
106         Preconditions.notNull("defaultPolicy", defaultPolicy);
107         if (maxDepth <= 0) {
108             throw new IllegalArgumentException("Max depth must be at least 1");
109         }
110         
111         this.functions = new ArrayList<BindingFunction>(bindFunctions);
112         this.triggerFunctions = new ArrayList<BindingFunction>(triggers);
113         this.maxDepth = maxDepth;
114         this.defaultPolicy = defaultPolicy;
115         
116         graph = DAGNode.singleton(ROOT_SATISFACTION);
117         backEdges = HashMultimap.create();
118         mergePool = MergePool.create();
119 
120         logger.info("DependencySolver created, max depth: {}", maxDepth);
121     }

155      * Get the back edge for a particular node and desire, if one exists.
156      * @return The back edge, or {@code null} if no edge exists.
157      * @see #getBackEdges()
158      */
159     public synchronized DAGNode<Component, Dependency> getBackEdge(DAGNode<Component, Dependency> parent,
160                                                                             Desire desire) {
161         Predicate<DAGEdge<?, Dependency>> pred = DAGEdge.labelMatches(Dependency.hasInitialDesire(desire));
162         return FluentIterable.from(backEdges.get(parent))
163                              .filter(pred)
164                              .first()
165                              .transform(DAGEdge.<Component, Dependency>extractTail())
166                              .orNull();

178     /**
179      * Update the dependency graph to include the given desire. An edge from the
180      * root node to the desire's resolved satisfaction will exist after this is
181      * finished.
182      * 
183      * @param desire The desire to include in the graph
184      */
185     public synchronized void resolve(Desire desire) throws ResolutionException {
186         logger.info("Resolving desire: {}", desire);
187 
188         Queue<Deferral> deferralQueue = new ArrayDeque<Deferral>();
189 
190         // before any deferred nodes are processed, we use a synthetic root
191         // and null original desire since nothing produced this root
192         deferralQueue.add(new Deferral(rootNode(), initialContext()));
193 
194         while(!deferralQueue.isEmpty()) {
195             Deferral current = deferralQueue.poll();
196             DAGNode<Component, Dependency> parent = current.node;
197             // deferred nodes are either root - depless - or having deferred dependencies
198             assert parent.getOutgoingEdges().isEmpty();
199 
200             if (current.node.getLabel().equals(ROOT_SATISFACTION)) {
201                 Pair<DAGNode<Component, Dependency>, Dependency> rootNode =
202                         resolveFully(desire, current.context, deferralQueue);
203                 // add this to the global graph
204                 graph = DAGNode.copyBuilder(graph)
205                                .addEdge(mergePool.merge(rootNode.getLeft()),
206                                         rootNode.getRight())
207                                .build();
208             } else if (graph.getReachableNodes().contains(parent)) {
209                 // the node needs to be re-scanned.  This means that it was not consolidated by
210                 // a previous merge operation.  This branch only arises with provider injection.
211                 Satisfaction sat = parent.getLabel().getSatisfaction();
212                 for (Desire d: sat.getDependencies()) {
213                     logger.debug("Attempting to resolve deferred dependency {} of {}", d, sat);
214                     // resolve the dependency
215                     Pair<DAGNode<Component, Dependency>, Dependency> result =
216                             resolveFully(d, current.context, deferralQueue);
217                     // merge it in
218                     DAGNode<Component, Dependency> merged = mergePool.merge(result.getLeft());
219                     // now see if there's a real cycle
220                     if (merged.getReachableNodes().contains(parent)) {
221                         // parent node is referenced from merged, we have a circle!
222                         // that means we need a back edge
223                         backEdges.put(parent, DAGEdge.create(parent, merged, result.getRight()));
224                     } else {
225                         // an edge from parent to merged does not add a cycle
226                         // we have to update graph right away so it's available to merge the next
227                         // dependency
228                         DAGNode<Component, Dependency> newP =
229                                 DAGNode.copyBuilder(parent)
230                                        .addEdge(merged, result.getRight())
231                                        .build();
232                         replaceNode(parent, newP);
233                         parent = newP;
234                     }
235                 }
236             } else {
237                 // node unreachable - it's a leftover or unneeded deferral
238                 logger.debug("node {} not in graph, ignoring", parent);
239             }
240         }
241     }

243     private void replaceNode(DAGNode<Component,Dependency> old,
244                              DAGNode<Component,Dependency> repl) {
245         Map<DAGNode<Component,Dependency>,
246                 DAGNode<Component,Dependency>> memory = Maps.newHashMap();
247         graph = graph.replaceNode(old, repl, memory);
248 
249         // loop over a snapshot of the list, replacing nodes
250         Collection<DAGEdge<Component, Dependency>> oldBackEdges = backEdges.values();
251         backEdges = HashMultimap.create();
252         for (DAGEdge<Component,Dependency> edge: oldBackEdges) {
253             DAGNode<Component,Dependency> newHead, newTail;
254             newHead = memory.get(edge.getHead());
255             if (newHead == null) {
256                 newHead = edge.getHead();
257             }
258             newTail = memory.get(edge.getTail());
259             if (newTail == null) {
260                 newTail = edge.getTail();
261             }
262             DAGEdge<Component,Dependency> newEdge;
263             if (newHead.equals(edge.getHead()) && newTail.equals(edge.getTail())) {
264                 newEdge = edge;
265             } else {
266                 newEdge = DAGEdge.create(newHead, newTail, edge.getLabel());
267             }
268             backEdges.put(newHead, newEdge);
269         }
270     }

272     /**
273      * Rewrite a dependency graph using the rules in this solver.  The accumulated global graph and
274      * back edges are ignored and not modified.
275      * <p>Graph rewrite walks the graph, looking for nodes to rewrite.  If the desire that leads
276      * to a node is matched by a trigger binding function, then it is resolved using the binding
277      * functions and replaced with the resulting (merged) node.  Rewriting proceeds from the root
278      * down, but does not consider the children of nodes generated by the rewriting process.</p>
279      *
280      * @param graph The graph to rewrite.
281      * @return A rewritten version of the graph.
282      */
283     public DAGNode<Component,Dependency> rewrite(DAGNode<Component,Dependency> graph) throws ResolutionException {
284         if (!graph.getLabel().getSatisfaction().getErasedType().equals(Void.TYPE)) {
285             throw new IllegalArgumentException("only full dependency graphs can be rewritten");
286         }
287 
288         logger.debug("rewriting graph with {} nodes", graph.getReachableNodes().size());
289         // We proceed in three stages.
290         Map<DAGEdge<Component, Dependency>, DAGEdge<Component,Dependency>> replacementSubtrees =
291                 Maps.newHashMap();
292         walkGraphForReplacements(graph,
293                                  InjectionContext.singleton(graph.getLabel().getSatisfaction()),
294                                  replacementSubtrees);
295 
296         DAGNode<Component, Dependency> stage2 =
297                 graph.transformEdges(Functions.forMap(replacementSubtrees, null));
298 
299         logger.debug("merging rewritten graph");
300         // Now we have a graph (stage2) with rewritten subtrees based on trigger rules
301         // We merge this graph with the original to deduplicate.
302         MergePool<Component,Dependency> pool = MergePool.create();
303         pool.merge(graph);
304         return pool.merge(stage2);
305     }

307     /**
308      * Walk the graph, looking for replacements.
309      * @param root The node to walk.
310      * @param context The context leading to this node.
311      * @param replacements The map of replacements to build. This maps edges to their replacement
312      *                     targets and labels.
313      * @throws ResolutionException If there is a resolution error rewriting the graph.
314      */
315     private void walkGraphForReplacements(DAGNode<Component, Dependency> root,
316                                           InjectionContext context,
317                                           Map<DAGEdge<Component, Dependency>, DAGEdge<Component, Dependency>> replacements) throws ResolutionException {
318         assert context.getTailValue().getLeft().equals(root.getLabel().getSatisfaction());
319         for (DAGEdge<Component, Dependency> edge: root.getOutgoingEdges()) {
320             logger.debug("considering {} for replacement", edge.getTail().getLabel());
321             Desire desire = edge.getLabel().getDesireChain().getInitialDesire();
322             DesireChain chain = DesireChain.singleton(desire);
323             Pair<DAGNode<Component, Dependency>, Dependency> repl = null;
324             if (!edge.getLabel().isFixed()) {
325                 for (BindingFunction bf: triggerFunctions) {
326                     BindingResult result = bf.bind(context, chain);
327                     if (result != null) {
328                         // resolve the node
329                         // we could reuse the resolution, but perf savings isn't worth complexity
330                         repl = resolveFully(desire, context, null);
331                         break;
332                     }
333                 }
334             } else {
335                 logger.debug("{} is fixed, skipping", edge.getTail().getLabel());
336             }
337             if (repl == null) {
338                 // no trigger bindings, walk the node's children
339                 InjectionContext next = context.extend(edge.getTail()
340                                                            .getLabel()
341                                                            .getSatisfaction(),
342                                                        edge.getLabel()
343                                                            .getDesireChain()
344                                                            .getInitialDesire()
345                                                            .getInjectionPoint());
346                 walkGraphForReplacements(edge.getTail(), next, replacements);
347             } else {
348                 // trigger binding, add a replacement
349                 logger.info("replacing {} with {}",
350                             edge.getTail().getLabel(),
351                             repl.getLeft().getLabel());
352                 replacements.put(edge, DAGEdge.create(root, repl.getLeft(), repl.getRight()));
353             }
354         }
355     }

357     /**
358      * Resolve a desire and its dependencies, inserting them into the graph.
359      *
360      * @param desire The desire to resolve.
361      * @param context The context of {@code parent}.
362      * @param deferQueue The queue of node deferrals.
363      * @throws ResolutionException if there is an error resolving the nodes.
364      */
365     private Pair<DAGNode<Component,Dependency>,Dependency>
366     resolveFully(Desire desire, InjectionContext context, Queue<Deferral> deferQueue) throws ResolutionException {
367         // check context depth against max to detect likely dependency cycles
368         if (context.size() > maxDepth) {
369             throw new CyclicDependencyException(desire, "Maximum context depth of " + maxDepth + " was reached");
370         }
371         
372         // resolve the current node
373         Resolution result = resolve(desire, context);
374 
375         InjectionContext newContext = context.extend(result.satisfaction, desire.getInjectionPoint());
376 
377         DAGNode<Component, Dependency> node;
378         if (result.deferDependencies) {
379             // extend node onto deferred queue and skip its dependencies for now
380             logger.debug("Deferring dependencies of {}", result.satisfaction);
381             node = DAGNode.singleton(result.makeSatisfaction());
382             deferQueue.add(new Deferral(node, newContext));
383         } else {
384             // build up a node with its outgoing edges
385             DAGNodeBuilder<Component,Dependency> nodeBuilder = DAGNode.newBuilder();
386             nodeBuilder.setLabel(result.makeSatisfaction());
387             for (Desire d: result.satisfaction.getDependencies()) {
388                 // complete the sub graph for the given desire
389                 // - the call to resolveFully() is responsible for adding the dependency edges
390                 //   so we don't need to process the returned node
391                 logger.debug("Attempting to satisfy dependency {} of {}", d, result.satisfaction);
392                 nodeBuilder.addEdge(resolveFully(d, newContext, deferQueue));
393             }
394             node = nodeBuilder.build();
395         }
396 
397         return Pair.of(node, result.makeDependency());
398     }

400     private Resolution resolve(Desire desire, InjectionContext context) throws ResolutionException {
401         DesireChain chain = DesireChain.singleton(desire);
402 
403         CachePolicy policy = CachePolicy.NO_PREFERENCE;
404         boolean fixed = false;
405         while(true) {
406             logger.debug("Current desire: {}", chain.getCurrentDesire());
407             
408             BindingResult binding = null;
409             for (BindingFunction bf: functions) {
410                 binding = bf.bind(context, chain);
411                 if (binding != null && !chain.getPreviousDesires().contains(binding.getDesire())) {
412                     // found a binding that hasn't been used before
413                     break;
414                 }
415             }
416             
417             boolean defer = false;
418             boolean terminate = true;
419             if (binding != null) {
420                 // update the desire chain
421                 chain = chain.extend(binding.getDesire());
422 
423                 terminate = binding.terminates();
424                 defer = binding.isDeferred();
425                 fixed |= binding.isFixed();
426                 
427                 // upgrade policy if needed
428                 if (binding.getCachePolicy().compareTo(policy) > 0) {
429                     policy = binding.getCachePolicy();
430                 }
431             }
432             
433             if (terminate && chain.getCurrentDesire().isInstantiable()) {
434                 logger.info("Satisfied {} with {}", desire, chain.getCurrentDesire().getSatisfaction());
435                 
436                 // update cache policy if a specific policy hasn't yet been selected
437                 if (policy.equals(CachePolicy.NO_PREFERENCE)) {
438                     policy = chain.getCurrentDesire().getSatisfaction().getDefaultCachePolicy();
439                     if (policy.equals(CachePolicy.NO_PREFERENCE)) {
440                         policy = defaultPolicy;
441                     }
442                 }
443                 
444                 return new Resolution(chain.getCurrentDesire().getSatisfaction(), policy, chain, fixed, defer);
445             } else if (binding == null) {
446                 // no more desires to process, it cannot be satisfied
447                 throw new UnresolvableDependencyException(chain, context);
448             }
449         }
450     }

452     /*
453      * Result tuple for resolve(Desire, InjectionContext)
454      */
455     private static class Resolution {
456         private final Satisfaction satisfaction;
457         private final CachePolicy policy;
458         private final DesireChain desires;
459         private final boolean fixed;
460         private final boolean deferDependencies;
461         
462         public Resolution(Satisfaction satisfaction, CachePolicy policy, 
463                           DesireChain desires, boolean fixed,
464                           boolean deferDependencies) {
465             this.satisfaction = satisfaction;
466             this.policy = policy;
467             this.desires = desires;
468             this.fixed = fixed;
469             this.deferDependencies = deferDependencies;
470         }
471 
472         public Component makeSatisfaction() {
473             return Component.create(satisfaction, policy);
474         }
475 
476         public Dependency makeDependency() {
477             EnumSet<Dependency.Flag> flags = Dependency.Flag.emptySet();
478             if (fixed) {
479                 flags.add(Dependency.Flag.FIXED);
480             }
481             return Dependency.create(desires, flags);
482         }
483 
484         @Override
485         public String toString() {
486             return "(" + satisfaction + ", " + policy + ")";
487         }
488     }
