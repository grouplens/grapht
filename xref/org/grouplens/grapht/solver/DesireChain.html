21  import com.google.common.base.Predicate;
22  import org.grouplens.grapht.reflect.Desire;
23  import org.grouplens.grapht.util.AbstractChain;
24  
25  import javax.annotation.Nonnull;
26  import javax.annotation.Nullable;
27  import java.util.Collections;
28  import java.util.List;
29  import java.util.UUID;

31  /**
32   * A sequence of desires.  When one desire is resolved, that resolution can be a desire that needs
33   * further resolution.  These desires are accumulated in a desire chain.  Desire chains are
34   * immutable; appending to one results in a new desire chain object pointing to the previous chain.
35   * They form a reverse singly linked list.  The chain maintains O(1) access to both the initial and
36   * current desires.
37   *
38   * <p>When iterating a desire chain the initial desire is first and the most recent desire is last.
39   *
40   * @since 0.7.0
41   * @author <a href="http://www.grouplens.org">GroupLens Research</a>
42   */
43  public class DesireChain extends AbstractChain<Desire> {
44      @Nonnull
45      private final Desire initialDesire;
46      private final UUID key;
47  
48      public static DesireChain singleton(Desire desire) {
49          return new DesireChain(null, desire);
50      }
51  
52      /**
53       * Create a new desire chain.
54       * @param prev The previous chain.
55       * @param d The desire.
56       */
57      private DesireChain(DesireChain prev, @Nonnull Desire d) {
58          super(prev, d);
59          key = prev == null ? UUID.randomUUID() : prev.key;
60          initialDesire = prev == null ? d : prev.getInitialDesire();
61      }
62  
63      public static Predicate<DesireChain> hasInitialDesire(final Desire d) {
64          return new Predicate<DesireChain>() {
65              @Override
66              public boolean apply(@Nullable DesireChain input) {
67                  return input != null && input.getInitialDesire().equals(d);
68              }
69          };
70      }
71  
72      @Nonnull
73      public Desire getCurrentDesire() {
74          return tailValue;
75      }
76  
77      @Nonnull
78      public Desire getInitialDesire() {
79          return initialDesire;
80      }
81  
82      /**
83       * Return the list of desires up to, but not including, the current desire.
84       * @return The previous desire chain.
85       */
86      @Nonnull
87      public List<Desire> getPreviousDesires() {
88          if (previous == null) {
89              return Collections.emptyList();
90          } else {
91              return previous;
92          }
93      }
94  
95      /**
96       * Get this chain's key. Each chain has a key, a unique object that is created when the chain
97       * is created (via {@link #singleton(org.grouplens.grapht.reflect.Desire)}), and preserved through
98       * {@link #extend(org.grouplens.grapht.reflect.Desire)} operations.  It can be used to remember
99       * state across invocations of a binding function as a desire chain is built up.
100      * @return The chain's key.
101      */
102     public Object getKey() {
103         return key;
104     }
105 
106     /**
107      * Extend this chain with a new desire. The chain is not modified; this method returns a new
108      * chain that includes the new desire as its current desire.
109      *
110      * @param d The new current desire.
111      * @return The new desire chain.
112      */
113     @Nonnull
114     public DesireChain extend(@Nonnull Desire d) {
115         return new DesireChain(this, d);
116     }
117 }
