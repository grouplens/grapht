30  import javax.annotation.concurrent.ThreadSafe;
31  import java.lang.annotation.Annotation;

33  /**
34   * <p>
35   * DefaultInjector is the default Injector implementation. When resolving the
36   * dependency graph for a desire, a "context" is built which consists of an
37   * ordering of qualified types that satisfy each dependency. The DefaultInjector
38   * uses the {@link DependencySolver} to manage dependency resolution. New
39   * injectors can easily be built to also use this solver.
40   * 
41   * @author <a href="http://grouplens.org">GroupLens Research</a>
42   */
43  @ThreadSafe
44  public class DefaultInjector implements Injector {
45      private static final Logger logger = LoggerFactory.getLogger(DefaultInjector.class);
46      
47      private final DependencySolver solver;
48      private final InjectionContainer instantiator;
49  
50      /**
51       * <p>
52       * Create a new DefaultInjector. The created resolver will use a max
53       * dependency depth of 100 to estimate if there are cycles in the dependency
54       * hierarchy. Bindings with a NO_PREFERENCE cache policy will be treated as
55       * NEW_INSTANCE.
56       * 
57       * @param functions The BindingFunctions to use, ordered with highest
58       *            priority function first
59       * @throws NullPointerException if spi or functions ar enull
60       */
61      public DefaultInjector(BindingFunction... functions) {
62          this(CachePolicy.MEMOIZE, functions);
63      }
64      
65      /**
66       * <p>
67       * Create a new DefaultInjector. The created resolver will use a max
68       * dependency depth of 100 to estimate if there are cycles in the dependency
69       * hierarchy. Bindings with a NO_PREFERENCE cache policy will use
70       * <tt>defaultPolicy</tt>.
71       * 
72       * @param defaultPolicy The CachePolicy used in place of NO_PREFERENCE
73       * @param functions The BindingFunctions to use, ordered with highest
74       *            priority functions first
75       * @throws IllegalArgumentException if defaultPolicy is NO_PREFERENCE
76       * @throws NullPointerException if spi or functions are null
77       */
78      public DefaultInjector(CachePolicy defaultPolicy, BindingFunction... functions) {
79          this(defaultPolicy, 100, functions);
80      }
81  
82      /**
83       * <p>
84       * Create a new DefaultInjector. <tt>maxDepth</tt> represents the maximum
85       * depth of the dependency hierarchy before it is assume that there is a
86       * cycle. Bindings with a NO_PREFERENCE cache policy will use
87       * <tt>defaultPolicy</tt>.
88       * <p>
89       * This constructor can be used to increase this depth in the event that
90       * configuration requires it, although for most purposes the default 100
91       * should be sufficient.
92       * 
93       * @param defaultPolicy The CachePolicy used in place of NO_PREFERENCE
94       * @param maxDepth The maximum depth of the dependency hierarchy
95       * @param functions The BindingFunctions to use, ordered with highest
96       *            priority functions first
97       * @throws IllegalArgumentException if maxDepth is less than 1, or if
98       *             defaultPolicy is NO_PREFERENCE
99       * @throws NullPointerException if spi or functions are null
100      */
101     public DefaultInjector(CachePolicy defaultPolicy, int maxDepth, BindingFunction... functions) {
102         if (defaultPolicy.equals(CachePolicy.NO_PREFERENCE)) {
103             throw new IllegalArgumentException("Default CachePolicy cannot be NO_PREFERENCE");
104         }
105 
106         solver = DependencySolver.newBuilder()
107                                  .addBindingFunctions(functions)
108                                  .setMaxDepth(maxDepth)
109                                  .build();
110         instantiator = InjectionContainer.create(defaultPolicy);
111     }
112     
113     /**
114      * @return The DependencySolver backing this injector
115      */
116     public DependencySolver getSolver() {
117         return solver;
118     }
119     
120     @Override
121     public <T> T getInstance(Class<T> type) throws InjectionException {
122         return getInstance(null, type);
123     }
124     
125     @Override
126     @SuppressWarnings("unchecked")
127     public <T> T getInstance(Annotation qualifier, Class<T> type) throws InjectionException {
128         // All Provider cache access, graph resolution, etc. occur
129         // within this exclusive lock so we know everything is thread safe
130         // albeit in a non-optimal way.
131         synchronized(this) {
132             Desire desire = Desires.create(qualifier, type, false);
133 
134             Predicate<Dependency> pred = Dependency.hasInitialDesire(desire);
135 
136             // check if the desire is already in the graph
137             DAGEdge<Component, Dependency> resolved =
138                     solver.getGraph().getOutgoingEdgeWithLabel(pred);
139 
140             // The edge is only non-null if instantiate() has been called before,
141             // it may be present in the graph at a deeper node. If that's the case
142             // it will be properly merged after regenerating the graph at the root context.
143             if (resolved == null) {
144                 logger.info("Must resolve desire: {}", desire);
145                 solver.resolve(desire);
146                 resolved = solver.getGraph().getOutgoingEdgeWithLabel(pred);
147             }
148 
149             // Check if the provider for the resolved node is in our cache
150             DAGNode<Component, Dependency> resolvedNode = resolved.getTail();
151             return type.cast(instantiator.makeInstantiator(resolvedNode, solver.getBackEdges()).instantiate());
152         }
153     }
