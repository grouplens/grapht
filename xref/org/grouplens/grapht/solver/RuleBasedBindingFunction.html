19  package org.grouplens.grapht.solver;

21  import com.google.common.collect.ImmutableListMultimap;
22  import com.google.common.collect.ListMultimap;
23  import com.google.common.collect.Multimap;
24  import org.apache.commons.lang3.tuple.Pair;
25  import org.grouplens.grapht.ResolutionException;
26  import org.grouplens.grapht.context.ContextMatch;
27  import org.grouplens.grapht.context.ContextMatcher;
28  import org.grouplens.grapht.reflect.QualifierMatcher;
29  import org.grouplens.grapht.util.Preconditions;
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  
33  import java.util.*;

35  /**
36   * <p>
37   * BindingFunction that uses BindRules created by the fluent API to bind desires
38   * to other desires or satisfactions.
39   * <p>
40   * For more details on context management, see {@link org.grouplens.grapht.context.ContextPattern},
41   * {@link org.grouplens.grapht.context.ContextElementMatcher}, and {@link QualifierMatcher}. This function uses the
42   * context to activate and select BindRules. A number of rules are used to order
43   * applicable BindRules and choose the best. When any of these rules rely on the
44   * current dependency context, the deepest node in the context has the most
45   * influence. Put another way, if contexts were strings, they could be ordered
46   * lexicographically from the right to the left.
47   * <p>
48   * When selecting BindRules to apply to a Desire, BindRules are ordered first by
49   * {@linkplain ContextMatch context match}, then by the ordering defined by the bind rule itself.
50   * <p>
51   * A summary of these rules is that the best specified BindRule is applied,
52   * where the context that the BindRule is activated in has more priority than
53   * the type of the BindRule. If multiple rules tie for best, then the solver
54   * fails with a checked exception.
55   * 
56   * @author <a href="http://grouplens.org">GroupLens Research</a>
57   */
58  public class RuleBasedBindingFunction implements BindingFunction {
59      private static final Map<Object,Set<BindRule>> bindRuleMemory
60              = new WeakHashMap<Object, Set<BindRule>>();
61  
62      private static final Logger logger = LoggerFactory.getLogger(RuleBasedBindingFunction.class);
63      
64      private final ImmutableListMultimap<ContextMatcher, BindRule> rules;
65      
66      public RuleBasedBindingFunction(Multimap<ContextMatcher, BindRule> rules) {
67          Preconditions.notNull("rules", rules);
68          
69          this.rules = ImmutableListMultimap.copyOf(rules);
70      }
71      
72      /**
73       * Get the rules underlying this binding function.
74       * @return The rules used by this BindingFunction
75       */
76      public ListMultimap<ContextMatcher, BindRule> getRules() {
77          return rules;
78      }
79      
80      @Override
81      public BindingResult bind(InjectionContext context, DesireChain desire) throws ResolutionException {
82          // FIXME Build a better way to remember the applied rules
83          Set<BindRule> appliedRules;
84          synchronized (bindRuleMemory) {
85              appliedRules = bindRuleMemory.get(desire.getKey());
86              if (appliedRules == null) {
87                  appliedRules = new HashSet<BindRule>();
88                  bindRuleMemory.put(desire.getKey(), appliedRules);
89              }
90          }
91  
92          // collect all bind rules that apply to this desire
93          List<Pair<ContextMatch, BindRule>> validRules = new ArrayList<Pair<ContextMatch, BindRule>>();
94          for (ContextMatcher matcher: rules.keySet()) {
95              ContextMatch match = matcher.matches(context);
96              if (match != null) {
97                  // the context applies to the current context, so go through all
98                  // bind rules within it and record those that match the desire
99                  for (BindRule br: rules.get(matcher)) {
100                     if (br.matches(desire.getCurrentDesire()) && !appliedRules.contains(br)) {
101                         validRules.add(Pair.of(match, br));
102                         logger.trace("Matching rule, context: {}, rule: {}", matcher, br);
103                     }
104                 }
105             }
106         }
107         
108         if (!validRules.isEmpty()) {
109             // we have a bind rule to apply
110             // pair's ordering is suitable for sorting the bind rules
111             Collections.sort(validRules);
112 
113             if (validRules.size() > 1) {
114                 // must check if other rules are equal to the first
115                 // we find the whole list of dupes for error reporting purposes
116                 List<BindRule> topRules = new ArrayList<BindRule>();
117                 topRules.add(validRules.get(0).getRight());
118                 for (int i = 1; i < validRules.size(); i++) {
119                     if (validRules.get(0).compareTo(validRules.get(i)) == 0) {
120                         topRules.add(validRules.get(i).getRight());
121                     }
122                 }
123                 
124                 if (topRules.size() > 1) {
125                     logger.error("{} bindings for {} in {}", topRules.size(),
126                                  desire, context);
127                     for (BindRule rule: topRules) {
128                         logger.info("matching rule: {}", rule);
129                     }
130                     // additional rules match just as well as the first, so fail
131                     throw new MultipleBindingsException(desire, context, topRules);
132                 }
133             }
134 
135             // apply the bind rule to get a new desire
136             BindRule selectedRule = validRules.get(0).getRight();
137             appliedRules.add(selectedRule);
138             
139             logger.debug("Applying rule: {} to desire: {}", selectedRule, desire);
140             return BindingResult.newBuilder()
141                                 .setDesire(selectedRule.apply(desire.getCurrentDesire()))
142                                 .setCachePolicy(selectedRule.getCachePolicy())
143                                 .setFlags(selectedRule.getFlags())
144                                 .build();
145         }
146         
147         // No rule to apply, so return null to delegate to the next binding function
148         return null;
149     }
