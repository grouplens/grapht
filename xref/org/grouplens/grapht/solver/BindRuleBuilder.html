19  package org.grouplens.grapht.solver;
20  
21  import com.google.common.base.Preconditions;
22  import org.grouplens.grapht.CachePolicy;
23  import org.grouplens.grapht.reflect.QualifierMatcher;
24  import org.grouplens.grapht.reflect.Qualifiers;
25  import org.grouplens.grapht.reflect.Satisfaction;
26  
27  import java.util.EnumSet;
28  
29  /**
30   * Builder for bind rules.
31   *
32   * @since 0.7
33   * @author <a href="http://www.grouplens.org">GroupLens Research</a>
34   */
35  public class BindRuleBuilder {
36      private Class<?> dependencyType;
37      private QualifierMatcher qualifierMatcher = Qualifiers.matchAny();
38  
39      private Satisfaction satisfaction;
40      private Class<?> implementation;
41  
42      private CachePolicy cachePolicy = CachePolicy.NO_PREFERENCE;
43      private EnumSet<BindingFlag> flags = BindingFlag.emptySet();
44  
45      public static BindRuleBuilder create() {
46          return new BindRuleBuilder();
47      }
48  
49      /**
50       * Get the dependency type to match.
51       * @return The dependency type to match.
52       */
53      public Class<?> getDependencyType() {
54          return dependencyType;
55      }
56  
57      /**
58       * Set the dependency type to match.
59       * @param type The dependency type to match.
60       */
61      public BindRuleBuilder setDependencyType(Class<?> type) {
62          dependencyType = type;
63          return this;
64      }
65  
66      /**
67       * Get the configured qualifer matcher.  The initial qualifier matcher is {@link org.grouplens.grapht.reflect.Qualifiers#matchAny()}.
68       * @return The qualifier matcher.
69       */
70      public QualifierMatcher getQualifierMatcher() {
71          return qualifierMatcher;
72      }
73  
74      /**
75       * Set the qualifier matcher.
76       * @param qm The qualifier matcher.
77       */
78      public BindRuleBuilder setQualifierMatcher(QualifierMatcher qm) {
79          qualifierMatcher = qm;
80          return this;
81      }
82  
83      /**
84       * Get the target satisfaction.
85       * @return The configured satisfaction, or {@code null} if none is configured.
86       */
87      public Satisfaction getSatisfaction() {
88          return satisfaction;
89      }
90  
91      /**
92       * Set the satisfaction to bind to.  This will unset the implementation class and result in
93       * a satisfaction binding.
94       *
95       * @param sat The satisfaction.
96       */
97      public BindRuleBuilder setSatisfaction(Satisfaction sat) {
98          satisfaction = sat;
99          return this;
100     }
101 
102     /**
103      * Get the target implementation.
104      * @return The target implementation, or {@code null} if none is configured.
105      */
106     public Class<?> getImplementation() {
107         return implementation;
108     }
109 
110     /**
111      * Set the target implementation. This will unset the satisfaction and result in an implementation
112      * class binding.
113      *
114      * @param type The implementation class.
115      */
116     public BindRuleBuilder setImplementation(Class<?> type) {
117         implementation = type;
118         return this;
119     }
120 
121     /**
122      * Query whether the binding will be terminal.
123      * @return {@code true} if the binding will be terminal.
124      */
125     public boolean isTerminal() {
126         return flags.contains(BindingFlag.TERMINAL);
127     }
128 
129     /**
130      * Set whether the binding will be terminal.
131      *
132      * @param term {@code true} to create a terminal binding.
133      * @see BindingFlag#TERMINAL
134      */
135     public BindRuleBuilder setTerminal(boolean term) {
136         if (term) {
137             flags.add(BindingFlag.TERMINAL);
138         } else {
139             flags.remove(BindingFlag.TERMINAL);
140         }
141         return this;
142     }
143 
144     /**
145      * Set the flags on this bind rule.
146      * @param fs The flags.
147      * @return The builder (for chaining).
148      */
149     public BindRuleBuilder setFlags(EnumSet<BindingFlag> fs) {
150         flags = fs.clone();
151         return this;
152     }
153 
154     /**
155      * Add a flag to the constructed bind rule.
156      * @param flag The flag to add.
157      * @return The builder (for chaining).
158      */
159     public BindRuleBuilder addFlag(BindingFlag flag) {
160         flags.add(flag);
161         return this;
162     }
163 
164     /**
165      * Get the cache policy.
166      * @return The cache policy.
167      */
168     public CachePolicy getCachePolicy() {
169         return cachePolicy;
170     }
171 
172     /**
173      * Set the cache policy.
174      * @param policy The cache policy.
175      */
176     public BindRuleBuilder setCachePolicy(CachePolicy policy) {
177         cachePolicy = policy;
178         return this;
179     }
180 
181     public BindRule build() {
182         Preconditions.checkState(dependencyType != null, "no dependency type specified");
183         if (implementation != null) {
184             assert satisfaction == null;
185             return new BindRuleImpl(dependencyType, implementation, cachePolicy, qualifierMatcher, flags);
186         } else if (satisfaction != null) {
187             return new BindRuleImpl(dependencyType, satisfaction, cachePolicy, qualifierMatcher, flags);
188         } else {
189             throw new IllegalStateException("no binding target specified");
190         }
191     }
192 }
