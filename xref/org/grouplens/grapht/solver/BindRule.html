19  package org.grouplens.grapht.solver;
20  
21  import org.grouplens.grapht.CachePolicy;
22  import org.grouplens.grapht.reflect.Desire;
23  
24  import java.util.EnumSet;
25  
26  /**
27   * BindRule is a partial function from desire to desire that acts as a binding.
28   * The {@link RuleBasedBindingFunction} takes a collection of BindRules grouped
29   * into their activating contexts to form a {@link BindingFunction}.
30   *
31   * <p><b>Note:</b> this interface's ordering is inconsistent with {@link Object#equals(Object)}. See
32   * {@link #compareTo(BindRule)} for more details.
33   *
34   * @see BindRules
35   */
36  public interface BindRule extends Comparable<BindRule> {
37      /**
38       * Get the rule's cache policy.
39       * @return The CachePolicy to use for satisfactions created with this rule.
40       */
41      CachePolicy getCachePolicy();
42  
43      /**
44       * Apply this BindRule to the given Desire, and return a restricted and
45       * possibly satisfied desire. It is assumed that {@link #matches(org.grouplens.grapht.reflect.Desire)}
46       * returns true.
47       *
48       * @param desire The desire that is input to this partial binding function
49       * @return The restricted desire
50       */
51      Desire apply(Desire desire);
52  
53      /**
54       * Query whether this rule is terminal.  Terminal rules have no further lookup
55       * applied in order to find the final binding target.
56       *
57       * @return True if this should be the last bind rule applied to the desire
58       *         chain when attempting to find a satisfaction
59       */
60      boolean isTerminal();
61  
62      /**
63       * Get the flags attached to this bind rule.
64       *
65       * @return The flags attached to the bind rule.
66       */
67      EnumSet<BindingFlag> getFlags();
68  
69      /**
70       * @param desire The input desire
71       * @return True if this desire matches this BindRule and can be passed to
72       *         {@link #apply(org.grouplens.grapht.reflect.Desire)}
73       */
74      boolean matches(Desire desire);
75  
76      /**
77       * Create a new bind rule builder initialized to copy this bind rule.  Use this to create a copy
78       * of this bind rule that differs in some way.
79       *
80       * @return A new bind rule builder whose {@link BindRuleBuilder#build()} method will return a
81       *         copy of this bind rule.
82       * @throws UnsupportedOperationException if the bind rule cannot be reconfigured in this manner.
83       */
84      BindRuleBuilder newCopyBuilder();
85  
86      /**
87       * Compare this bind rule to another. More-specific bind rules compare less than less-specific
88       * rules.
89       *
90       * <p><b>Note:</b> This implements an ordering inconsistent with {@link Object#equals(Object)}. Equal
91       * rules will compare the same (and implementations are required to enforce this), but unequal
92       * rules may compare equal to each other when compared with this method.
93       *
94       * @param other The bind rule to compare with.
95       * @return The comparison result.
96       * @throws IllegalArgumentException if called with an incompatible bind rule.  This should not
97       *                                  arise in practice, but if you implement your own bind rules
98       *                                  they will not be comparable with the ones provided by
99       *                                  Grapht.
100      */
101     @Override
102     int compareTo(BindRule other);
103 }
