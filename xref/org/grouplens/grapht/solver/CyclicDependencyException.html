19  package org.grouplens.grapht.solver;
20  
21  import org.grouplens.grapht.ResolutionException;
22  import org.grouplens.grapht.reflect.Desire;
23  
24  /**
25   * Thrown by when a cyclic dependency is detected and could not be broken or
26   * bypassed by the solver.
27   * 
28   * @author <a href="http://grouplens.org">GroupLens Research</a>
29   */
30  public class CyclicDependencyException extends ResolutionException {
31      private static final long serialVersionUID = 1L;
32  
33      private final Desire desire;
34      
35      public CyclicDependencyException(Desire desire, String msg) {
36          super(msg);
37          this.desire = desire;
38      }
39      
40      /**
41       * @return The current desire that triggered the cycle detection
42       */
43      public Desire getDesire() {
44          return desire;
45      }
46      
47      @Override
48      public String getMessage() {
49          return new StringBuilder("Unable to satisfy desire: ")
50              .append(format(desire.getInjectionPoint()))
51              .append('\n')
52              .append("Reason: ")
53              .append(super.getMessage())
54              .toString();
55      }
56  }
