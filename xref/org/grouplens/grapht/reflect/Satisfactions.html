19  package org.grouplens.grapht.reflect;
20  
21  import org.grouplens.grapht.reflect.internal.*;
22  
23  import javax.annotation.Nonnull;
24  import javax.inject.Provider;
25  
26  /**
27   * Class to construct specific {@link Satisfaction} implementations.
28   *
29   * @since 0.7
30   * @author <a href="http://www.grouplens.org">GroupLens Research</a>
31   */
32  public final class Satisfactions {
33      private Satisfactions() {}
34  
35      public static Satisfaction type(@Nonnull Class<?> type) {
36          return new ClassSatisfaction(type);
37      }
38  
39      public static Satisfaction nullOfType(@Nonnull Class<?> type) {
40          return new NullSatisfaction(type);
41      }
42  
43      public static Satisfaction instance(@Nonnull Object o) {
44          return new InstanceSatisfaction(o);
45      }
46  
47      public static Satisfaction providerType(@Nonnull Class<? extends Provider<?>> providerType) {
48          return new ProviderClassSatisfaction(providerType);
49      }
50  
51      public static Satisfaction providerInstance(@Nonnull Provider<?> provider) {
52          return new ProviderInstanceSatisfaction(provider);
53      }
54  }
