32  /**
33   * A concrete type. It has a set of dependencies which must be satisfied in
34   * order to instantiate it. It can also be viewed as an instantiable extension
35   * of {@link Type}.
36   * <p>
37   * Satisfactions are expected to provide a reasonable implementation of
38   * {@link Object#equals(Object)} and {@link Object#hashCode()} so that they can
39   * be de-duplicated, etc.
40   * 
41   * @author <a href="http://grouplens.org">GroupLens Research</a>
42   */
43  public interface Satisfaction extends Serializable {
44      /**
45       * Get this satisfaction's dependencies.
46       *
47       * @return A list of dependencies which must be satisfied in order to
48       *         instantiate this satisfaction.
49       */
50      List<Desire> getDependencies();
51  
52      /**
53       * Get the type of this satisfaction. If this is a synthetic Satisfaction,
54       * then a null type is returned.
55       * 
56       * @return The type of objects to be instantiated by this satisfaction.
57       */
58      Type getType();
59  
60      /**
61       * Get the type-erased class of this satisfaction's type. If this is a
62       * synthetic Satisfaction, then null is returned.
63       * 
64       * @return The class object for this satisfaction's type.
65       */
66      Class<?> getErasedType();
67  
68      /**
69       * Query whether this satisfaction already has an instance to return. If
70       * true, then this satisfaction's provider will just return an instance
71       * that is already in existence; if false, then its provider may create
72       * new instances.
73       *
74       * @return whether the satisfaction already has an instance.
75       */
76      boolean hasInstance();
77  
78      /**
79       * Visit the satisfaction. This method invokes the appropriate method on the
80       * provided visitor to report information on itself.
81       *
82       * @param visitor The visitor object.
83       * @param <T> The type returned from the visitor.
84       * @return The return value from the invoked visitor method.
85       */
86      <T> T visit(SatisfactionVisitor<T> visitor);
87      
88      /**
89       * Get the default cache policy for instances created by this satisfaction.
90       * In most cases this should be NO_PREFERENCE, but annotations such as
91       * {@link Singleton} can be used to specify a default. BindingFunctions are
92       * allowed to overrule the default cache policy.
93       * 
94       * @return The default cache policy if no function overrules it
95       */
96      CachePolicy getDefaultCachePolicy();
97  
98      /**
99       * <p>
100      * Create an instantiator from this satisfaction.
101      * <p>
102      * If the details of the Satisfaction require instantiation (e.g. a class or
103      * provider class satisfaction), the returned provider should be a new
104      * instance, and it should perform no memoization of its own. Caching is the
105      * purview of {@link Injector} implementations.
106      * <p>
107      * If the satisfaction is configured to use specific instances, this rule is
108      * obviously void.
109      * 
110      * @param dependencies A function mapping desires to providers of their
111      *            instances.
112      * @return An instantiator of new instances of the type specified by this
113      *         satisfaction, instantiated using the specified dependency
114      *         mapping.
115      */
116     Instantiator makeInstantiator(Map<Desire,Instantiator> dependencies);
117 }
