19  package org.grouplens.grapht.reflect;
20  
21  import java.io.Serializable;
22  
23  /**
24   * A possibly-not-concrete type. This represents the type of a dependency; it
25   * may or may not be concrete. It can effectively be any type. Desires are
26   * iteratively resolved and narrowed until they finally correspond to
27   * {@link Satisfaction}s.
28   *
29   * @author <a href="http://grouplens.org">GroupLens Research</a>
30   *
31   */
32  public interface Desire extends Serializable {
33      /**
34       * Query whether this desire is instantiable, that is, resolved to a
35       * concrete type. If it is instantiable, then it can be converted to a Satisfaction
36       * with {@link #getSatisfaction()}.
37       *
38       * @return <tt>true</tt> if the desire is for a concrete class. The only
39       *         further desires or satisfactions that can satisfy it are for subclasses
40       *         of the desire type.
41       */
42      boolean isInstantiable();
43  
44      /**
45       * Get the satisfaction (concrete type) if this desire is fully resolved.
46       *
47       * @return The satisfaction for this desire, or <tt>null</tt> if the desire is not a
48       *         concrete type.
49       */
50      Satisfaction getSatisfaction();
51  
52      /**
53       * @return The injection point of this desire
54       */
55      InjectionPoint getInjectionPoint();
56      
57      /**
58       * @return The desired type, potentially more constrained than the injection
59       *         point's type
60       */
61      Class<?> getDesiredType();
62      
63      /**
64       * Return a new Desire that restricts the type of this desire to the given
65       * class. The type must be a subclass of the desired type.
66       * 
67       * @param type The restricted type
68       * @return A restricted Desire
69       */
70      Desire restrict(Class<?> type);
71      
72      /**
73       * Return a new Desire that restricts the type of this desire to the erased
74       * type of the satisfaction. The returned Desire will also be instantiable
75       * and return the provided satisfaction from {@link #getSatisfaction()}.
76       * 
77       * @param satisfaction The satisfaction to restrict this desire to
78       * @return A restricted and satisfied desire
79       */
80      Desire restrict(Satisfaction satisfaction);
81  }
