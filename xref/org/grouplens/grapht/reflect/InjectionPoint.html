19  package org.grouplens.grapht.reflect;
20  
21  import javax.annotation.Nonnull;
22  import javax.annotation.Nullable;
23  import java.io.Serializable;
24  import java.lang.annotation.Annotation;
25  import java.lang.reflect.Member;
26  import java.lang.reflect.Type;
27  import java.util.Collection;
28  
29  
30  /**
31   * InjectionPoint represents a point of injection for an instantiable type.
32   * Examples include a constructor parameter, a setter method, or a field.
33   * 
34   * @author <a href="http://grouplens.org">GroupLens Research</a>
35   */
36  public interface InjectionPoint extends Serializable {
37      /**
38       * Return the type required to satisfy the injection point.
39       * 
40       * @return The type of the injection point
41       */
42      Type getType();
43      
44      /**
45       * @return Return the erased type of {@link #getType()}
46       */
47      Class<?> getErasedType();
48  
49      /**
50       * Return the qualifier annotation added to the injection point. The
51       * returned annotation's type will have been annotated with
52       * {@link javax.inject.Qualifier}. If the injection point is not qualified, this will
53       * return null.
54       *
55       * @return Any qualifier applied to the injection point
56       */
57      @Nullable
58      Annotation getQualifier();
59  
60      /**
61       * Return the attribute of type A that is applied to the injection point. If
62       * the injection point does not have an attribute of type {@code A}, then null is
63       * returned.
64       *
65       * @param atype Attribute annotation type.  It must be annotated with {@link org.grouplens.grapht.annotation.Attribute}.
66       * @return The instance of A applied to the injection point, or null
67       * @throws NullPointerException if atype is null
68       */
69      @Nullable <A extends Annotation> A getAttribute(Class<A> atype);
70  
71      /**
72       * @return Immutable collection of attribute annotations (does not include
73       *         the qualifier)
74       */
75      @Nonnull
76      Collection<Annotation> getAttributes();
77      
78      /**
79       * Return the Member that produced this injection point. Synthetic injection
80       * points can have a null member.
81       * 
82       * @return The Member that produces this injection point
83       */
84      @Nullable Member getMember();
85  
86      /**
87       * @return True if this injection point accepts null values
88       */
89      boolean isNullable();
90  }
