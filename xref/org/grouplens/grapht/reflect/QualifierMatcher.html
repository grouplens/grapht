19  package org.grouplens.grapht.reflect;

21  import com.google.common.base.Predicate;
22  
23  import javax.annotation.Nullable;
24  import javax.inject.Qualifier;
25  import java.io.Serializable;
26  import java.lang.annotation.Annotation;
27  
28  
29  /**
30   * <p>
31   * QualifierMatcher encapsulates the logic used to determine if a BindRule or
32   * ContextElementMatcher match a particular Qualifier. Common qualifier matching rules
33   * are:
34   * <ol>
35   * <li>Any qualifier</li>
36   * <li>No qualifier</li>
37   * <li>Annotation type</li>
38   * <li>Annotation instance equality</li>
39   * </ol>
40   * All QualifierMatchers created by the same InjectSPI must be comparable,
41   * matchers from different SPIs do not need to be comparable.
42   * 
43   * @author <a href="http://grouplens.org">GroupLens Research</a>
44   */
45  public interface QualifierMatcher extends Predicate<Annotation>, Comparable<QualifierMatcher>, Serializable {
46      /**
47       * Return true if this matcher matches the given qualifier annotation. It
48       * can be assumed that the annotation type has been annotated with
49       * {@link Qualifier}. The qualifier will be null if the injection point
50       * being matched did not have a qualifier.
51       * 
52       * @param q The qualifier to match
53       * @return True if matched
54       */
55      boolean matches(@Nullable Annotation q);
56  
57      /**
58       * Return true if this matcher matches the given qualifier annotation. It
59       * can be assumed that the annotation type has been annotated with
60       * {@link Qualifier}. The qualifier will be null if the injection point
61       * being matched did not have a qualifier.
62       *
63       * @param q The qualifier to match
64       * @return True if matched
65       */
66      @Override
67      boolean apply(@Nullable Annotation q);
68  
69      /**
70       * Get the priority of this matcher. Lower priority values have precedence in selecting
71       * the final bind rule. All Grapht matchers have priorities of at least 0; negative
72       * priority values are reserved for custom extensions.
73       *
74       * @return The priority.
75       */
76      int getPriority();
77  }
