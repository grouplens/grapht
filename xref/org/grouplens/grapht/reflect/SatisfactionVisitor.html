19  package org.grouplens.grapht.reflect;
20  
21  import javax.inject.Provider;
22  
23  /**
24   * An interface for visiting {@link Satisfaction}s.
25   *
26   * @param <T> The type returned from this satisfaction's values.
27   * @author <a href="http://grouplens.org">GroupLens Research</a>
28   * @since 0.5
29   * @see Satisfaction
30   */
31  public interface SatisfactionVisitor<T> {
32      /**
33       * Called when visiting a null satisfaction.
34       *
35       * @return The return value.
36       */
37      T visitNull();
38  
39      /**
40       * Called when visiting a satisfaction that will instantiate a class.
41       *
42       * @param clazz The implementation class.
43       * @return The return value.
44       */
45      T visitClass(Class<?> clazz);
46  
47      /**
48       * Called when visiting a satisfaction that will return a pre-configured instance.
49       *
50       * @param instance The instance that will be returned.  The visitor should not modify it
51       *                 in any way.
52       * @return The return value.
53       */
54      T visitInstance(Object instance);
55  
56      /**
57       * Called when visiting a satisfaction that will instantiate and invoke a provider class.
58       *
59       * @param pclass The provider class.
60       * @return The return value.
61       */
62      T visitProviderClass(Class<? extends Provider<?>> pclass);
63  
64      /**
65       * Called when visiting a satisfaction that will invoke a pre-instantiated provider.
66       *
67       * @param provider The provider instance.  The visitor should not modify it in any way.
68       * @return The return value.
69       */
70      T visitProviderInstance(Provider<?> provider);
71  }
