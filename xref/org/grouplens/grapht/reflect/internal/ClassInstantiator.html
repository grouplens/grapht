38  /**
39   * Instantiates class instances.
40   * 
41   * @author <a href="http://grouplens.org">GroupLens Research</a>
42   */
43  public class ClassInstantiator implements Instantiator {
44      private static final Logger logger = LoggerFactory.getLogger(ClassInstantiator.class);
45      
46      private final Class<?> type;
47      private final List<Desire> desires;
48      private final Map<Desire, Instantiator> providers;
49  
50      /**
51       * Create an ClassInstantiator that will provide instances of the given
52       * type, with given the list of desires and a function mapping that
53       * satisfies those providers.
54       * 
55       * @param type The type of instance created
56       * @param desires The dependency desires for the instance
57       * @param providers The providers that satisfy the desires of the type
58       */
59      public ClassInstantiator(Class<?> type, List<Desire> desires, Map<Desire,Instantiator> providers) {
60          Preconditions.notNull("type", type);
61          Preconditions.notNull("desires", desires);
62          Preconditions.notNull("providers", providers);
63          
64          this.type = type;
65          this.desires = desires;
66          this.providers = providers;
67      }
68  
69      @Override
70      public Class getType() {
71          return type;
72      }
73  
74      @Override
75      public Object instantiate() throws ConstructionException {
76          // find constructor and build up necessary constructor arguments
77          Constructor<?> ctor = getConstructor();
78          Object[] ctorArgs = new Object[ctor.getParameterTypes().length];
79          for (Desire d: desires) {
80              if (d.getInjectionPoint() instanceof ConstructorParameterInjectionPoint) {
81                  // this desire is a constructor argument so create it now
82                  Instantiator provider = providers.get(d);
83                  ConstructorParameterInjectionPoint cd = (ConstructorParameterInjectionPoint) d.getInjectionPoint();
84                  ctorArgs[cd.getParameterIndex()] = checkNull(cd, provider.instantiate());
85              }
86          }
87          
88          // create the instance that we are injecting
89          Object instance;
90          try {
91              logger.trace("Invoking constructor {} with arguments {}", ctor, ctorArgs);
92              ctor.setAccessible(true);
93              instance = ctor.newInstance(ctorArgs);
94          } catch (InvocationTargetException e) {
95              throw new ConstructionException(ctor, "Constructor " + ctor + " failed", e);
96          } catch (InstantiationException e) {
97              throw new ConstructionException(ctor, "Could not instantiate " + type, e);
98          } catch (IllegalAccessException e) {
99              throw new ConstructionException(ctor, "Access violation on " + ctor, e);
100         }
101 
102         // satisfy dependencies in the order of the list, which was
103         // prepared to comply with JSR 330
104         Map<Method, InjectionArgs> settersAndArguments = new HashMap<Method, InjectionArgs>();
105         for (Desire d: desires) {
106             if (d.getInjectionPoint() instanceof FieldInjectionPoint) {
107                 FieldInjectionPoint fd = (FieldInjectionPoint) d.getInjectionPoint();
108                 Object value = checkNull(fd, providers.get(d).instantiate());
109                 Field field = fd.getMember();
110 
111                 try {
112                     logger.trace("Setting field {} with arguments {}", field, value);
113                     field.setAccessible(true);
114                     field.set(instance, value);
115                 } catch (IllegalAccessException e) {
116                     throw new ConstructionException(fd, e);
117                 }
118             } else if (d.getInjectionPoint() instanceof SetterInjectionPoint) {
119                 // collect parameters before invoking
120                 SetterInjectionPoint sd = (SetterInjectionPoint) d.getInjectionPoint();
121                 InjectionArgs args = settersAndArguments.get(sd.getMember());
122                 Method setter = sd.getMember();
123 
124                 if (args == null) {
125                     // first encounter of this method
126                     args = new InjectionArgs(setter.getParameterTypes().length);
127                     settersAndArguments.put(setter, args);
128                 }
129                 
130                 Instantiator provider = providers.get(d);
131                 args.set(sd.getParameterIndex(), checkNull(sd, provider.instantiate()));
132                 
133                 if (args.isCompleted()) {
134                     // all parameters initialized, invoke the setter with all arguments
135                     try {
136                         logger.trace("Invoking setter {} with arguments {}", setter, args.arguments);
137                         setter.setAccessible(true);
138                         setter.invoke(instance, args.arguments);
139                     } catch (InvocationTargetException e) {
140                         String message = "Exception thrown by ";
141                         if (args.arguments.length == 1) {
142                             message += sd;
143                         } else {
144                             message += setter;
145                         }
146                         throw new ConstructionException(sd, message, e);
147                     } catch (IllegalAccessException e) {
148                         String message = "Access violation calling ";
149                         if (args.arguments.length == 1) {
150                             message += sd;
151                         } else {
152                             message += setter;
153                         }
154                         throw new ConstructionException(sd, message, e);
155                     }
156                 }
157             } else if (d.getInjectionPoint() instanceof NoArgumentInjectionPoint) {
158                 // just invoke the method
159                 Method method = ((NoArgumentInjectionPoint) d.getInjectionPoint()).getMember();
160                 try {
161                     logger.trace("Invoking no-argument injection point {}", d.getInjectionPoint());
162                     method.setAccessible(true);
163                     method.invoke(instance);
164                 } catch (InvocationTargetException e) {
165                     throw new ConstructionException(d.getInjectionPoint(), "Exception throw by " + method, e);
166                 } catch (IllegalAccessException e) {
167                     throw new ConstructionException(d.getInjectionPoint(), "Access violation invoking " + method, e);
168                 }
169             }
170         }
171         
172         // the instance has been fully configured
173         return instance;
174     }
175     
176     @SuppressWarnings("unchecked")
177     private Constructor<?> getConstructor() {
178         for (Desire d: desires) {
179             if (d.getInjectionPoint() instanceof ConstructorParameterInjectionPoint) {
180                 // since we only allow one injectable constructor, any ConstructorParameterInjectionPoint
181                 // will have the same constructor as all other constructor parameter injection points
182                 Constructor<?> ctor = ((ConstructorParameterInjectionPoint) d.getInjectionPoint()).getMember();
183                 logger.debug("Using constructor annotated with @Inject: {}", ctor);
184                 return ctor;
185             }
186         }
187         
188         try {
189             logger.debug("Using default constructor for {}", type);
190             return type.getDeclaredConstructor();
191         } catch (NoSuchMethodException e) {
192             // this constructor is being invoked for a ClassSatisfaction or a 
193             // ProviderClassSatisfaction, both of which assert that the type is
194             // instantiable, so this should never happen
195             throw new RuntimeException("Unexpected exception", e);
196         }
197     }
198     
199     private static Object checkNull(InjectionPoint injectPoint, Object value) throws NullDependencyException {
200         if (value == null && !injectPoint.isNullable()) {
201             throw new NullDependencyException(injectPoint);
202         } else {
203             return value;
204         }
205     }
206     
207     private static class InjectionArgs {
208         private final Object[] arguments;
209         private final boolean[] injected;
210         
211         public InjectionArgs(int num) {
212             arguments = new Object[num];
213             injected = new boolean[num];
214         }
215         
216         public void set(int i, Object o) {
217             arguments[i] =o;
218             injected[i] = true;
219         }
220         
221         public boolean isCompleted() {
222             for (int i = 0; i < injected.length; i++) {
223                 if (!injected[i]) {
224                     return false;
225                 }
226             }
227             return true;
228         }
229     }
