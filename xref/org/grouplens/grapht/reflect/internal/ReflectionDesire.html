38  /**
39   * ReflectionDesire is an implementation of desire that contains all necessary
40   * implementation to represent a desire, except that the point of injection is
41   * abstracted by an {@link InjectionPoint}.
42   * 
43   * @author <a href="http://grouplens.org">GroupLens Research</a>
44   */
45  public class ReflectionDesire implements Desire, Serializable {
46      private static final long serialVersionUID = -1L;
47  
48      /**
49       * Return a list of desires that must satisfied in order to instantiate the
50       * given type.
51       *
52       * @param type The class type whose dependencies will be queried
53       * @return The dependency desires for the given type
54       * @throws NullPointerException if the type is null
55       */
56      public static List<Desire> getDesires(Class<?> type) {
57          List<Desire> desires = Lists.newArrayList();
58  
59          boolean ctorFound = false;
60          for (Constructor<?> ctor: type.getDeclaredConstructors()) {
61              if (ctor.getAnnotation(Inject.class) != null) {
62                  if (!ctorFound) {
63                      ctorFound = true;
64                      for (int i = 0; i < ctor.getParameterTypes().length; i++) {
65                          desires.add(new ReflectionDesire(new ConstructorParameterInjectionPoint(ctor, i)));
66                      }
67                  } else {
68                      // at the moment there can only be one injectable constructor
69                      throw new InvalidBindingException(type, "More than one constructor with @Inject is not allowed");
70                  }
71              }
72          }
73          
74          // JSR 330 mandates that super class methods are injected first, so we
75          // collect method injection points into a separate list and then reverse
76          // it to get the ordering correct.
77          List<Desire> groupDesires = Lists.newArrayList();
78          
79          // Must also keep track of methods overridden in the subtypes.
80          Set<Signature> visitedMethods = new HashSet<Signature>();
81          while(type != null) {
82              for (Method m: type.getDeclaredMethods()) {
83                  Signature s = new Signature(m);
84                  if (!visitedMethods.contains(s) && m.getAnnotation(Inject.class) != null
85                      && !Modifier.isStatic(m.getModifiers())) {
86                      // have not seen this signature, and its an injection point
87                      if (m.getParameterTypes().length > 0) {
88                          for (int i = 0; i < m.getParameterTypes().length; i++) {
89                              groupDesires.add(new ReflectionDesire(new SetterInjectionPoint(m, i)));
90                          }
91                      } else {
92                          // hack to invoke no-argument injectable methods required by JSR 330
93                          groupDesires.add(new ReflectionDesire(new NoArgumentInjectionPoint(m)));
94                      }
95                  }
96                  // always add signature, because a subclass without @Inject
97                  // overrides any @Inject on the superclass's method declaration
98                  visitedMethods.add(s);
99              }
100             for (Field f: type.getDeclaredFields()) {
101                 if (f.getAnnotation(Inject.class) != null && !Modifier.isStatic(f.getModifiers())) {
102                     // have not seen this field
103                     groupDesires.add(new ReflectionDesire(new FieldInjectionPoint(f)));
104                 }
105             }
106             
107             type = type.getSuperclass();
108         }
109         
110         // after reversing this list, fields will be injected 
111         // before methods as required
112         Collections.reverse(groupDesires);
113         desires.addAll(groupDesires);
114         
115         return Collections.unmodifiableList(desires);
116     }

122     /**
123      * Create a ReflectionDesire that immediately wraps the given
124      * InjectionPoint. The desired type equals the type declared by the
125      * injection point. The created desire will have a satisfaction if the
126      * injection point's type is satisfiable.
127      * 
128      * @param injectPoint The injection point to wrap
129      * @throws NullPointerException if injectPoint is null
130      */
131     public ReflectionDesire(InjectionPoint injectPoint) {
132         this(injectPoint.getErasedType(), injectPoint, null);
133     }

135     /**
136      * Create a ReflectionDesire that represents the dependency for
137      * <tt>desiredType</tt> that will be injected into the given InjectionPoint.
138      * The optional satisfaction will satisfy this desire. If null is provided,
139      * and the desired type instantiable, a ClassSatisfaction is created.
140      * 
141      * @param desiredType The desired type of the dependency
142      * @param injectPoint The injection point of the desire
143      * @param satisfaction The satisfaction satisfying this desire, if there is
144      *            one
145      * @throws NullPointerException if desiredType, injectPoint, or dfltSource is null
146      * @throws IllegalArgumentException if desiredType is not assignable to the
147      *             type of the injection point, or if the satisfaction's type is
148      *             not assignable to the desired type
149      */
150     public ReflectionDesire(Class<?> desiredType, InjectionPoint injectPoint,
151                             Satisfaction satisfaction) {
152         Preconditions.notNull("desired type", desiredType);
153         Preconditions.notNull("injection point", injectPoint);
154 
155         desiredType = Types.box(desiredType);
156         Preconditions.isAssignable(injectPoint.getErasedType(), desiredType);
157         if (satisfaction != null) {
158             Preconditions.isAssignable(desiredType, satisfaction.getErasedType());
159         }
160 
161         // try and find a satisfaction
162         if (satisfaction == null) {
163             if (Types.shouldBeInstantiable(desiredType)) {
164                 if (Types.isInstantiable(desiredType)) {
165                     satisfaction = new ClassSatisfaction(desiredType);
166                 } // else don't satisfy this, even if the injection point is null
167             } else if (injectPoint.isNullable()) {
168                 // we only default to null if the injection point depended on
169                 // an interface. if they ask for a concrete type, it's a bug
170                 // for that type not to be injectable
171                 satisfaction = new NullSatisfaction(desiredType);
172             }
173         }
174 
175         this.desiredType = desiredType;
176         this.injectPoint = injectPoint;
177         this.satisfaction = satisfaction;
178     }

212         if (!(o instanceof ReflectionDesire)) {
213             return false;

216         return (r.desiredType.equals(desiredType) && 
217                 r.injectPoint.equals(injectPoint) && 
218                 (r.satisfaction == null ? satisfaction == null : r.satisfaction.equals(satisfaction)));

239     private static class SerialProxy implements Serializable {
240         private static final long serialVersionUID = 1L;
241 
242         private final InjectionPoint injectionPoint;
243         private final ClassProxy desiredType;
244         private final Satisfaction satisfaction;
245 
246         public SerialProxy(Class<?> type, InjectionPoint ip, Satisfaction sat) {
247             injectionPoint = ip;
248             desiredType = ClassProxy.of(type);
249             satisfaction = sat;
250         }
251 
252         @SuppressWarnings("unchecked")
253         private Object readResolve() throws ObjectStreamException {
254             try {
255                 return new ReflectionDesire(desiredType.resolve(),
256                                             injectionPoint,
257                                             satisfaction);
258             } catch (ClassNotFoundException e) {
259                 InvalidObjectException ex = new InvalidObjectException("cannot resolve " + desiredType);
260                 ex.initCause(e);
261                 throw ex;
262             } catch (InvalidBindingException e) {
263                 InvalidObjectException ex = new InvalidObjectException("invalid binding");
264                 ex.initCause(e);
265                 throw ex;
266             }
267         }

270     /*
271      * Internal class to track a methods signature. Java's default reflection
272      * doesn't give us a convenient way to record just this information.
273      *
274      * FIXME Document why we need this class more clearly
275      */
276     public static class Signature {
277         private final String name;
278         private final Type[] args;
279         
280         public Signature(Method m) {
281             // FIXME Make it clearer what this code is supposed to do
282             int mods = m.getModifiers();
283             if (Modifier.isPublic(mods) || Modifier.isProtected(mods)) {
284                 // method overrides depends solely on method name
285                 name = m.getName();
286             } else if (Modifier.isPrivate(mods)) {
287                 // method overrides depend on method name and class name
288                 name = m.getName() + m.getDeclaringClass().getCanonicalName();
289             } else {
290                 // method overrides depend on method name and package,
291                 // since it is package-private
292                 Package pkg = m.getDeclaringClass().getPackage();
293                 if (pkg != null) {
294                     name = m.getName() + pkg.getName();
295                 } else {
296                     name = m.getName();
297                 }
298             }
299             args = m.getGenericParameterTypes();
300         }
301         
302         @Override
303         public boolean equals(Object o) {
304             if (!(o instanceof Signature)) {
305                 return false;
306             }
307             Signature s = (Signature) o;
308             return s.name.equals(name) && Arrays.equals(args, s.args);
309         }
310         
311         @Override
312         public int hashCode() {
313             return (name.hashCode() ^ Arrays.hashCode(args));
314         }
315     }
