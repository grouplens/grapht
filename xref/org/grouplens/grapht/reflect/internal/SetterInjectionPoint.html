37  /**
38   * SetterInjectionPoint represents an injection point via a setter method.
39   * 
40   * @author <a href="http://grouplens.org">GroupLens Research</a>
41   */
42  public class SetterInjectionPoint implements InjectionPoint, Serializable {
43      private static final long serialVersionUID = -1L;
44      // transient because we use a serialization proxy
45      private final transient Method setter;
46      private final transient int parameter;
47      private final transient AnnotationHelper annotations;
48  
49      /**
50       * Create a SetterInjectionPoint that wraps the given setter method.
51       * 
52       * @param setter The setter method
53       * @param parameter The parameter index to apply
54       */
55      public SetterInjectionPoint(@Nonnull Method setter, int parameter) {
56          Preconditions.notNull("setter method", setter);
57          Preconditions.inRange(parameter, 0, setter.getParameterTypes().length);
58          
59          this.annotations = new AnnotationHelper(setter.getParameterAnnotations()[parameter]);
60          this.setter = setter;
61          this.parameter = parameter;
62      }
63      
64      /**
65       * @return The setter method wrapped by this injection point
66       */
67      @Override @Nonnull
68      public Method getMember() {
69          return setter;
70      }
71      
72      /**
73       * @return The parameter index of this injection point within the
74       *         setter's parameters
75       */
76      public int getParameterIndex() {
77          return parameter;
78      }
79      
80      @Override
81      public boolean isNullable() {
82          // we'll check both setter and parameter annotations
83          return Types.hasNullableAnnotation(setter.getAnnotations()) || 
84                 Types.hasNullableAnnotation(setter.getParameterAnnotations()[parameter]);
85      }
86  
87      @Override
88      public Type getType() {
89          return Types.box(setter.getGenericParameterTypes()[parameter]);
90      }
91      
92      @Override
93      public Class<?> getErasedType() {
94          return Types.box(setter.getParameterTypes()[parameter]);
95      }
96  
97      @Nullable
98      @Override
99      public Annotation getQualifier() {
100         return annotations.getQualifier();
101     }
102 
103     @Nullable
104     @Override
105     public <A extends Annotation> A getAttribute(Class<A> atype) {
106         return annotations.getAttribute(atype);
107     }
108 
109     @Nonnull
110     @Override
111     public Collection<Annotation> getAttributes() {
112         return annotations.getAttributes();
113     }
114     
115     @Override
116     public boolean equals(Object o) {
117         if (!(o instanceof SetterInjectionPoint)) {
118             return false;
119         }
120         SetterInjectionPoint p = (SetterInjectionPoint) o;
121         return p.setter.equals(setter) && p.parameter == parameter;
122     }
123     
124     @Override
125     public int hashCode() {
126         return setter.hashCode() ^ (37 * 17 * parameter);
127     }
128     
129     @Override
130     public String toString() {
131         // method setFoo(..., @Qual Type argN, ...)
132         StringBuilder sb = new StringBuilder();
133         sb.append("method ")
134           .append(setter.getName())
135           .append("(");
136         if (parameter > 0) {
137             sb.append("..., ");
138         }
139         if (annotations.getQualifier() != null) {
140             sb.append(annotations.getQualifier())
141               .append(" ");
142         }
143         sb.append(setter.getParameterTypes()[parameter].getName())
144           .append(" arg")
145           .append(parameter);
146         if (parameter < setter.getParameterTypes().length) {
147             sb.append(", ...");
148         }
149         sb.append(")");
150 
151         return sb.toString();
152     }
153     
154     private Object writeReplace() {
155         return new SerialProxy(setter, parameter);
156     }
157 
158     private void readObject(ObjectInputStream stream) throws ObjectStreamException {
159         throw new InvalidObjectException("Serialization proxy required");
160     }
161 
162     private static class SerialProxy implements Serializable {
163         private static final long serialVersionUID = 1L;
164         private final MethodProxy method;
165         private final int parameterIndex;
166         public SerialProxy(Method m, int pidx) {
167             method = MethodProxy.of(m);
168             parameterIndex = pidx;
169         }
170 
171         private Object readResolve() throws ObjectStreamException {
172             try {
173                 return new SetterInjectionPoint(method.resolve(), parameterIndex);
174             } catch (ClassNotFoundException e) {
175                 InvalidObjectException ex =
176                         new InvalidObjectException("no class for " + method.toString());
177                 ex.initCause(e);
178                 throw ex;
179             } catch (NoSuchMethodException e) {
180                 InvalidObjectException ex =
181                         new InvalidObjectException("cannot resolve " + method.toString());
182                 ex.initCause(e);
183                 throw ex;
184             }
185         }
186     }
