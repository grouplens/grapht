34  /**
35   * Utilities related to Qualifier implementations.
36   * 
37   * @author <a href="http://grouplens.org">GroupLens Research</a>
38   */
39  public final class Qualifiers {
40      private Qualifiers() { }

42      /**
43       * Return true or false whether or not the annotation type represents a
44       * {@link Qualifier}
45       * 
46       * @param type The annotation type
47       * @return True if the annotation is a {@link Qualifier} or parameter
48       * @throws NullPointerException if the type is null
49       */
50      public static boolean isQualifier(Class<? extends Annotation> type) {
51          return type.getAnnotation(javax.inject.Qualifier.class) != null;
52      }
53  
54      /**
55       * The default qualifier matcher. This is currently the {@linkplain #matchAny() any matcher}.
56       * @return A QualifierMatcher that matches using the default policy.
57       */
58      public static QualifierMatcher matchDefault() {
59          return new DefaultMatcher();
60      }
61      
62      /**
63       * @return A QualifierMatcher that matches any qualifier
64       */
65      public static QualifierMatcher matchAny() {
66          return new AnyMatcher();
67      }
68      
69      /**
70       * @return A QualifierMatcher that matches only the null qualifier
71       */
72      public static QualifierMatcher matchNone() {
73          return new NullMatcher();
74      }
75      
76      /**
77       * @param annotType Annotation type class to match; {@code null} to match only the lack of a
78       *                  qualifier.
79       * @return A QualifierMatcher that matches any annotation of the given class
80       *         type.
81       */
82      public static QualifierMatcher match(Class<? extends Annotation> annotType) {
83          if (annotType == null) {
84              return matchNone();
85          } else {
86              return new AnnotationClassMatcher(annotType);
87          }
88      }
89      
90      /**
91       * @param annot Annotation instance to match, or {@code null} to match only the lack of a qualifier.
92       * @return A QualifierMatcher that matches annotations equaling annot
93       */
94      public static QualifierMatcher match(@Nonnull Annotation annot) {
95          if (annot == null) {
96              return matchNone();
97          } else {
98              return new AnnotationMatcher(annot);
99          }
100     }

102     private abstract static class AbstractMatcher implements QualifierMatcher {
103         private static final long serialVersionUID = 1L;
104 
105         @Override
106         @Deprecated
107         public boolean matches(Annotation q) {
108             return apply(q);
109         }
110 
111         @Override
112         public int compareTo(QualifierMatcher o) {
113             if (o == null) {
114                 // other type is unknown, so extend it to the front
115                 return 1;
116             } else {
117                 // lower priorities sort lower (higher precedence)
118                 return getPriority() - o.getPriority();
119             }
120         }
121     }

123     private static class DefaultMatcher extends AbstractMatcher {
124         private static final long serialVersionUID = 1L;
125 
126         @Override
127         public int getPriority() {
128             return 3;
129         }
130 
131         @Override
132         public boolean apply(Annotation q) {
133             if (q == null) {
134                 return true;
135             } else {
136                 Class<? extends Annotation> atype = q.annotationType();
137                 return atype.isAnnotationPresent(AllowUnqualifiedMatch.class);
138             }
139         }
140 
141         @Override
142         public boolean equals(Object o) {
143             return o instanceof DefaultMatcher;
144         }
145 
146         @Override
147         public int hashCode() {
148             return DefaultMatcher.class.hashCode();
149         }
150 
151         @Override
152         public String toString() {
153             return "%";
154         }
155     }

215     private static class AnnotationClassMatcher extends AbstractMatcher {
216         private static final long serialVersionUID = -1L;
217         private final Class<? extends Annotation> type;
218         
219         public AnnotationClassMatcher(Class<? extends Annotation> type) {
220             Preconditions.notNull("type", type);
221             Preconditions.isQualifier(type);
222             this.type = type;
223         }
224         
225         @Override
226         public int getPriority() {
227             return 1;
228         }
229 
230         @Override
231         public boolean apply(Annotation q) {
232             Class<? extends Annotation> qtype = (q == null ? null : q.annotationType());
233             return type.equals(qtype);
234         }
235         
236         @Override
237         public boolean equals(Object o) {
238             return o instanceof AnnotationClassMatcher
239                    && ((AnnotationClassMatcher) o).type.equals(type);
240         }
241         
242         @Override
243         public int hashCode() {
244             return type.hashCode();
245         }
246         
247         @Override
248         public String toString() {
249             return type.toString();
250         }
251 
252         private Object writeReplace() {
253             return new SerialProxy(type);
254         }
255 
256         private void readObject(ObjectInputStream stream) throws ObjectStreamException {
257             throw new InvalidObjectException("must use serialization proxy");
258         }
259 
260         private static class SerialProxy implements Serializable {
261             private static final long serialVersionUID = 1L;
262 
263             private final ClassProxy type;
264 
265             public SerialProxy(Class<?> cls) {
266                 type = ClassProxy.of(cls);
267             }
268 
269             private Object readResolve() throws ObjectStreamException {
270                 try {
271                     return new AnnotationClassMatcher(type.resolve().asSubclass(Annotation.class));
272                 } catch (ClassNotFoundException e) {
273                     InvalidObjectException ex = new InvalidObjectException("cannot resolve " + type);
274                     ex.initCause(e);
275                     throw ex;
276                 } catch (ClassCastException e) {
277                     InvalidObjectException ex =
278                             new InvalidObjectException("class " + type + " not an annotation");
279                     ex.initCause(e);
280                     throw ex;
281                 }
282             }
283         }
284     }

286     private static class AnnotationMatcher extends AbstractMatcher implements Serializable {
287         private static final long serialVersionUID = 1L;
288 
289         private final Annotation annotation;
290         
291         public AnnotationMatcher(Annotation annot) {
292             Preconditions.notNull("annotation", annot);
293             Preconditions.isQualifier(annot.annotationType());
294             annotation = annot;
295         }
296         
297         @Override
298         public int getPriority() {
299             return 0;
300         }
301 
302         @Override
303         public boolean apply(Annotation q) {
304             return annotation.equals(q);
305         }
306         
307         @Override
308         public boolean equals(Object o) {
309             return (o instanceof AnnotationMatcher)
310                    && ((AnnotationMatcher) o).annotation.equals(annotation);
311         }
312         
313         @Override
314         public int hashCode() {
315             return annotation.hashCode();
316         }
317         
318         @Override
319         public String toString() {

321         }
