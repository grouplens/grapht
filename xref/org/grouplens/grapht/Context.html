19  package org.grouplens.grapht;
20  
21  import org.grouplens.grapht.context.ContextPattern;
22  
23  import javax.annotation.Nullable;
24  import javax.inject.Qualifier;
25  import java.lang.annotation.Annotation;
26  
27  /**
28   * <p>
29   * Context is the main entry point for configuring bind rules using the fluent
30   * API. The dependency injector uses the contexts to limit the scope of a
31   * binding. Every time a dependency is satisfied, that type (and possibly {@link Qualifier})
32   * is pushed onto the context stack. Thus, if two different types each require a
33   * Foo, there can be two different bindings activated depending on which first
34   * type is in the context stack.
35   * <p>
36   * The "root" context is an empty stack, and will always be matched. When
37   * creating bindings, the context stack can be configured by calling
38   * {@link #in(Class)} or {@link #in(Class, Class)}.
39   * 
40   * @author <a href="http://grouplens.org">GroupLens Research</a>
41   */
42  public interface Context {
43      /**
44       * Start a new binding for the given type T within the scope of this
45       * context.  The returned Binding instance can be configured and completed by
46       * invoking one of its various to() methods.  Unless further configuration is
47       * done, this binding will match unqualified dependencies and dependencies
48       * with a qualifier annotated with
49       * {@link org.grouplens.grapht.annotation.AllowUnqualifiedMatch}.
50       * 
51       * @param <T> The matched source type
52       * @param type The raw class that is matched
53       * @return A new binding in this context for type T
54       */
55      <T> Binding<T> bind(Class<T> type);
56  
57      /**
58       * Start a new binding for a qualified type. A shortcut for
59       * {@code bind(type).withQualifier(qual)}.
60       * @param qual The type's qualifier.
61       * @param type The type to bind.
62       * @param <T> The type to bind.
63       * @return A new binding in this context for T with qualifier qual.
64       * @see Binding#withQualifier(Class)
65       */
66      <T> Binding<T> bind(Class<? extends Annotation> qual, Class<T> type);
67  
68      /**
69       * Start a new binding for a type irrespective of qualifier.  This is a
70       * shortcut for {@code bind(type).withAnyQualifier()}.
71       * @param type The type.
72       * @return A new binding in this context for type T with any (or no) 
73       *         qualifier.
74       */
75      <T> Binding<T> bindAny(Class<T> type);
76  
77      /**
78       * @deprecated Use {@link #within(Class)}.
79       */
80      @Deprecated
81      Context in(Class<?> type);
82  
83      /**
84       * @deprecated Use {@link #within(Class, Class)}.
85       */
86      @Deprecated
87      Context in(@Nullable Class<? extends Annotation> qualifier, Class<?> type);
88  
89      /**
90       * @deprecated Use {@link #within(Annotation, Class)}.
91       */
92      @Deprecated
93      Context in(@Nullable Annotation qualifier, Class<?> type);
94  
95      /**
96       * Create a new Context that extends the current context stack with the
97       * given class type. This matches with the default {@link Qualifier}. This is equivalent
98       * to <code>within(null, type);</code>
99       *
100      * @param type The type to extend this context by
101      * @return A new Context with a longer context stack
102      */
103     Context within(Class<?> type);
104 
105     /**
106      * Create a new Context that extends the current context stack with the
107      * given class and {@link Qualifier} annotation. If the qualifier is null,
108      * the default or null qualifier is used.
109      *
110      * @param qualifier The qualifier type that must be matched along with the type
111      * @param type The type to extend this context by
112      * @return A new Context with a longer context stack
113      */
114     Context within(@Nullable Class<? extends Annotation> qualifier, Class<?> type);
115 
116     /**
117      * Create a new Context that extends the current context stack with the
118      * given class, qualified by the specific Annotation instance. If the
119      * qualifier is null, the default or null qualifier is used.
120      *
121      * <p>The annotation provided must be serializable.  Annotations built by {@link
122      * org.grouplens.grapht.annotation.AnnotationBuilder} (recommended) or retrieved from the Java
123      * reflection API are serializable; if you use some other annotation implementation, it must be
124      * serializable.
125      *
126      * @param qualifier The qualifier instance that must be matched along with
127      *            the type
128      * @param type The type to extend this context by
129      * @return A new Context with a longer context stack
130      */
131     Context within(@Nullable Annotation qualifier, Class<?> type);
132 
133     /**
134      * Restruct context to matching a particular pattern.  The pattern is appended to the pattern
135      * generated by the other context-restriction methods.  This is a high-powered method that
136      * won't be needed in most situations, but allows fine-grained control of context matching.
137      *
138      * @param pattern The context pattern to match.
139      * @return A context with longer context.
140      */
141     Context matching(ContextPattern pattern);
142 
143     /**
144      * Create a new Context that extends the current context stack with the given class type as an
145      * anchored match. This matches with the default {@link Qualifier}. This is equivalent to
146      * <code>at(null, type);</code>
147      *
148      * @param type The type to extend this context by
149      * @return A new Context with a longer context stack
150      * @see #at(Class, Class)
151      */
152     Context at(Class<?> type);
153 
154     /**
155      * Create a new Context that extends the current context stack with the given class and {@link
156      * Qualifier} annotation as an anchored match. If the qualifier is null, the default or null
157      * qualifier is used.
158      * <p>
159      * Unlike {@link #in(Class,Class)}, this match is <em>anchored</em> &mdash; that is, it only
160      * matches at the end of a context chain.  Context is matched if it ends with this or,
161      * if further context is opened inside this context, if the inner context matches immediately.
162      * </p>
163      *
164      * @param qualifier The qualifier type that must be matched along with the type
165      * @param type      The type to extend this context by
166      * @return A new Context with a longer context stack
167      */
168     Context at(@Nullable Class<? extends Annotation> qualifier, Class<?> type);
169 
170     /**
171      * Create a new Context that extends the current context stack with the given class, qualified
172      * by the specific Annotation instance. as an anchored match. If the qualifier is null, the
173      * default or null qualifier is used.
174      *
175      * <p>The annotation provided must be serializable.  Annotations built by {@link
176      * org.grouplens.grapht.annotation.AnnotationBuilder} (recommended) or retrieved from the Java
177      * reflection API are serializable; if you use some other annotation implementation, it must be
178      * serializable.
179      *
180      * @param qualifier The qualifier instance that must be matched along with the type
181      * @param type      The type to extend this context by
182      * @return A new Context with a longer context stack
183      */
184     Context at(@Nullable Annotation qualifier, Class<?> type);
185 }
