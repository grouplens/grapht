19  package org.grouplens.grapht.util;
20  
21  import org.apache.commons.lang3.builder.HashCodeBuilder;
22  
23  import javax.annotation.Nullable;
24  import java.io.Serializable;
25  import java.lang.reflect.Constructor;
26  import java.util.Arrays;
27  
28  /**
29   * Proxy class for serializing constructors
30   */
31  public final class ConstructorProxy implements Serializable {
32      private static final long serialVersionUID = 1L;
33  
34      private final ClassProxy declaringClass;
35      private final ClassProxy[] parameterTypes;
36      @Nullable
37      private transient volatile Constructor constructor;
38      private transient volatile int hash;
39      @Nullable
40      private transient volatile String stringRepr;
41  
42      @SuppressWarnings("PMD.ArrayIsStoredDirectly")
43      private ConstructorProxy(ClassProxy cls, ClassProxy[] ptypes) {
44          declaringClass = cls;
45          parameterTypes = ptypes;
46      }
47  
48      @Override
49      public String toString() {
50          String repr = stringRepr;
51          if (repr == null) {
52              StringBuilder bld = new StringBuilder();
53              bld.append("proxy of ")
54                 .append(declaringClass.getClassName())
55                 .append("(");
56              boolean first = true;
57              for (ClassProxy p: parameterTypes) {
58                  if (!first) {
59                      bld.append(", ");
60                  }
61                  bld.append(p.getClassName());
62                  first = false;
63              }
64              bld.append(")");
65              stringRepr = repr = bld.toString();
66          }
67          return repr;
68      }
69  
70      @Override
71      public boolean equals(Object o) {
72          if (o == this) {
73              return true;
74          } else if (o instanceof ConstructorProxy) {
75              ConstructorProxy op = (ConstructorProxy) o;
76              return declaringClass.equals(op.declaringClass)
77                      && Arrays.deepEquals(parameterTypes, op.parameterTypes);
78          } else {
79              return true;
80          }
81      }
82  
83      @Override
84      public int hashCode() {
85          if (hash == 0) {
86              HashCodeBuilder hcb = new HashCodeBuilder();
87              hash = hcb.append(declaringClass)
88                        .append(parameterTypes)
89                        .hashCode();
90          }
91          return hash;
92      }
93  
94      /**
95       * Resolve this proxy into a {@link java.lang.reflect.Constructor} instance.
96       * @return The {@link java.lang.reflect.Constructor} represented by this proxy.
97       * @throws ClassNotFoundException If the proxy's declaring type cannot be resolved.
98       * @throws NoSuchMethodException If the constructor does not exist on the declaring type.
99       */
100     public Constructor resolve() throws ClassNotFoundException, NoSuchMethodException {
101         Constructor ctor = constructor;
102         if (ctor == null) {
103             Class<?> cls = declaringClass.resolve();
104             Class<?>[] ptypes = new Class<?>[parameterTypes.length];
105             for (int i = ptypes.length - 1; i >= 0; i--) {
106                 ptypes[i] = parameterTypes[i].resolve();
107             }
108             constructor = ctor = cls.getDeclaredConstructor(ptypes);
109         }
110         return ctor;
111     }
112 
113     /**
114      * Construct a proxy for a constructor.
115      * @param constructor The constructor to proxy.
116      * @return The constructor proxy representing {@code constructor}.
117      */
118     public static ConstructorProxy of(Constructor constructor) {
119         Class<?>[] ptypes = constructor.getParameterTypes();
120         ClassProxy[] proxies = new ClassProxy[ptypes.length];
121         for (int i = ptypes.length - 1; i >= 0; i--) {
122             proxies[i] = ClassProxy.of(ptypes[i]);
123         }
124         ConstructorProxy proxy =
125                 new ConstructorProxy(ClassProxy.of(constructor.getDeclaringClass()),
126                                      proxies);
127         proxy.constructor = constructor;
128         return proxy;
129     }
