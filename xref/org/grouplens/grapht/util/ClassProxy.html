40  /**
41   * A serialization proxy for class instances.  This serializable class encapsulates a simple
42   * representation for classes when serializing object graphs.
43   * <p>
44   *     When using this class, classes are serialized as their binary name, as returned by
45   *     {@link Class#getName()}.  The name encodes array information, so this is adequate
46   *     to fully reconstruct the class.
47   * </p>
48   *
49   * @author <a href="http://grouplens.org">GroupLens Research</a>
50   */
51  @Immutable
52  public final class ClassProxy implements Serializable {
53      private static final long serialVersionUID = 1;
54      private static final Logger logger = LoggerFactory.getLogger(ClassProxy.class);
55  
56      private final String className;
57      private final long checksum;
58      @Nullable
59      private transient volatile WeakReference<Class<?>> theClass;
60      private transient ClassLoader classLoader;
61  
62      private ClassProxy(String name, long check) {
63          className = name;
64          checksum = check;
65          classLoader = ClassLoaders.inferDefault(ClassProxy.class);
66      }
67  
68      private void readObject(ObjectInputStream stream) throws IOException, ClassNotFoundException {
69          stream.defaultReadObject();
70          classLoader = ClassLoaders.inferDefault(ClassProxy.class);
71      }

73      /**
74       * Get the class name. This name does not include any array information.
75       * @return The class name.
76       */
77      public String getClassName() {
78          return className;
79      }
80  
81      @Override
82      public String toString() {
83          return "proxy of " + className;

88          if (o == this) {
89              return true;
90          } else if (o instanceof ClassProxy) {
91              ClassProxy op = (ClassProxy) o;
92              return className.equals(op.className);
93          } else {
94              return false;
95          }
96      }

98      @Override

101     }

103     /**
104      * Resolve a class proxy to a class.
105      * @return The class represented by this proxy.
106      * @throws ClassNotFoundException if the class represented by this proxy cannot be found.
107      */
108     public Class<?> resolve() throws ClassNotFoundException {
109         WeakReference<Class<?>> ref = theClass;
110         Class<?> cls = ref == null ? null : ref.get();
111         if (cls == null) {
112             if(className.equals("void")) {
113                 // special case
114                 cls = Void.TYPE;
115             } else {
116                 cls = ClassUtils.getClass(classLoader, className);
117             }
118             long check = checksumClass(cls);
119             if (!isSerializationPermissive() && checksum != check) {
120                 throw new ClassNotFoundException("checksum mismatch for " + cls.getName());
121             } else {
122                 if (checksum != check) {
123                     logger.warn("checksum mismatch for {}", cls);
124                 }
125                 theClass = new WeakReference<Class<?>>(cls);
126             }
127         }
128         return cls;
129     }

131     private static final Map<Class<?>, ClassProxy> proxyCache = new WeakHashMap<Class<?>, ClassProxy>();
132 
133     /**
134      * Construct a class proxy for a class.
135      *
136      * @param cls The class.
137      * @return The class proxy.
138      */
139     public static synchronized ClassProxy of(Class<?> cls) {
140         ClassProxy proxy = proxyCache.get(cls);
141         if (proxy == null) {
142             proxy = new ClassProxy(cls.getName(), checksumClass(cls));
143             proxy.theClass = new WeakReference<Class<?>>(cls);
144             proxyCache.put(cls, proxy);
145         }
146         return proxy;
147     }

155     /**
156      * Compute a checksum for a class. These checksums are used to see if a class has changed
157      * its definition since being serialized.
158      * <p>
159      * The checksum used here is not cryptographically strong. It is intended only as a sanity
160      * check to detect incompatible serialization, not to robustly prevent tampering. The
161      * checksum algorithm currently is to compute an MD5 checksum over class member signatures
162      * and XOR the lower and upper halves of the checksum.
163      * </p>
164      *
165      * @param type The class to checksum.
166      * @return The
167      */
168     private static long checksumClass(Class<?> type) {
169         MessageDigest digest;
170         try {
171             digest = MessageDigest.getInstance("MD5");
172         } catch (NoSuchAlgorithmException e) {
173             throw new RuntimeException("JVM does not support MD5");
174         }
175         checksumClass(type, digest);
176 
177         ByteBuffer buf = ByteBuffer.wrap(digest.digest());
178         return buf.getLong() ^ buf.getLong();
179     }
180 
181     private static void checksumClass(Class<?> type, MessageDigest digest) {
182         // we compute a big hash of all the members of the class, and its superclasses.
183 
184         List<String> members = new ArrayList<String>();
185         for (Constructor<?> c: type.getDeclaredConstructors()) {
186             if (isInjectionSensitive(c)) {
187                 members.add(String.format("%s(%s)", c.getName(),
188                                           StringUtils.join(c.getParameterTypes(), ", ")));
189             }
190         }
191         for (Method m: type.getDeclaredMethods()) {
192             if (isInjectionSensitive(m)) {
193                 members.add(String.format("%s(%s): %s", m.getName(),
194                                           StringUtils.join(m.getParameterTypes(), ", "),
195                                           m.getReturnType()));
196             }
197         }
198         for (Field f: type.getDeclaredFields()) {
199             if (isInjectionSensitive(f)) {
200                 members.add(f.getName() + ":" + f.getType().getName());
201             }
202         }
203 
204         Collections.sort(members);
205 
206         Class<?> sup = type.getSuperclass();
207         if (sup != null) {
208             checksumClass(sup, digest);
209         }
210         for (String mem: members) {
211             digest.update(mem.getBytes(UTF8));
212         }
213     }

215     /**
216      * Check whether a member is injection-sensitive and should be checked for validity in
217      * deserialization.
218      *
219      * @param m The member.
220      * @param <M> The type of member (done so we can check multiple types).
221      * @return {@code true} if the member should be checksummed, {@code false} to ignore it.
222      */
223     private static <M extends Member & AnnotatedElement>boolean isInjectionSensitive(M m) {
224         // static methods are not sensitive
225         if (Modifier.isStatic(m.getModifiers())) {
226             return false;
227         }
228 
229         // private members w/o @Inject are not sensitive
230         if (Modifier.isPrivate(m.getModifiers()) && m.getAnnotation(Inject.class) == null) {
231             return false;
232         }
233 
234         // public, protected, or @Inject - it's sensitive (be conservative)
235         return true;
236     }
