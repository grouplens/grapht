59      /**
60       * Return the boxed version of the given type if the type is primitive.
61       * Otherwise, if the type is not a primitive the original type is returned.
62       * As an example, int.class is converted to Integer.class, but List.class is
63       * unchanged. This version of box preserves generics.
64       * 
65       * @param type The possibly unboxed type
66       * @return The boxed type
67       */
68      public static Type box(Type type) {
69          if (type instanceof Class) {
70              return box((Class<?>) type);
71          } else {
72              return type;
73          }
74      }

76      /**
77       * Return the boxed version of the given type if the type is primitive.
78       * Otherwise, if the type is not primitive the original class is returned.
79       * 
80       * @param type The possibly unboxed type
81       * @return The boxed type
82       */
83      public static Class<?> box(Class<?> type) {
84          if (int.class.equals(type)) {
85              return Integer.class;
86          } else if (short.class.equals(type)) {
87              return Short.class;
88          } else if (byte.class.equals(type)) {
89              return Byte.class;
90          } else if (long.class.equals(type)) {
91              return Long.class;
92          } else if (boolean.class.equals(type)) {
93              return Boolean.class;
94          } else if (char.class.equals(type)) {
95              return Character.class;
96          } else if (float.class.equals(type)) {
97              return Float.class;
98          } else if (double.class.equals(type)) {
99              return Double.class;
100         } else {
101             return type;
102         }
103     }

105     /**
106      * Compute the erasure of a type.
107      * 
108      * @param type The type to erase.
109      * @return The class representing the erasure of the type.
110      * @throws IllegalArgumentException if <var>type</var> is unerasable (e.g.
111      *             it is a type variable or a wildcard).
112      */
113     public static Class<?> erase(Type type) {
114         if (type instanceof Class) {
115             return (Class<?>) type;
116         } else if (type instanceof ParameterizedType) {
117             ParameterizedType pt = (ParameterizedType) type;
118             Type raw = pt.getRawType();
119             try {
120                 return (Class<?>) raw;
121             } catch (ClassCastException e) {
122                 throw new RuntimeException("raw type not a Class", e);
123             }
124         } else {
125             throw new IllegalArgumentException();
126         }
127     }

129     /**
130      * Return the type distance between the child and parent types. The child type
131      * must be a subtype of the parent. The type distance between a class and itself is 0;
132      * the distance from a class to one of its immediate supertypes (superclass or a directly
133      * implemented interface) is 1; deeper distances are computed recursively.
134      * 
135      * @param child The child type
136      * @param parent The parent type
137      * @return The type distance
138      * @throws IllegalArgumentException if {@code child} is not a subtype of {@code parent}.
139      */
140     public static int getTypeDistance(@Nonnull Class<?> child, @Nonnull Class<?> parent) {
141         Preconditions.notNull("child class", child);
142         Preconditions.notNull("parent class", parent);
143 
144         if (child.equals(parent)) {
145             // fast-path same-class tests
146             return 0;
147         } else if (!parent.isAssignableFrom(child)) {
148             // if child does not extend from the parent, return -1
149             throw new IllegalArgumentException("child not a subclass of parent");
150         } else if (!parent.isInterface()) {
151             // if the parent is not an interface, we only need to follower superclasses
152             int distance = 0;
153             Class<?> cur = child;
154             while (!cur.equals(parent)) {
155                 distance++;
156                 cur = cur.getSuperclass();
157             }
158             return distance;
159         } else {
160             // worst case, recursively compute the type
161             // recursion is safe, as types aren't too deep except in crazy-land
162             int minDepth = Integer.MAX_VALUE;
163             Class<?> sup = child.getSuperclass();
164             if (sup != null && parent.isAssignableFrom(sup)) {
165                 minDepth = getTypeDistance(sup, parent);
166             }
167             for (Class<?> iface: child.getInterfaces()) {
168                 if (parent.isAssignableFrom(iface)) {
169                     int d = getTypeDistance(iface, parent);
170                     if (d < minDepth) {
171                         minDepth = d;
172                     }
173                 }
174             }
175             // minDepth now holds the depth of the superclass with shallowest depth
176             return minDepth + 1;
177         }
178     }

180     /**
181      * Get the type that is provided by a given implementation of
182      * {@link Provider}.
183      * 
184      * @param providerClass The provider's class
185      * @return The provided class type
186      * @throws IllegalArgumentException if the class doesn't actually implement
187      *             Provider
188      */
189     public static Class<?> getProvidedType(Class<? extends Provider<?>> providerClass) {
190         com.google.common.base.Preconditions.checkArgument(Provider.class.isAssignableFrom(providerClass),
191                                                            "class is not Provider class");
192         Map<TypeVariable<?>, Type> bindings = TypeUtils.getTypeArguments(providerClass, Provider.class);
193         if(!bindings.containsKey(PROVIDER_TYPE_VAR)){
194             throw new IllegalArgumentException("Class provided by " + providerClass.getName() + " is generic");
195         }
196         final Class<?> inferredType = TypeUtils.getRawType(bindings.get(PROVIDER_TYPE_VAR), null);
197         try{
198             final Class<?> observedType = providerClass.getMethod("get").getReturnType();
199             if (inferredType != null && inferredType.isAssignableFrom(observedType)) {
200                 return observedType;
201             } else {
202                 return inferredType;
203             }
204         } catch (NoSuchMethodException e) {
205             throw new IllegalArgumentException("Class does not implement get()");
206         }
207     }

225     /**
226      * Return true if the type is not abstract and not an interface, and has
227      * a constructor annotated with {@link Inject} or its only constructor
228      * is the default constructor.
229      * 
230      * @param type A class type
231      * @return True if the class type is instantiable
232      */
233     public static boolean isInstantiable(Class<?> type) {
234         if (!Modifier.isAbstract(type.getModifiers()) && !type.isInterface()) {
235             // first check for a constructor annotated with @Inject, 
236             //  - this doesn't care how many we'll let the injector complain
237             //    if there are more than one
238             for (Constructor<?> c: type.getDeclaredConstructors()) {
239                 if (c.getAnnotation(Inject.class) != null) {
240                     return true;
241                 }
242             }
243             
244             // check if we only have the public default constructor
245             if (type.getConstructors().length == 1 
246                 && type.getConstructors()[0].getParameterTypes().length == 0) {
247                 return true;
248             }
249         }
250         
251         // no constructor available
252         return false;
253     }

255     /**
256      * <p>
257      * Return true if the type is not abstract and not an interface. This will
258      * return true essentially when the class "should" have a default
259      * constructor or a constructor annotated with {@link Inject @Inject} to be
260      * used properly.
261      * <p>
262      * As another special rule, if the input type is {@link Void}, false is
263      * returned because for most intents and purposes, it is not instantiable.
264      * 
265      * @param type The type to test
266      * @return True if it should be instantiable
267      */
268     public static boolean shouldBeInstantiable(Class<?> type) {
269         return !Modifier.isAbstract(type.getModifiers()) && !type.isInterface() && !Void.class.equals(type);
270     }

272     /**
273      * Return true if the array of Annotations contains an Annotation with a
274      * simple name of 'Nullable'. It does not matter which actual Nullable
275      * annotation is present.
276      * 
277      * @param annotations Array of annotations, e.g. from a setter or
278      *            constructor
279      * @return True if there exists a Nullable annotation in the array
280      */
281     public static boolean hasNullableAnnotation(Annotation[] annotations) {
282         for (Annotation a: annotations) {
283             if (a.annotationType().getSimpleName().equals("Nullable")) {
284                 return true;
285             }
286         }
287         return false;
288     }
