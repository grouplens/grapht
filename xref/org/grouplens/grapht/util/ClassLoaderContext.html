19  package org.grouplens.grapht.util;
20  
21  /**
22   * A class loader context, used to restore the state from {@link ClassLoaders#pushContext(ClassLoader)}.
23   *
24   * @since 0.8.1
25   * @author <a href="http://www.grouplens.org">GroupLens Research</a>
26   */
27  public class ClassLoaderContext {
28      private final Thread thread;
29      private final ClassLoader original;
30      private boolean popped = false;
31  
32      ClassLoaderContext(Thread th, ClassLoader orig) {
33          thread = th;
34          original = orig;
35      }
36  
37      /**
38       * Restore the original class loader before this context was entered.
39       * @throws IllegalStateException if the context was already popped.
40       */
41      public void pop() {
42          if (popped) {
43              throw new IllegalStateException("loader context already popped");
44          }
45          thread.setContextClassLoader(original);
46          popped = true;
47      }
48  }
